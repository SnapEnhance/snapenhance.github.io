+    data class PropertyDataProcessor<T>

+    val BOOLEAN = PropertyDataProcessor(

+        serialize = {

+        deserialize = { it.asBoolean },

+
+    val INTEGER = PropertyDataProcessor(

+        serialize = { JsonPrimitive(it) },
