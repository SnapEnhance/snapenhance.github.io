     30 fn init(mut env: JNIEnv, _class: JObject, signature_cache: JString) -> jstring {
     31     debug!("Initializing native lib");
     32 
     33     let start_time = std::time::Instant::now();
     34 
     35     // load signature cache
     36     
     37     if !signature_cache.is_null() {
     38         let sig_cache_str = get_jni_string(&mut env, signature_cache).expect("Failed to convert mappings to string");
     39         
     40         if let Ok(signature_cache) = serde_json::from_str(sig_cache_str.as_str()) {
     41             sig::add_signatures(signature_cache);
     42         } else {
     43             error!("Failed to load signature cache");
     44         }
     45     }
     46 
     47     common::set_native_lib_instance(env.new_global_ref(_class).ok().expect("Failed to create global ref"));
     48 
     49     let _ = common::CLIENT_MODULE;
     50 
     51     // initialize modules asynchronously
     52 
     53     let mut threads: Vec<JoinHandle<()>> = Vec::new();
     54 
     55     macro_rules! async_init {
     56         ($($f:expr),*) => {
     57             $(
     58                 threads.push(std::thread::spawn(move || {
     59                     $f;
     60                 }));
     61             )*
     62         };
     63     }
     64 
     65     async_init!(
     66         duplex_hook::init(),
     67         unary_call_hook::init(),
     68         composer_hook::init(),
     69         fstat_hook::init(),
     70         sqlite_hook::init(),
     71         custom_font_hook::init()
     72     );
     73     
     74     threads.into_iter().for_each(|t| t.join().unwrap());
     75 
     76     info!("native init took {:?}", start_time.elapsed());
     77 
     78     // send back the signature cache
     79     if let Ok(signature_cache) = serde_json::to_string(&sig::get_signatures()) {
     80         env.new_string(signature_cache).ok().expect("Failed to create new string").into_raw()
     81     } else {
     82         std::ptr::null_mut()
     83     }
     84 }
