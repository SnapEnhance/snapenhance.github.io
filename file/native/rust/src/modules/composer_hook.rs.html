    118 pub unsafe fn composer_eval(env: JNIEnv, _: *mut c_void, script: JString) -> jobject {
    119     #[cfg(not(target_arch = "aarch64"))]
    120     {
    121         return env.new_string("Architecture not supported").unwrap().into_raw();
    122     }
    123 
    124     let mut env = env;
    125 
    126     let script_str = get_jni_string(&mut env, script).expect("Failed to get script");
    127     let script_length = script_str.len();
    128 
    129     let js_value = JS_EVAL_ORIGINAL2.expect("No js eval found")(
    130         GLOBAL_INSTANCE.expect("No global instance found"), 
    131         GLOBAL_CTX.expect("No global context found"),
    132         std::ptr::null_mut(),
    133         (script_str + "\0").as_ptr() as *mut u8, 
    134         script_length, 
    135         "<eval>\0".as_ptr(), 
    136         0
    137     );
    138 
    139     let result: String =  if js_value.tag == JS_TAG_STRING {
    140         let string = js_value.u.ptr as *mut JsString;
    141         CStr::from_ptr((*string).str8.as_ptr() as *const u8).to_str().unwrap().into()
    142     } else if js_value.tag == JS_TAG_INT {
    143         js_value.u.int32.to_string()
    144     } else if js_value.tag == JS_TAG_BOOL {
    145         if js_value.u.int32 == 1 { "true" } else { "false" }.into()
    146     } else if js_value.tag == JS_TAG_NULL {
    147         "null".into()
    148     } else if js_value.tag == JS_TAG_UNDEFINED {
    149         "undefined".into()
    150     } else if js_value.tag == JS_TAG_OBJECT {
    151         "[object]".into()
    152     } else if js_value.tag == JS_TAG_FLOAT64 {
    153         js_value.u.float64.to_string()
    154     } else if js_value.tag == JS_TAG_EXCEPTION {
    155         "Failed to evaluate script".into()
    156     } else {
    157         "[unknown tag ".to_owned() + &js_value.tag.to_string() + "]".into()
    158     };
    159     
    160     env.new_string(result).unwrap().into_raw()
    161 }

    165         return
    166     }

    168     if let Some(signature) = sig::find_signature(

    181         warn!("Unable to find js_eval signature");
