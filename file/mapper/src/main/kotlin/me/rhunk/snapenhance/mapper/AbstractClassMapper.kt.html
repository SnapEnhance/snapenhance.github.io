     72     fun readFromJson(json: JsonObject) {
     73         values.forEach { (key, _) ->
     74             runCatching {
     75                 val jsonElement = json.get(key) ?: return@forEach
     76                 when (jsonElement) {
     77                     is JsonObject -> values[key] = gson.fromJson(jsonElement, HashMap::class.java)
     78                     else -> values[key] = jsonElement.asString
     79                 }
     80             }.onFailure {
     81                 Log.e("Mapper","Failed to deserialize property $key")
     82             }
     83         }

     86     fun writeFromJson(json: JsonObject) {
     87         values.forEach { (key, value) ->
     88             runCatching {
     89                 when (value) {
     90                     is String -> json.addProperty(key, value)
     91                     is Class<*> -> json.addProperty(key, value.name)
     92                     is Map<*, *> -> json.add(key, gson.toJsonTree(value))
     93                     else -> json.addProperty(key, value.toString())
     94                 }
     95             }.onFailure {
     96                 Log.e("Mapper","Failed to serialize property $key")
     97             }
     98         }
