     47 class FFMpegProcessor(
     48     private val logManager: LogManager,
     49     private val ffmpegOptions: DownloaderConfig.FFMpegOptions,
     50     private val onStatistics: (Statistics) -> Unit = {}

     52     companion object {

     55         fun newFFMpegProcessor(context: RemoteSideContext, pendingTask: PendingTask) = FFMpegProcessor(

     57             ffmpegOptions = context.config.root.downloader.ffmpegOptions,
     58             onStatistics = {
     59                 pendingTask.updateProgress("Processing (frames=${it.videoFrameNumber}, fps=${it.videoFps}, time=${it.time}, bitrate=${it.bitrate}, speed=${it.speed})")

     73         val inputs: List<String>,

     85     private suspend fun newFFMpegTask(globalArguments: ArgumentList, inputArguments: ArgumentList, outputArguments: ArgumentList) = suspendCancellableCoroutine<FFmpegSession> {
     86         val stringBuilder = StringBuilder()
     87         arrayOf(globalArguments, inputArguments, outputArguments).forEach { argumentList ->
     88             argumentList.forEach { (key, value) ->
     89                 stringBuilder.append("$key ${value.takeIf { it.isNotEmpty() }?.plus(" ") ?: ""}")
     90             }
     91         }
     92 
     93         logManager.debug("arguments: $stringBuilder", "FFMpegProcessor")
     94 
     95         FFmpegKit.executeAsync(stringBuilder.toString(),
     96             { session ->
     97                 it.resumeWith(
     98                     if (session.returnCode.isValueSuccess) {
     99                         Result.success(session)
    100                     } else {
    101                         Result.failure(Exception(session.output))
    102                     }
    103                 )
    104             }, logFunction@{ log ->
    105                 logManager.internalLog(TAG, when (log.level) {
    106                     Level.AV_LOG_ERROR, Level.AV_LOG_FATAL -> LogLevel.ERROR
    107                     Level.AV_LOG_WARNING -> LogLevel.WARN
    108                     Level.AV_LOG_VERBOSE -> LogLevel.VERBOSE
    109                     else -> return@logFunction
    110                 }, log.message)
    111             }, { onStatistics(it) }, Executors.newSingleThreadExecutor())
    112     }
    113 
    114     suspend fun execute(args: Request) {
    115         // load ffmpeg native sync to avoid native crash
    116         synchronized(this) { FFmpegKit.listSessions() }
    117         val globalArguments = ArgumentList().apply {
    118             this += "-hwaccel ${if (ffmpegOptions.customVideoCodec.get().isNotEmpty()) "auto" else "mediacodec"}" //Use hwaccel If Available
    119             this += "-y"
    120             this += "-threads" to ffmpegOptions.threads.get().toString()
    121         }
    122 
    123         val inputArguments = ArgumentList().apply {
    124             args.inputs.forEach { path ->
    125                 this += "-i" to path
    126             }
    127         }
    128 
    129         val outputArguments = ArgumentList().apply {
    130             this += "-preset" to (ffmpegOptions.preset.getNullable() ?: "ultrafast")
    131             this += "-c:v" to (ffmpegOptions.customVideoCodec.get().takeIf { it.isNotEmpty() } ?: "h264_mediacodec")
    132             this += "-c:a" to (ffmpegOptions.customAudioCodec.get().takeIf { it.isNotEmpty() } ?: "copy")
    133             this += "-crf" to ffmpegOptions.constantRateFactor.get().let { "\"$it\"" }
    134             this += "-b:v" to ffmpegOptions.videoBitrate.get().toString() + "K"
    135             this += "-b:a" to ffmpegOptions.audioBitrate.get().toString() + "K"
    136         }
    137 
    138         when (args.action) {
    139             Action.DOWNLOAD_DASH -> {
    140                 outputArguments += "-ss" to "'${args.startTime}ms'"
    141                 if (args.duration != null) {
    142                     outputArguments += "-t" to "'${args.duration}ms'"
    143                 }
    144             }
    145             Action.MERGE_OVERLAY -> {
    146                 inputArguments += "-i" to args.overlay!!.absolutePath
    147                 outputArguments += "-filter_complex" to "\"[0]scale2ref[img][vid];[img]setsar=1[img];[vid]nullsink;[img][1]overlay=(W-w)/2:(H-h)/2,scale=2*trunc(iw*sar/2):2*trunc(ih/2)\""
    148             }
    149             Action.CONVERSION -> {
    150                 if (ffmpegOptions.customAudioCodec.isEmpty()) {
    151                     outputArguments -= "-c:a"
    152                 }
    153                 outputArguments -= "-c:v"
    154                 args.videoCodec?.let {
    155                     outputArguments += "-c:v" to it
    156                 } ?: run {
    157                     outputArguments += "-vn"
    158                 }
    159                 args.audioCodec?.let {
    160                     outputArguments -= "-c:a"
    161                     outputArguments += "-c:a" to it
    162                 }
    163             }
    164             Action.MERGE_MEDIA -> {
    165                 inputArguments.clear()
    166                 val filesInfo = args.inputs.mapNotNull { file ->
    167                     runCatching {
    168                         MediaMetadataRetriever().apply { setDataSource(file) }
    169                     }.getOrNull()?.let { file to it }
    170                 }
    171 
    172                 val (maxWidth, maxHeight) = filesInfo.maxByOrNull { (_, r) ->
    173                     r.extractMetadata(MediaMetadataRetriever.METADATA_KEY_VIDEO_WIDTH)?.toIntOrNull() ?: 0
    174                 }?.let { (_, r) ->
    175                     r.extractMetadata(MediaMetadataRetriever.METADATA_KEY_VIDEO_WIDTH)?.toIntOrNull() to
    176                     r.extractMetadata(MediaMetadataRetriever.METADATA_KEY_VIDEO_HEIGHT)?.toIntOrNull()
    177                 } ?: throw Exception("Failed to get video size")
    178 
    179                 val filterFirstPart = StringBuilder()
    180                 val filterSecondPart = StringBuilder()
    181                 var containsNoSound = false
    182 
    183                 filesInfo.forEachIndexed { index, (file, retriever) ->
    184                     filterFirstPart.append("[$index:v]scale=$maxWidth:$maxHeight,setsar=1[v$index];")
    185                     if (retriever.extractMetadata(MediaMetadataRetriever.METADATA_KEY_HAS_AUDIO) == "yes") {
    186                         filterSecondPart.append("[v$index][$index:a]")
    187                     } else {
    188                         containsNoSound = true
    189                         filterSecondPart.append("[v$index][${filesInfo.size}]")
    190                     }
    191                     inputArguments += "-i" to file
    192                 }
    193 
    194                 if (containsNoSound) {
    195                     inputArguments += "-f" to "lavfi"
    196                     inputArguments += "-t" to "0.1"
    197                     inputArguments += "-i" to "anullsrc=channel_layout=stereo:sample_rate=44100"
    198                 }
    199 
    200                 if (outputArguments["-c:a"] == "copy") {
    201                     outputArguments -= "-c:a"
    202                 }
    203 
    204                 outputArguments += "-fps_mode" to "vfr"
    205 
    206                 outputArguments += "-filter_complex" to "\"$filterFirstPart ${filterSecondPart}concat=n=${args.inputs.size}:v=1:a=1[vout][aout]\""
    207                 outputArguments += "-map" to "\"[aout]\""
    208                 outputArguments += "-map" to "\"[vout]\""
    209 
    210                 filesInfo.forEach { it.second.close() }
    211             }
    212             Action.DOWNLOAD_AUDIO_STREAM -> {
    213                 outputArguments.clear()
    214                 globalArguments += "-f" to when (args.audioStreamFormat!!.encoding) {
    215                     AudioFormat.ENCODING_PCM_8BIT -> "u8"
    216                     AudioFormat.ENCODING_PCM_16BIT -> "s16le"
    217                     AudioFormat.ENCODING_PCM_FLOAT -> "f32le"
    218                     AudioFormat.ENCODING_PCM_32BIT -> "s32le"
    219                     else -> throw IllegalArgumentException("Unsupported audio encoding")
    220                 }
    221                 globalArguments += "-ar" to args.audioStreamFormat!!.sampleRate.toString()
    222                 globalArguments += "-ac" to args.audioStreamFormat!!.channels.toString()
    223             }
    224         }
    225         outputArguments += args.output.absolutePath
    226         newFFMpegTask(globalArguments, inputArguments, outputArguments)
    227     }
