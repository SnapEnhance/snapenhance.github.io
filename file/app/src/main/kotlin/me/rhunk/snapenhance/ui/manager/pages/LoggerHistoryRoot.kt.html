     59     private inline fun decodeMessage(message: LoggedMessage, result: (contentType: ContentType, messageReader: ProtoReader, attachments: List<DecodedAttachment>) -> Unit) {
     60         runCatching {
     61             val messageObject = JsonParser.parseString(String(message.messageData, Charsets.UTF_8)).asJsonObject
     62             val messageContent = messageObject.getAsJsonObject("mMessageContent")
     63             val messageReader = messageContent.getAsJsonArray("mContent").map { it.asByte }.toByteArray().let { ProtoReader(it) }
     64             result(ContentType.fromMessageContainer(messageReader) ?: ContentType.UNKNOWN, messageReader, MessageDecoder.decode(messageContent))
     65         }.onFailure {
     66             context.log.error("Failed to decode message", it)
     67         }
     68     }

     70     private fun downloadAttachment(creationTimestamp: Long, attachment: DecodedAttachment) {
     71         context.shortToast("Download started!")
     72         val attachmentHash = attachment.mediaUniqueId!!.longHashCode().absoluteValue.toString()
     73 
     74         DownloadProcessor(
     75             remoteSideContext = context,
     76             callback = object: DownloadCallback.Default() {
     77                 override fun onSuccess(outputPath: String?) {
     78                     context.shortToast("Downloaded to $outputPath")
     79                 }
     80 
     81                 override fun onFailure(message: String?, throwable: String?) {
     82                     context.shortToast("Failed to download $message")
     83                 }
     84             }
     85         ).enqueue(
     86             DownloadRequest(
     87                 inputMedias = arrayOf(attachment.createInputMedia()!!)
     88             ),
     89             DownloadMetadata(
     90                 mediaIdentifier = attachmentHash,
     91                 outputPath = createNewFilePath(
     92                     context.config.root,
     93                     attachment.mediaUniqueId!!,
     94                     MediaDownloadSource.MESSAGE_LOGGER,
     95                     attachmentHash,
     96                     creationTimestamp
     97                 ),
     98                 iconUrl = null,
     99                 mediaAuthor = null,
    100                 downloadSource = MediaDownloadSource.MESSAGE_LOGGER.translate(context.translation),
    101             )
    102         )
    103     }
