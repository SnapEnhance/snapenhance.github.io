     29 /**
     30  * A nested scroll modifier that provides scroll events to [state].
     31  *
     32  * Note that this modifier must be added above a scrolling container, such as a lazy column, in
     33  * order to receive scroll events. For example:
     34  *
     35  * @sample androidx.compose.material.samples.PullRefreshSample
     36  *
     37  * @param state The [PullRefreshState] associated with this pull-to-refresh component.
     38  * The state will be updated by this modifier.
     39  * @param enabled If not enabled, all scroll delta and fling velocity will be ignored.
     40  */
     41 // TODO(b/244423199): Move pullRefresh into its own material library similar to material-ripple.
     42 fun Modifier.pullRefresh(
     43     state: PullRefreshState,
     44     enabled: Boolean = true,
     45 ) = inspectable(
     46     inspectorInfo = debugInspectorInfo {
     47         name = "pullRefresh"
     48         properties["state"] = state
     49         properties["enabled"] = enabled
     50     },
     51 ) {
     52     Modifier.pullRefresh(state::onPull, state::onRelease, enabled)
     53 }

     55 /**
     56  * A nested scroll modifier that provides [onPull] and [onRelease] callbacks to aid building custom
     57  * pull refresh components.
     58  *
     59  * Note that this modifier must be added above a scrolling container, such as a lazy column, in
     60  * order to receive scroll events. For example:
     61  *
     62  * @sample androidx.compose.material.samples.CustomPullRefreshSample
     63  *
     64  * @param onPull Callback for dispatching vertical scroll delta, takes float pullDelta as argument.
     65  * Positive delta (pulling down) is dispatched only if the child does not consume it (i.e. pulling
     66  * down despite being at the top of a scrollable component), whereas negative delta (swiping up) is
     67  * dispatched first (in case it is needed to push the indicator back up), and then the unconsumed
     68  * delta is passed on to the child. The callback returns how much delta was consumed.
     69  * @param onRelease Callback for when drag is released, takes float flingVelocity as argument.
     70  * The callback returns how much velocity was consumed - in most cases this should only consume
     71  * velocity if pull refresh has been dragged already and the velocity is positive (the fling is
     72  * downwards), as an upwards fling should typically still scroll a scrollable component beneath the
     73  * pullRefresh. This is invoked before any remaining velocity is passed to the child.
     74  * @param enabled If not enabled, all scroll delta and fling velocity will be ignored and neither
     75  * [onPull] nor [onRelease] will be invoked.
     76  */
     77 fun Modifier.pullRefresh(
     78     onPull: (pullDelta: Float) -> Float,
     79     onRelease: suspend (flingVelocity: Float) -> Float,
     80     enabled: Boolean = true,
     81 ) = inspectable(
     82     inspectorInfo = debugInspectorInfo {
     83         name = "pullRefresh"
     84         properties["onPull"] = onPull
     85         properties["onRelease"] = onRelease
     86         properties["enabled"] = enabled
     87     },
     88 ) {
     89     Modifier.nestedScroll(PullRefreshNestedScrollConnection(onPull, onRelease, enabled))
     90 }
     91 
     92 private class PullRefreshNestedScrollConnection(
     93     private val onPull: (pullDelta: Float) -> Float,
     94     private val onRelease: suspend (flingVelocity: Float) -> Float,
     95     private val enabled: Boolean,
     96 ) : NestedScrollConnection {
     97 
     98     override fun onPreScroll(
     99         available: Offset,
    100         source: NestedScrollSource,
    101     ): Offset = when {
    102         !enabled -> Offset.Zero
    103         source == Drag && available.y < 0 -> Offset(0f, onPull(available.y)) // Swiping up
    104         else -> Offset.Zero
    105     }
    106 
    107     override fun onPostScroll(
    108         consumed: Offset,
    109         available: Offset,
    110         source: NestedScrollSource,
    111     ): Offset = when {
    112         !enabled -> Offset.Zero
    113         source == Drag && available.y > 0 -> Offset(0f, onPull(available.y)) // Pulling down
    114         else -> Offset.Zero
    115     }
    116 
    117     override suspend fun onPreFling(available: Velocity): Velocity {
    118         return Velocity(0f, onRelease(available.y))
    119     }
    120 }
