     16 class MappingsScreen : SetupScreen() {
     17     @Composable
     18     override fun Content() {
     19         val coroutineScope = rememberCoroutineScope()
     20         var infoText by remember { mutableStateOf(null as String?) }
     21         var isGenerating by remember { mutableStateOf(false) }
     22 
     23         if (infoText != null) {
     24             Dialog(onDismissRequest = {
     25                 infoText = null
     26             }) {
     27                 remember { AlertDialogs(context.translation) }.InfoDialog(title = infoText!!) {
     28                     infoText = null
     29                     goNext()
     30                 }
     31             }
     32         }
     33 
     34         LaunchedEffect(Unit) {
     35             coroutineScope.launch(Dispatchers.IO) {
     36                 if (isGenerating) return@launch
     37                 isGenerating = true
     38                 runCatching {
     39                     if (context.installationSummary.snapchatInfo == null) {
     40                         throw Exception(context.translation["setup.mappings.generate_failure_no_snapchat"])
     41                     }
     42                     context.mappings.refresh()
     43                     goNext()
     44                 }.onFailure {
     45                     isGenerating = false
     46                     infoText = context.translation["setup.mappings.generate_failure"] + "\n\n" + it.message
     47                     context.log.error("Failed to generate mappings", it)
     48                 }
     49             }
     50         }
     51 
     52         if (isGenerating) {
     53             DialogText(text = context.translation["setup.mappings.dialog"])
     54             CircularProgressIndicator(
     55                 modifier = Modifier
     56                     .padding()
     57                     .size(50.dp),
     58                 strokeWidth = 3.dp,
     59                 color = MaterialTheme.colorScheme.onPrimary
     60             )
     61         }
     62     }
