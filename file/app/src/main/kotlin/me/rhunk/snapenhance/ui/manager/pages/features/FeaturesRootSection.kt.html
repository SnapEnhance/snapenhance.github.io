    679             content = { innerPadding ->

    696     override val floatingActionButton: @Composable () -> Unit = {
    697         fun saveConfig() {
    698             context.coroutineScope.launch(Dispatchers.IO) {
    699                 context.config.writeConfig()
    700                 context.log.verbose("saved config!")
    701             }
    702         }
    703 
    704         OnLifecycleEvent { _, event ->
    705             if (event == Lifecycle.Event.ON_PAUSE || event == Lifecycle.Event.ON_STOP) {
    706                 saveConfig()
    707             }
    708         }
    709 
    710         DisposableEffect(Unit) {
    711             onDispose {
    712                 saveConfig()
    713             }
    714         }
    715     }
