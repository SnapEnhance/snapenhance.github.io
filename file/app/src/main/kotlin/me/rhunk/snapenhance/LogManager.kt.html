     19 class LogLine(
     20     val logLevel: LogLevel,
     21     val dateTime: String,
     22     val tag: String,
     23     val message: String
     24 ) {
     25     companion object {
     26         fun fromString(line: String) = runCatching {
     27             val parts = line.trimEnd().split("/")
     28             if (parts.size != 4) return@runCatching null
     29             LogLine(
     30                 LogLevel.fromLetter(parts[0]) ?: return@runCatching null,
     31                 parts[1],
     32                 parts[2],
     33                 parts[3]
     34             )
     35         }.getOrNull()
     36     }
     37 
     38     override fun toString(): String {
     39         return "${logLevel.letter}/$dateTime/$tag/$message"
     40     }

     44 class LogReader(
     45     logFile: File
     46 ) {
     47     private val randomAccessFile = RandomAccessFile(logFile, "r")
     48     private var startLineIndexes = mutableListOf<Long>()
     49     var lineCount = queryLineCount()
     50 
     51     private fun readLogLine(): LogLine? {
     52         val lines = StringBuilder()
     53         val lastPointer = randomAccessFile.filePointer
     54         var lastChar: Int = -1
     55         var bufferLength = 0
     56         while (true) {
     57             val char = randomAccessFile.read()
     58             if (char == -1) {
     59                 randomAccessFile.seek(lastPointer)
     60                 return null
     61             }
     62             if ((char == '|'.code && lastChar == '\n'.code) || bufferLength > 4096) {
     63                 break
     64             }
     65             lines.append(char.toChar())
     66             bufferLength++
     67             lastChar = char
     68         }
     69 
     70         return LogLine.fromString(lines.trimEnd().toString())
     71             ?: LogLine(LogLevel.ERROR, "1970-01-01 00:00:00", "LogReader", "Failed to parse log line: $lines")
     72     }
     73 
     74     fun incrementLineCount() {
     75         synchronized(randomAccessFile) {
     76             randomAccessFile.seek(randomAccessFile.length())
     77             startLineIndexes.add(randomAccessFile.filePointer + 1)
     78             lineCount++
     79         }
     80     }
     81 
     82     private fun queryLineCount(): Int {
     83         synchronized(randomAccessFile) {
     84             randomAccessFile.seek(0)
     85             var lineCount = 0
     86             var lastPointer: Long
     87             var line: String?
     88 
     89             while (randomAccessFile.also {
     90                     lastPointer = it.filePointer
     91                 }.readLine().also { line = it } != null) {
     92                 if (line?.startsWith('|') == true) {
     93                     lineCount++
     94                     startLineIndexes.add(lastPointer + 1)
     95                 }
     96             }
     97 
     98             return lineCount
     99         }
    100     }
    101 
    102     private fun getLine(index: Int): String? {
    103         if (index <= 0 || index > lineCount) return null
    104         synchronized(randomAccessFile) {
    105             randomAccessFile.seek(startLineIndexes[index])
    106             return readLogLine()?.toString()
    107         }
    108     }
    109 
    110     fun getLogLine(index: Int): LogLine? {
    111         return getLine(index)?.let { LogLine.fromString(it) }
    112     }
    113 }

    116 class LogManager(
    117     private val remoteSideContext: RemoteSideContext
    118 ): AbstractLogger(LogChannel.MANAGER) {
    119     companion object {
    120         private val LOG_LIFETIME = 24.hours
    121     }
    122 
    123     private val anonymizeLogs by lazy { !remoteSideContext.config.root.scripting.disableLogAnonymization.get() }
    124 
    125     var lineAddListener = { _: LogLine -> }
    126 
    127     private val logFolder = File(remoteSideContext.androidContext.cacheDir, "logs")
    128     private var logFile: File? = null
    129 
    130     private val uuidRegex by lazy { Regex("[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}", RegexOption.MULTILINE) }
    131     private val contentUriRegex by lazy { Regex("content://[a-zA-Z0-9_\\-./]+") }
    132     private val filePathRegex by lazy { Regex("([a-zA-Z0-9_\\-./]+)\\.(${FileType.entries.joinToString("|") { file -> file.fileExtension.toString() }})") }
    133 
    134     fun init() {
    135         if (!logFolder.exists()) {
    136             logFolder.mkdirs()
    137         }
    138         logFile = remoteSideContext.sharedPreferences.getString("log_file", null)?.let { File(it) }?.takeIf { it.exists() } ?: run {
    139             newLogFile()
    140             logFile
    141         }
    142 
    143         if (System.currentTimeMillis() - remoteSideContext.sharedPreferences.getLong("last_created", 0) > LOG_LIFETIME.inWholeMilliseconds) {
    144             newLogFile()
    145         }
    146     }
    147 
    148     fun internalLog(tag: String, logLevel: LogLevel, message: Any?) {
    149         runCatching {
    150             val anonymizedMessage = message.toString().let {
    151                 if (remoteSideContext.config.isInitialized() && anonymizeLogs)
    152                     it.replace(uuidRegex, "xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx")
    153                         .replace(contentUriRegex, "content://xxx")
    154                         .replace(filePathRegex, "xxxxxxxx.$2")
    155                 else it
    156             }
    157             val line = LogLine(
    158                 logLevel = logLevel,
    159                 dateTime = getCurrentDateTime(),
    160                 tag = tag,
    161                 message = anonymizedMessage
    162             )
    163             remoteSideContext.coroutineScope.launch(Dispatchers.IO) {
    164                 logFile?.appendText("|$line\n", Charsets.UTF_8)
    165             }
    166             lineAddListener(line)
    167             Log.println(logLevel.priority, tag, anonymizedMessage)
    168         }.onFailure {
    169             Log.println(Log.ERROR, tag, "Failed to log message: $message")
    170             Log.println(Log.ERROR, tag, it.stackTraceToString())
    171         }
    172     }
    173 
    174     private fun getCurrentDateTime(pathSafe: Boolean = false): String {
    175         return DateTimeFormatter.ofPattern(if (pathSafe) "yyyy-MM-dd_HH-mm-ss" else "yyyy-MM-dd HH:mm:ss").format(
    176             java.time.LocalDateTime.now()
    177         )
    178     }
    179 
    180     private fun newLogFile() {
    181         val currentTime = System.currentTimeMillis()
    182         logFile = File(logFolder, "snapenhance_${getCurrentDateTime(pathSafe = true)}.log").also {
    183             it.createNewFile()
    184             remoteSideContext.sharedPreferences.edit().putString("log_file", it.absolutePath).putLong("last_created", currentTime).apply()
    185         }
    186     }
    187 
    188     fun clearLogs() {
    189         logFolder.listFiles()?.forEach { it.delete() }
    190         newLogFile()
    191     }

    193     fun exportLogsToZip(outputStream: OutputStream) {
    194         val zipOutputStream = ZipOutputStream(outputStream).apply {
    195             setMethod(ZipOutputStream.DEFLATED)
    196         }
    197 
    198         // add device info to zip
    199         zipOutputStream.putNextEntry(ZipEntry("device_info.json"))
    200         val gson = GsonBuilder().setPrettyPrinting().create()
    201         zipOutputStream.write(gson.toJson(remoteSideContext.installationSummary).toByteArray())
    202         zipOutputStream.closeEntry()
    203 
    204         // add config
    205         zipOutputStream.putNextEntry(ZipEntry("config.json"))
    206         zipOutputStream.write(remoteSideContext.config.exportToString(exportSensitiveData = false).toByteArray())
    207         zipOutputStream.closeEntry()
    208 
    209         //add logFolder to zip
    210         logFolder.walk().forEach {
    211             if (it.isFile) {
    212                 zipOutputStream.putNextEntry(ZipEntry(it.name))
    213                 it.inputStream().copyTo(zipOutputStream)
    214                 zipOutputStream.closeEntry()
    215             }
    216         }
    217 
    218         zipOutputStream.close()
    219     }
