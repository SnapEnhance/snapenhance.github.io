    103     fun reload() {
    104         runCatching {
    105             runBlocking(Dispatchers.IO) {
    106                 log.init()
    107                 log.verbose("Loading RemoteSideContext")
    108                 config.load()
    109                 launch {
    110                     mappings.apply {
    111                         init(androidContext)
    112                     }
    113                 }
    114                 translation.apply {
    115                     userLocale = config.locale
    116                     load()
    117                 }
    118                 database.init()
    119                 streaksReminder.init()
    120                 scriptManager.init()
    121                 launch {
    122                     taskManager.init()
    123                     config.root.messaging.messageLogger.takeIf {
    124                         it.globalState == true
    125                     }?.autoPurge?.let { getPurgeTime(it.getNullable()) }?.let {
    126                         messageLogger.purgeAll(it)
    127                     }
    128 
    129                     config.root.friendTracker.takeIf {
    130                         it.globalState == true
    131                     }?.autoPurge?.let { getPurgeTime(it.getNullable()) }?.let {
    132                         messageLogger.purgeTrackerLogs(it)
    133                     }
    134                 }
    135                 coroutineScope.launch {
    136                     remoteSharedLibraryManager.init()
    137                 }
    138             }
    139         }.onFailure {
    140             log.error("Failed to load RemoteSideContext", it)
    141             androidContext.fatalCrash(it)
    142         }
    143 
    144         scriptManager.runtime.eachModule {
    145             callFunction("module.onSnapEnhanceLoad", androidContext)
    146         }
    147     }

    200     fun checkForRequirements(overrideRequirements: Int? = null): Boolean {
    201         var requirements = overrideRequirements ?: 0
    202         if (!config.wasPresent) {
    203             requirements = requirements or Requirements.FIRST_RUN
    204         }
    205 
    206         config.root.downloader.saveFolder.get().let {
    207             if (it.isEmpty() || run {
    208                     val documentFile = runCatching { DocumentFile.fromTreeUri(androidContext, Uri.parse(it)) }.getOrNull()
    209                     documentFile == null || !documentFile.exists() || !documentFile.canWrite()
    210                 }) {
    211                 requirements = requirements or Requirements.SAVE_FOLDER
    212             }
    213         }
    214 
    215         if (!sharedPreferences.getBoolean("debug_disable_mapper", false) && mappings.getSnapchatPackageInfo() != null && mappings.isMappingsOutdated()) {
    216             requirements = requirements or Requirements.MAPPINGS
    217         }
    218 
    219         /*if (sharedPreferences.getString("sif", null) == null) {
    220             requirements = requirements or Requirements.SIF
    221         }*/
    222 
    223         if (requirements == 0) return false
    224 
    225         val currentContext = activity ?: androidContext
    226 
    227         Intent(currentContext, SetupActivity::class.java).apply {
    228             putExtra("requirements", requirements)
    229             if (currentContext !is Activity) {
    230                 addFlags(Intent.FLAG_ACTIVITY_NEW_TASK)
    231             }
    232             currentContext.startActivity(this)
    233             return true
    234         }
    235     }

    237     fun launchActionIntent(action: EnumAction) {
    238         val intent = androidContext.packageManager.getLaunchIntentForPackage(
    239             Constants.SNAPCHAT_PACKAGE_NAME
    240         )
    241         if (intent == null) {
    242             shortToast("Can't execute action: Snapchat is not installed")
    243             return
    244         }
    245         intent.putExtra(EnumAction.ACTION_PARAMETER, action.key)
    246         androidContext.startActivity(intent)
