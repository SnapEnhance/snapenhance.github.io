    271             context.database.isScriptEnabled(script.name)

    294                 .fillMaxWidth()

    297         ) {
    298             Row(

    338                         context.coroutineScope.launch(Dispatchers.IO) {
    339                             runCatching {
    340                                 val modulePath = context.scriptManager.getModulePath(script.name)!!
    341                                 context.scriptManager.unloadScript(modulePath)
    342                                 if (isChecked) {
    343                                     context.scriptManager.loadScript(modulePath)
    344                                     context.scriptManager.runtime.getModuleByName(script.name)
    345                                         ?.callFunction("module.onSnapEnhanceLoad")
    346                                     context.shortToast("Loaded script ${script.name}")
    347                                 } else {
    348                                     context.shortToast("Unloaded script ${script.name}")
    349                                 }
    350 
    351                                 context.database.setScriptEnabled(script.name, isChecked)
    352                                 withContext(Dispatchers.Main) {
    353                                     enabled = isChecked
    354                                 }
    355                             }.onFailure { throwable ->
    356                                 withContext(Dispatchers.Main) {
    357                                     enabled = !isChecked
    358                                 }
    359                                 ("Failed to ${if (isChecked) "enable" else "disable"} script. Check logs for more details").also {
    360                                     context.log.error(it, throwable)
    361                                     context.shortToast(it)
    362                                 }
    363                             }
    364                         }
    365                     }
    366                 )
    367             }
    368 
    369             if (openSettings) {
    370                 ScriptSettings(script)
    371             }
    372         }
    373 
    374         if (openActions) {
    375             ModuleActions(
    376                 script = script,
    377                 canUpdate = latestUpdate != null,
    378             ) { openActions = false }
    379         }
    380     }

    449     override val content: @Composable (NavBackStackEntry) -> Unit = {
    450         val scriptingFolder by rememberAsyncMutableState(
    451             defaultValue = null,
    452             updateDispatcher = reloadDispatcher
    453         ) {
    454             context.scriptManager.getScriptsFolder()
    455         }
    456         val scriptModules by rememberAsyncMutableState(
    457             defaultValue = emptyList(),
    458             updateDispatcher = reloadDispatcher
    459         ) {
    460             context.scriptManager.sync()
    461             context.scriptManager.getSyncedModules()
    462         }
    463 
    464         val coroutineScope = rememberCoroutineScope()
    465 
    466         var refreshing by remember {
    467             mutableStateOf(false)
    468         }
    469 
    470         LaunchedEffect(Unit) {
    471             refreshing = true
    472             withContext(Dispatchers.IO) {
    473                 reloadDispatcher.dispatch()
    474                 refreshing = false
    475             }
    476         }
    477 
    478         val pullRefreshState = rememberPullRefreshState(refreshing, onRefresh = {
    479             refreshing = true
    480             coroutineScope.launch(Dispatchers.IO) {
    481                 reloadDispatcher.dispatch()
    482                 refreshing = false
    483             }
    484         })
    485 
    486         Box(
    487             modifier = Modifier.fillMaxSize()
    488         ) {
    489             LazyColumn(
    490                 modifier = Modifier
    491                     .fillMaxSize()
    492                     .pullRefresh(pullRefreshState),
    493                 horizontalAlignment = Alignment.CenterHorizontally
    494             ) {
    495                 item {
    496                     if (scriptingFolder == null && !refreshing) {
    497                         Text(
    498                             text = "No scripts folder selected",
    499                             style = MaterialTheme.typography.bodySmall,
    500                             modifier = Modifier.padding(8.dp)
    501                         )
    502                         Spacer(modifier = Modifier.height(8.dp))
    503                         Button(onClick = {
    504                             activityLauncherHelper.chooseFolder {
    505                                 context.config.root.scripting.moduleFolder.set(it)
    506                                 context.config.writeConfig()
    507                                 coroutineScope.launch {
    508                                     reloadDispatcher.dispatch()
    509                                 }
    510                             }
    511                         }) {
    512                             Text(text = "Select folder")
    513                         }
    514                     } else if (scriptModules.isEmpty()) {
    515                         Text(
    516                             text = "No scripts found",
    517                             style = MaterialTheme.typography.bodySmall,
    518                             modifier = Modifier.padding(8.dp)
    519                         )
    520                     }
    521                 }
    522                 items(scriptModules.size, key = { scriptModules[it].hashCode() }) { index ->
    523                     ModuleItem(scriptModules[index])
    524                 }
    525                 item {
    526                     Spacer(modifier = Modifier.height(200.dp))
    527                 }
    528             }
    529 
    530             PullRefreshIndicator(
    531                 refreshing = refreshing,
    532                 state = pullRefreshState,
    533                 modifier = Modifier.align(Alignment.TopCenter)
    534             )
    535         }
    536 
    537         var scriptingWarning by remember {
    538             mutableStateOf(context.sharedPreferences.run {
    539                 getBoolean("scripting_warning", true).also {
    540                     edit().putBoolean("scripting_warning", false).apply()
    541                 }
    542             })
    543         }
    544 
    545         if (scriptingWarning) {
    546             var timeout by remember {
    547                 mutableIntStateOf(10)
    548             }
    549 
    550             LaunchedEffect(Unit) {
    551                 while (timeout > 0) {
    552                     delay(1000)
    553                     timeout--
    554                 }
    555             }
    556 
    557             AlertDialog(onDismissRequest = {
    558                 if (timeout == 0) {
    559                     scriptingWarning = false
    560                 }
    561             }, title = {
    562                 Text(text = context.translation["manager.dialogs.scripting_warning.title"])
    563             }, text = {
    564                 Text(text = context.translation["manager.dialogs.scripting_warning.content"])
    565             }, confirmButton = {
    566                 TextButton(
    567                     onClick = {
    568                         scriptingWarning = false
    569                     },
    570                     enabled = timeout == 0
    571                 ) {
    572                     Text(text = "OK " + if (timeout > 0) "($timeout)" else "")
    573                 }
    574             })
    575         }
    576     }
