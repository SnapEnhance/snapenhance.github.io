     95     private fun openExternalLink(link: String) {
     96         kotlin.runCatching {
     97             context.activity?.startActivity(Intent(Intent.ACTION_VIEW).apply {
     98                 flags = Intent.FLAG_ACTIVITY_NEW_TASK
     99                 data = Uri.parse(link)
    100             })
    101         }.onFailure {
    102             context.log.error("Failed to open external link", it)
    103             context.shortToast("Failed to open external link. Check logs for more details.")
    104         }
