     41     fun fetchRepoIndexes(): Map<String, RepositoryIndex>? {
     42         val indexes = mutableMapOf<String, RepositoryIndex>()
     43 
     44         context.database.getRepositories().forEach { rootUri ->
     45             val indexUri = rootUri.toUri().buildUpon().appendPath("index.json").build()
     46 
     47             runCatching {
     48                 root.okHttpClient.newCall(
     49                     Request.Builder().url(indexUri.toString()).build()
     50                 ).execute().use { response ->
     51                     if (!response.isSuccessful) {
     52                         context.log.error("Failed to fetch theme index from $indexUri: ${response.code}")
     53                         context.shortToast("Failed to fetch index of $indexUri")
     54                         return@forEach
     55                     }
     56 
     57                     runCatching {
     58                         indexes[rootUri] = context.gson.fromJson(response.body.charStream(), RepositoryIndex::class.java)
     59                     }.onFailure {
     60                         context.log.error("Failed to parse theme index from $indexUri", it)
     61                         context.shortToast("Failed to parse index of $indexUri")
     62                     }
     63                 }
     64             }.onFailure {
     65                 context.log.error("Failed to fetch theme index from $indexUri", it)
     66                 context.shortToast("Failed to fetch index of $indexUri")
     67             }
     68         }
     69 
     70         return indexes
     71     }

     73     suspend fun installTheme(themeUri: Uri) {
     74         root.okHttpClient.newCall(
     75             Request.Builder().url(themeUri.toString()).build()
     76         ).execute().use { response ->
     77             if (!response.isSuccessful) {
     78                 context.log.error("Failed to fetch theme from $themeUri: ${response.code}")
     79                 context.shortToast("Failed to fetch theme from $themeUri")
     80                 return
     81             }
     82 
     83             val themeContent = response.body.bytes().toString(Charsets.UTF_8)
     84             root.importTheme(themeContent, themeUri.toString())
     85         }
     86     }

     90     suspend fun refreshCachedIndexes() {
     91         isRefreshing = true
     92         coroutineScope {
     93             launch(Dispatchers.IO) {
     94                 fetchRepoIndexes()?.let {
     95                     context.log.verbose("Fetched ${it.size} theme indexes")
     96                     it.forEach { (t, u) ->
     97                         context.log.verbose("Fetched theme index from $t with ${u.themes.size} themes")
     98                     }
     99                     synchronized(cachedRepoIndexes) {
    100                         cachedRepoIndexes.clear()
    101                         cachedRepoIndexes += it
    102                     }
    103                     cacheReloadDispatcher.dispatch()
    104                     delay(600)
    105                     isRefreshing = false
    106                 }
    107             }
