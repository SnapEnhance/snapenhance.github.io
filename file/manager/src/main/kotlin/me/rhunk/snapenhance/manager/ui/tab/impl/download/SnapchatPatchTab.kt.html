    105     @Composable
    106     private fun SelectSnapchatVersionDialog(onSelected: (DownloadItem) -> Unit = {}) {
    107         val coroutineScope = rememberCoroutineScope()
    108         var isFetching by remember { mutableStateOf(false) }
    109         val downloadItems = remember { cachedDownloadItems.toMutableStateList() }
    110 
    111         LazyColumn {
    112             items(downloadItems, key = { it.hash }) { item ->
    113                 DownloadItemRow(item) {
    114                     onSelected(item)
    115                 }
    116             }
    117             item {
    118                 Box(
    119                     modifier = Modifier
    120                         .fillMaxWidth()
    121                         .padding(20.dp),
    122                     contentAlignment = Alignment.Center
    123                 ) {
    124                     CircularProgressIndicator(modifier = Modifier.alpha(if (isFetching) 1f else 0f))
    125                 }
    126                 SideEffect {
    127                     if (isFetching) return@SideEffect
    128                     isFetching = true
    129                     coroutineScope.launch {
    130                         runCatching {
    131                             withContext(Dispatchers.IO) {
    132                                 apkMirror.fetchSnapchatVersions(currentPage)?.let {
    133                                     withContext(Dispatchers.Main) {
    134                                         cachedDownloadItems.addAll(it)
    135                                         downloadItems.addAll(it)
    136                                     }
    137                                 }
    138                             }
    139                         }.onFailure {
    140                             it.printStackTrace()
    141                         }
    142                         ++currentPage
    143                         isFetching = false
    144                     }
    145                 }
    146             }
    147         }
    148     }
