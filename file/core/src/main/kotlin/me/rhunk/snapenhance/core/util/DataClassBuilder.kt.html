     19     fun set(fieldName: String, value: Any?) {
     20         val field = instance::class.java.declaredFields.firstOrNull { it.name == fieldName } ?: return
     21         val fieldType = field.type
     22         field.isAccessible = true
     23 
     24         when {
     25             fieldType.isEnum -> {
     26                 val enumValue = fieldType.enumConstants.firstOrNull { it.toString() == value } ?: return
     27                 field.set(instance, enumValue)
     28             }
     29             fieldType.isPrimitive -> {
     30                 when (fieldType) {
     31                     Boolean::class.javaPrimitiveType -> field.setBoolean(instance, value as Boolean)
     32                     Byte::class.javaPrimitiveType -> field.setByte(instance, value as Byte)
     33                     Char::class.javaPrimitiveType -> field.setChar(instance, value as Char)
     34                     Short::class.javaPrimitiveType -> field.setShort(instance, value as Short)
     35                     Int::class.javaPrimitiveType -> field.setInt(instance, value as Int)
     36                     Long::class.javaPrimitiveType -> field.setLong(instance, value as Long)
     37                     Float::class.javaPrimitiveType -> field.setFloat(instance, value as Float)
     38                     Double::class.javaPrimitiveType -> field.setDouble(instance, value as Double)
     39                 }
     40             }
     41             else -> field.set(instance, value)
     42         }
     43     }

     54     fun from(fieldName: String, new: Boolean = false, callback: DataClassBuilder.() -> Unit) {
     55         val field = instance::class.java.declaredFields.firstOrNull { it.name == fieldName } ?: return
     56         field.isAccessible = true
     57
