     41             if (callingState.toString() == "IN_CALL") {

     48             getConstructor(
     49                 AudioAttributes::class.java,
     50                 AudioFormat::class.java,
     51                 Int::class.javaPrimitiveType,
     52                 Int::class.javaPrimitiveType,
     53                 Int::class.javaPrimitiveType,

     60                 lateinit var streamUrl: String

     62                     object: HttpServer.HttpContent() {
     63                         override val contentType: String = "audio/wav"
     64                         override val chunked: Boolean = true

     66                         override val newBody: () -> HttpServer.HttpBody = {
     67                             object: HttpServer.HttpBody() {
     68                                 val outputStream = PipedOutputStream()
     69                                 val inputStream = PipedInputStream(outputStream)
     70 
     71                                 val handler: (byteArray: ByteArray) -> Unit = handler@{ byteArray ->
     72                                     if (byteArray.isEmpty()) {
     73                                         httpServer.removeUrl(streamUrl)
     74                                         return@handler
     75                                     }
     76                                     runCatching {
     77                                         outputStream.write(byteArray)
     78                                         outputStream.flush()
     79                                     }.onFailure {
     80                                         context.log.warn("Failed to write to streaming url ${it.localizedMessage}")
     81                                     }
     82                                 }

     86                                 }
     87 
     88                                 override val readBytes: (byteArray: ByteArray) -> Int = { byteArray ->
     89                                     runBlocking {
     90                                         withTimeoutOrNull(3000L) {
     91                                             inputStream.read(byteArray)
     92                                         } ?: -1
     93                                     }
     94                                 }
     95 
     96                                 override val onClose: () -> Unit = {
     97                                     context.log.verbose("Streaming url closed")
     98                                     streamHandlers[hashCode]?.remove(handler)
     99                                     outputStream.close()
    100                                     inputStream.close()
    101                                 }
    102                             }
    103                         }
    104                     }
    105                 ) ?: return@hook
    106 
    107                 context.log.verbose("streaming url = $streamUrl, sampleRate = ${audioFormat.sampleRate}, audioFormat = ${audioFormat.encoding}")
    108 
    109                 context.feature(MediaDownloader::class).provideDownloadManagerClient(
    110                     UUID.randomUUID().toString(),
    111                     participants.mapNotNull { context.database.getFriendInfo(it)?.mutableUsername }.joinToString("-"),
    112                     System.currentTimeMillis(),
    113                     MediaDownloadSource.VOICE_CALL
    114                 ).downloadStream(streamUrl, AudioStreamFormat(audioFormat.channelCount, audioFormat.sampleRate, audioFormat.encoding))
    115             }
    116 
    117             getMethod("write", ByteBuffer::class.java, Int::class.javaPrimitiveType, Int::class.javaPrimitiveType).hook(HookStage.BEFORE) { param ->
    118                 streamHandlers[param.thisObject<Any>().hashCode()]?.let { handlers ->
    119                     val byteBuffer = param.arg<ByteBuffer>(0)
    120                     val position = byteBuffer.position()
    121                     val buffer = ByteArray(param.arg(1))
    122                     byteBuffer.get(buffer)
    123                     byteBuffer.position(position)
    124                     handlers.forEach { it(buffer) }
    125                 }
    126             }
    127 
    128             hook("release", HookStage.BEFORE) {
    129                 streamHandlers.remove(it.thisObject<Any>().hashCode())?.forEach { it(ByteArray(0)) }
    130             }
