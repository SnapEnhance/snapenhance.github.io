    104     fun editCurrentMessage(
    105         localContext: Context,
    106         dismissActionMenu: () -> Unit,
    107     ) {
    108         val messaging = context.feature(Messaging::class)
    109         messaging.conversationManager?.fetchMessage(
    110             messaging.openedConversationUUID.toString(),
    111             messaging.lastFocusedMessageId,
    112             onSuccess = onSuccess@{ message ->
    113                 dismissActionMenu()
    114                 if (message.senderId.toString() != context.database.myUserId) {
    115                     context.shortToast("You can only edit your own messages")
    116                     return@onSuccess
    117                 }
    118 
    119                 val editText = EditText(localContext).apply {
    120                     setText(ProtoReader(message.messageContent?.content ?: return@apply).getString(2, 1) ?: run {
    121                         this@NewChatActionMenu.context.shortToast("You can only edit text messages")
    122                         return@onSuccess
    123                     })
    124                     setTextColor(resources.getColor(android.R.color.white, context.theme))
    125                     postDelayed({
    126                         requestFocus()
    127                         setSelection(text.length)
    128                         context.getSystemService(android.content.Context.INPUT_METHOD_SERVICE)
    129                             .let { it as android.view.inputmethod.InputMethodManager }
    130                             .showSoftInput(this, android.view.inputmethod.InputMethodManager.SHOW_IMPLICIT)
    131                     }, 200)
    132                 }
    133 
    134                 context.runOnUiThread {
    135                     ViewAppearanceHelper.newAlertDialogBuilder(context.mainActivity!!)
    136                         .setPositiveButton("Save") { _, _ ->
    137                             val newMessageContent = ProtoWriter().apply {
    138                                 from(2) { addString(1, editText.text.toString()) }
    139                             }.toByteArray()
    140                             message.messageContent?.content = newMessageContent
    141                             messaging.conversationManager?.editMessage(
    142                                 message.messageDescriptor?.conversationId.toString(),
    143                                 message.messageDescriptor?.messageId ?: return@setPositiveButton,
    144                                 newMessageContent,
    145                                 onSuccess = {
    146                                     context.coroutineScope.launch(Dispatchers.Main) {
    147                                         message.messageMetadata?.isEdited = true
    148                                         messaging.localUpdateMessage(
    149                                             message.messageDescriptor?.conversationId.toString(),
    150                                             message,
    151                                             forceUpdate = true
    152                                         )
    153                                     }
    154                                 },
    155                                 onError = {
    156                                     context.shortToast("Failed to edit message: $it")
    157                                 }
    158                             )
    159                         }
    160                         .setNegativeButton("Cancel") { dialog, _ -> dialog.dismiss() }
    161                         .setView(editText)
    162                         .setTitle("Edit message content")
    163                         .show()
    164                 }
    165             }
    166         )
    167     }
