    218     fun handle(event: AddViewEvent) {
    219         if (event.parent is LinearLayout) return
    220         val closeActionMenu = { event.parent.iterateParent {
    221             it.triggerCloseTouchEvent()

    225         val mediaDownloader = context.feature(MediaDownloader::class)
    226         val messageLogger = context.feature(MessageLogger::class)

    228 
    229         val composeView = createComposeView(event.view.context) {
    230             val primaryColor = remember { if (event.view.context.isDarkTheme()) Color.White else Color.Black }

    232                 FontFamily(

    237             @Composable
    238             fun ListButton(
    239                 modifier: Modifier = Modifier,
    240                 icon: ImageVector,
    241                 text: String,
    242             ) {
    243                 Row(
    244                     modifier = Modifier
    245                         .fillMaxWidth()
    246                         .then(modifier)
    247                         .padding(top = 11.dp, bottom = 11.dp),
    248                     horizontalArrangement = Arrangement.spacedBy(16.dp)
    249                 ) {
    250                     Icon(
    251                         modifier = Modifier
    252                             .padding(start = 16.dp),
    253                         imageVector = icon,
    254                         tint = primaryColor,
    255                         contentDescription = text
    256                     )
    257                     Text(text, color = primaryColor, fontFamily = avenirNextMediumFont, fontSize = 16.sp)
    258                 }
    259                 Spacer(modifier = Modifier
    260                     .height(1.dp)
    261                     .fillMaxWidth()
    262                     .border(1.dp, MaterialTheme.colorScheme.onSurface.copy(alpha = 0.05f)))
    263             }
    264 
    265             Column(
    266                 modifier = Modifier.fillMaxWidth(),
    267             ) {
    268                 if (context.config.downloader.downloadContextMenu.get()) {
    269                     ListButton(icon = Icons.Outlined.RemoveRedEye, text = context.translation["chat_action_menu.preview_button"], modifier = Modifier.clickable {
    270                         closeActionMenu()
    271                         mediaDownloader.onMessageActionMenu(true)
    272                     })
    273                     ListButton(icon = Icons.Outlined.Download, text = context.translation["chat_action_menu.download_button"], modifier = Modifier.pointerInput(Unit) {
    274                         detectTapGestures(
    275                             onTap = {
    276                                 closeActionMenu()
    277                                 mediaDownloader.onMessageActionMenu(false)
    278                             },
    279                             onLongPress = {
    280                                 context.androidContext.vibrateLongPress()
    281                                 mediaDownloader.onMessageActionMenu(isPreviewMode = false, forceAllowDuplicate = true)
    282                             }
    283                         )
    284                     })
    285                 }
    286 
    287                 if (context.config.messaging.messageLogger.globalState == true) {
    288                     val chatEdits by rememberAsyncMutableState(defaultValue = null) {
    289                         context.feature(MessageLogger::class).getChatEdits(
    290                             messaging.openedConversationUUID.toString(),
    291                             messaging.lastFocusedMessageId
    292                         )
    293                     }
    294 
    295                     if (chatEdits != null && chatEdits?.isNotEmpty() == true) {
    296                         ListButton(icon = Icons.Outlined.History, text = context.translation["chat_action_menu.show_chat_edit_history"], modifier = Modifier.clickable {
    297                             closeActionMenu()
    298                             showChatEditHistory(chatEdits!!)
    299                         })
    300                     }
    301 
    302                     ListButton(icon = Icons.Outlined.BookmarkRemove, text = context.translation["chat_action_menu.delete_logged_message_button"], modifier = Modifier.clickable {
    303                         closeActionMenu()
    304                         context.executeAsync {
    305                             messageLogger.deleteMessage(messaging.openedConversationUUID.toString(), messaging.lastFocusedMessageId)
    306                         }
    307                     })
    308                 }
    309 
    310                 if (context.config.experimental.convertMessageLocally.get()) {
    311                     ListButton(icon = Icons.Outlined.Image, text = context.translation["chat_action_menu.convert_message"], modifier = Modifier.clickable {
    312                         closeActionMenu()
    313                         messaging.conversationManager?.fetchMessage(
    314                             messaging.openedConversationUUID.toString(),
    315                             messaging.lastFocusedMessageId,
    316                             onSuccess = {
    317                                 context.runOnUiThread {
    318                                     runCatching {
    319                                         context.feature(ConvertMessageLocally::class)
    320                                             .convertMessageInterface(it)
    321                                     }.onFailure {
    322                                         context.log.verbose("Failed to convert message: $it")
    323                                         context.shortToast("Failed to edit message: $it")
    324                                     }
    325                                 }
    326                             },
    327                             onError = {
    328                                 context.shortToast("Failed to fetch message: $it")
    329                             }
    330                         )
    331                     })
    332                 }
    333             }
    334         }.apply {
    335             layoutParams = LinearLayout.LayoutParams(
    336                 ViewGroup.LayoutParams.MATCH_PARENT,
    337                 ViewGroup.LayoutParams.WRAP_CONTENT
    338             )
    339         }
    340         event.view = ScrollView(event.view.context).apply {
    341             addView(LinearLayout(context).apply {
    342                 orientation = LinearLayout.VERTICAL
    343                 layoutParams = ViewGroup.LayoutParams(
    344                     ViewGroup.LayoutParams.MATCH_PARENT,
    345                     ViewGroup.LayoutParams.MATCH_PARENT
    346                 )
    347                 addView(composeView)
    348                 composeView.post {
    349                     (event.parent.layoutParams as ViewGroup.MarginLayoutParams).apply {
    350                         if (height < composeView.measuredHeight) {
    351                             height += composeView.measuredHeight
    352                         } else {
    353                             setObjectField("a", null) // remove drag callback
    354                         }
    355                     }
    356                     event.parent.requestLayout()
    357                 }
    358                 addView(event.view)
    359             })
    360         }
    361     }
