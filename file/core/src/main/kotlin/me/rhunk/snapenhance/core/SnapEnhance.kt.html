     51     private fun hookMainActivity(methodName: String, stage: HookStage = HookStage.AFTER, block: Activity.(param: HookAdapter) -> Unit) {
     52         Activity::class.java.hook(methodName, stage, { isBridgeInitialized }) { param ->
     53             val activity = param.thisObject() as Activity
     54             if (!activity.packageName.equals(Constants.SNAPCHAT_PACKAGE_NAME)) return@hook
     55             block(activity, param)
     56         }
     57     }

     59     fun init(context: Context) {
     60         appContext = ModContext(
     61             androidContext = context.also { classLoader = it.classLoader }
     62         )
     63         appContext.apply {
     64             bridgeClient = BridgeClient(this)
     65             initConfigListener()
     66             bridgeClient.addOnConnectedCallback {
     67                 bridgeClient.registerMessagingBridge(messagingBridge)
     68                 coroutineScope.launch {
     69                     runCatching {
     70                         syncRemote()
     71                     }.onFailure {
     72                         log.error("Failed to sync remote", it)
     73                     }
     74                 }
     75             }
     76         }
     77 
     78         runBlocking {
     79             var throwable: Throwable? = null
     80             val canLoad = appContext.bridgeClient.connect { throwable = it }
     81             if (canLoad == null) {
     82                 InAppOverlay.showCrashOverlay(
     83                     buildString {
     84                         append("Snapchat timed out while trying to connect to SnapEnhance\n\n")
     85                         append("Make sure you:\n")
     86                         append(" - Have installed the latest SnapEnhance version (https://github.com/rhunk/SnapEnhance)\n")
     87                         append(" - Disabled battery optimizations\n")
     88                         append(" - Excluded SnapEnhance and Snapchat in HideMyApplist")
     89                     },
     90                     throwable
     91                 )
     92                 appContext.logCritical("Cannot connect to the SnapEnhance app")
     93                 return@runBlocking
     94             }
     95             if (!canLoad) exitProcess(1)
     96             runCatching {
     97                 LSPatchUpdater.onBridgeConnected(appContext)
     98             }.onFailure {
     99                 appContext.log.error("Failed to init LSPatchUpdater", it)
    100             }
    101             jetpackComposeResourceHook()
    102             runCatching {
    103                 measureTimeMillis {
    104                     init(this)
    105                 }.also {
    106                     appContext.log.verbose("init took ${it}ms")
    107                 }
    108 
    109                 hookMainActivity("onPostCreate") {
    110                     appContext.mainActivity = this
    111                     if (!appContext.mappings.isMappingsLoaded) return@hookMainActivity
    112                     appContext.isMainActivityPaused = false
    113                     onActivityCreate(this)
    114                     appContext.actionManager.onNewIntent(intent)
    115                 }
    116 
    117                 hookMainActivity("onPause") {
    118                     appContext.bridgeClient.closeOverlay()
    119                     appContext.isMainActivityPaused = true
    120                 }
    121 
    122                 hookMainActivity("onNewIntent") { param ->
    123                     appContext.actionManager.onNewIntent(param.argNullable(0))
    124                 }
    125 
    126                 hookMainActivity("onResume") {
    127                     if (appContext.isMainActivityPaused.also {
    128                         appContext.isMainActivityPaused = false
    129                     }) {
    130                         appContext.reloadConfig()
    131                         appContext.executeAsync {
    132                             syncRemote()
    133                         }
    134                     }
    135                 }
    136             }.onSuccess {
    137                 isBridgeInitialized = true
    138             }.onFailure {
    139                 appContext.logCritical("Failed to initialize bridge", it)
    140                 InAppOverlay.showCrashOverlay("SnapEnhance failed to initialize. Please check logs for more details.", it)
    141             }
    142         }
    143     }

    145     private fun init(scope: CoroutineScope) {
    146         with(appContext) {
    147             Thread::class.java.hook("dispatchUncaughtException", HookStage.BEFORE) { param ->

    154             reloadConfig()

    160             }
    161 
    162             mappings.init(androidContext)

    164             eventDispatcher.init()

    167             features.init()
    168             scriptRuntime.init()
    169             scriptRuntime.eachModule { callFunction("module.onSnapApplicationLoad", androidContext) }
    170         }

    196     private fun initNative() {
    197         val nativeSigCacheFileHandle = appContext.fileHandlerManager.getFileHandle(FileHandleScope.INTERNAL.key, InternalFileHandleType.NATIVE_SIG_CACHE.key).toWrapper()
    198 
    199         val oldSignatureCache = nativeSigCacheFileHandle.readBytes()
    200             .takeIf {
    201                 it.isNotEmpty()
    202             }?.toString(Charsets.UTF_8)?.also {
    203                 appContext.native.signatureCache = it
    204                 appContext.log.verbose("old signature cache $it")
    205             }
    206 
    207         val lateInit = appContext.native.initOnce {
    208             nativeUnaryCallCallback = { request ->
    209                 appContext.event.post(NativeUnaryCallEvent(request.uri, request.buffer)) {
    210                     request.buffer = buffer
    211                     request.canceled = canceled
    212                 }
    213             }
    214             appContext.reloadNativeConfig()
    215         }.let { init ->
    216             {
    217                 init()
    218                 appContext.native.signatureCache.takeIf { it != oldSignatureCache }?.let {
    219                     appContext.log.verbose("new signature cache $it")
    220                     nativeSigCacheFileHandle.writeBytes(it.toByteArray(Charsets.UTF_8))
    221                 }
    222             }
    223         }
    224 
    225         appContext.config.experimental.nativeHooks.customSharedLibrary.get().takeIf { it.isNotEmpty() }?.let {
    226             runCatching {
    227                 appContext.native.loadSharedLibrary(
    228                     appContext.fileHandlerManager.getFileHandle(FileHandleScope.USER_IMPORT.key, it).toWrapper().readBytes()
    229                 )
    230                 appContext.log.verbose("loaded custom shared library")
    231             }.onFailure {
    232                 appContext.log.error("Failed to load custom shared library", it)
    233             }
    234         }
    235 
    236         if (appContext.bridgeClient.getDebugProp("disable_sif", "false") != "true") {
    237             runCatching {
    238                 appContext.native.loadSharedLibrary(
    239                     appContext.fileHandlerManager.getFileHandle(FileHandleScope.INTERNAL.key, InternalFileHandleType.SIF.key)
    240                         .toWrapper()
    241                         .readBytes()
    242                         .takeIf {
    243                             it.isNotEmpty()
    244                         } ?: throw IllegalStateException("buffer is empty")
    245                 )
    246                 appContext.log.verbose("loaded sif")
    247             }.onFailure {
    248                 safeMode = true
    249                 appContext.log.error("Failed to load sif", it)
    250             }
    251         } else {
    252             appContext.log.warn("sif is disabled")
    253         }
    254 
    255         Runtime::class.java.findRestrictedMethod {
    256             it.name == "loadLibrary0" && it.parameterTypes.contentEquals(
    257                 if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) arrayOf(Class::class.java, String::class.java)
    258                 else arrayOf(ClassLoader::class.java, String::class.java)
    259             )
    260         }!!.apply {
    261             if (safeMode) {
    262                 hook(HookStage.BEFORE) { param ->
    263                     if (param.arg<String>(1) != "scplugin") return@hook
    264                     param.setResult(null)
    265                     appContext.log.warn("Can't load scplugin in safe mode")
    266                     runCatching {
    267                         Thread.sleep(Long.MAX_VALUE)
    268                     }.onFailure {
    269                         appContext.log.error(it)
    270                     }
    271                     exitProcess(1)
    272                 }
    273             }
    274 
    275             lateinit var unhook: () -> Unit
    276             hook(HookStage.AFTER) { param ->
    277                 if (param.arg<String>(1) != "client") return@hook
    278                 unhook()
    279                 appContext.log.verbose("libclient lateInit")
    280                 lateInit()
    281             }.also { unhook = { it.unhook() } }
    282         }
    283     }

    287         hookMainActivity("onResume") {
    288             tasks.forEach { it() }

    292             if (appContext.isMainActivityPaused) {
    293                 tasks.add(task)
    294             } else {
    295                 task()

    297         }

    326     private fun initWidgetListener() {
    327         appContext.event.subscribe(SnapWidgetBroadcastReceiveEvent::class) { event ->
    328             if (event.action != ReceiversConfig.BRIDGE_SYNC_ACTION) return@subscribe
    329             event.canceled = true
    330             val feedEntries = appContext.database.getFeedEntries(Int.MAX_VALUE)
    331 
    332             val groups = feedEntries.filter { it.friendUserId == null }.map {
    333                 MessagingGroupInfo(
    334                     it.key!!,
    335                     it.feedDisplayName!!,
    336                     it.participantsSize
    337                 )
    338             }
    339 
    340             val friends = feedEntries.filter { it.friendUserId != null }.map {
    341                 MessagingFriendInfo(
    342                     it.friendUserId!!,
    343                     appContext.database.getConversationLinkFromUserId(it.friendUserId!!)?.clientConversationId,
    344                     it.friendDisplayName,
    345                     it.friendDisplayUsername!!.split("|")[1],
    346                     it.bitmojiAvatarId,
    347                     it.bitmojiSelfieId,
    348                     streaks = null
    349                 )
    350             }
    351 
    352             appContext.bridgeClient.passGroupsAndFriends(groups, friends)
    353         }
    354     }

    389     private fun jetpackComposeResourceHook() {
    390         fun strings(vararg classes: KClass<*>): Map<Int, String> {
    391             return classes.fold(mapOf()) { map, clazz ->
    392                 map + clazz.java.fields.filter {
    393                     java.lang.reflect.Modifier.isStatic(it.modifiers) && it.type == Int::class.javaPrimitiveType
    394                 }.associate { it.getInt(null) to it.name }
    395             }
    396         }
    397         val stringResources = strings(androidx.compose.material3.R.string::class, androidx.compose.ui.R.string::class)
    398         Resources::class.java.getMethod("getString", Int::class.javaPrimitiveType).hook(HookStage.BEFORE) { param ->
    399             val key = param.arg<Int>(0)
    400             val name = stringResources[key]?.replaceFirst("m3c_", "") ?: return@hook
    401             param.setResult(appContext.translation.getOrNull("material3_strings.${name}") ?: "")
    402         }
    403     }
