     41     private fun transact(option: Int, option2: Long) = runCatching { Os.prctl(option, option2, 0, 0, 0) }.getOrNull()

     55         // load user shared library

     64                 lateinit var composable: CustomComposable

     66                     Row(
     67                         modifier = Modifier

     95             "/messagingcoreservice.MessagingCoreService/",

    102         context.event.subscribe(UnaryCallEvent::class) { event ->
    103             val callOptions = event.adapter.arg<Any>(2).let { it.javaClass.getMethod("build").invoke(it) } ?: return@subscribe
    104             if (callOptions.getObjectField("mAttestation") != null) {
    105                 context.log.verbose("blocked ep ${event.adapter.arg<Any>(0)}", "UnaryCallEvent")
    106                 event.canceled = true
    107                 val eventHandler = event.adapter.arg<Any>(3)
    108                 eventHandler.javaClass.methods.first { it.name == "onEvent" }.also { method ->

    122             }
    123             loadClass("com.snapchat.client.client_attestation.ArgosClient").hook("createInstance", HookStage.BEFORE) { param ->
    124                 param.setResult(argosClientClass.declaredConstructors.first().also { it.isAccessible = true }.newInstance(0))
    125             }
    126             loadClass("com.snap.security.attestation.impl.SCClientAttestationDurableJob").hookConstructor(HookStage.BEFORE) { param ->
    127                 param.setArg(0, null)
    128             }
    129             loadClass("com.snapchat.client.grpc.AuthContext").hookConstructor(HookStage.AFTER) { param ->
    130                 val headers by lazy { (param.thisObject<Any>().getObjectField("mHeaders") as? List<*>)?.filterNotNull() ?: emptyList() }
    131 
    132                 if (param.thisObject<Any>().getObjectField("mAuthTokenErrorCode") != null ||
    133                     headers.isEmpty() ||
    134                     headers.mapNotNull { it.getObjectField("mKey")?.toString()?.lowercase() }.any { it != "x-snap-access-token" }
    135                 ) {
    136                     context.log.error("invalid headers ${headers.size}")

    138                 }
    139             }

    142         context.mappings.useMapper(CallbackMapper::class) {
    143             callbacks.getClass("AuthContextDelegate")?.hook("getAuthContext", HookStage.BEFORE) { param ->

    145                 val requestPath = authContextRequest.getObjectField("mRequestPath").toString()
    146 
    147                 if (authContextRequest.getObjectField("mAttestationRequired") == true) {
    148                     if (allowedEPs.any { requestPath.contains(it) }) {

    169                         )
    170                         return@hook
    171                     }
    172 
    173                     param.setResult(null)
    174                 }
