     41     private fun isSafeMode(): Boolean {
     42         val snapchatVersionCode = context.androidContext.packageManager?.getPackageInfo(context.androidContext.packageName, 0)?.longVersionCode ?: throw IllegalStateException("Failed to get version code")
     43         val shouldUseSafeMode = MOD_DETECTION_VERSION_CHECK.checkVersion(snapchatVersionCode)?.second == VersionRequirement.OLDER_REQUIRED
     44 
     45         context.config.experimental.nativeHooks.customSharedLibrary.get().takeIf { it.isNotEmpty() }?.let {
     46             runCatching {
     47                 context.native.loadSharedLibrary(
     48                     context.fileHandlerManager.getFileHandle(FileHandleScope.USER_IMPORT.key, it).toWrapper().readBytes()
     49                 )
     50                 context.log.verbose("loaded custom shared library")
     51             }.onFailure {
     52                 context.log.error("Failed to load custom shared library", it)
     53                 return true
     54             }
     55         } ?: context.bridgeClient.getDebugProp("enable_security_features", "false").takeIf { it == "true" }?.runCatching {
     56             context.native.loadSharedLibrary(
     57                 context.fileHandlerManager.getFileHandle(FileHandleScope.INTERNAL.key, InternalFileHandleType.SIF.key)
     58                     .toWrapper()
     59                     .readBytes()
     60                     .takeIf {
     61                         it.isNotEmpty()
     62                     } ?: throw IllegalStateException("Binary is empty")
     63             )
     64             context.log.verbose("loaded sif")
     65         }?.onFailure {
     66             context.log.error("Failed to load sif: " + it.message)
     67             return shouldUseSafeMode
     68         } ?: context.log.warn("Security features are disabled")
     69 
     70         token // pre init token
     71 
     72         val status = getStatus()
     73         val safeMode = shouldUseSafeMode && (status == null || status < 2)
     74 
     75         if (status != null && status >= 2) {
     76             context.log.verbose("status=$status")
     77             lateinit var composable: CustomComposable
     78             composable = {
     79                 Row(
     80                     modifier = Modifier
     81                         .padding(16.dp)
     82                         .align(Alignment.TopCenter),
     83                 ) {
     84                     Icon(Icons.Filled.Check, contentDescription = null, tint = Color(0xFF85A947))
     85                 }
     86 
     87                 LaunchedEffect(Unit) {
     88                     delay(2500)
     89                     context.inAppOverlay.removeCustomComposable(composable)
     90                 }
     91             }
     92             context.inAppOverlay.addCustomComposable(composable)
     93         }
     94 
     95         return safeMode
     96     }
