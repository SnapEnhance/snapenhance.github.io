    136                     val viewItemContainer = param.arg<Any>(0)

    177     fun clearConversationFromFeed(conversationId: String, onError : (String) -> Unit = {}, onSuccess : () -> Unit = {}) {
    178         conversationManager?.clearConversation(conversationId, onError = { onError(it) }, onSuccess = {
    179             runCatching {
    180                 conversationManagerDelegate!!.let {
    181                     it::class.java.methods.first { method ->
    182                         method.name == "onConversationRemoved"
    183                     }.invoke(conversationManagerDelegate, conversationId.toSnapUUID().instanceNonNull())
    184                 }
    185                 onSuccess()
    186             }.onFailure {
    187                 context.log.error("Failed to invoke onConversationRemoved: $it")
    188                 onError(it.message ?: "Unknown error")
    189             }
    190         })
    191     }

    193     fun localUpdateMessage(conversationId: String, message: Message, forceUpdate: Boolean = false) {
    194         if (forceUpdate) {
    195             message.messageMetadata?.screenRecordedBy = ArrayList<SnapUUID>(message.messageMetadata?.screenRecordedBy ?: emptyList()).apply {
    196                 add(SnapUUID(UUID.randomUUID().toString()))
    197             }
    198         }
    199         conversationManagerDelegate?.let {
    200             it::class.java.methods.first { method ->
    201                 method.name == "onConversationUpdated"
    202             }.invoke(conversationManagerDelegate, conversationId.toSnapUUID().instanceNonNull(), null, mutableListOf(message.instanceNonNull()), mutableListOf<Any>())
    203         }
    204     }

    206     fun fetchSnapchatterInfos(userIds: List<String>): List<Snapchatter> {
    207         val identity = identityDelegate ?: return emptyList()
    208         val snapUUIDs = userIds.map {
    209             it.toSnapUUID().instanceNonNull()
    210         }
    211 
    212         val future = identity::class.java.methods.first {
    213             it.name == "fetchSnapchatterInfos"
    214         }.let { method ->
    215             if (method.parameterCount == 2) method.invoke(identity, snapUUIDs, false)
    216             else method.invoke(identity, snapUUIDs)
    217         } as Future<*>
    218 
    219         return (future.get() as? List<*>)?.map { Snapchatter(it) } ?: return emptyList()
    220     }
