    414     private fun tryDecryptMessage(senderId: String, clientMessageId: Long, conversationId: String, contentType: ContentType, messageBuffer: ByteArray): Pair<ContentType, ByteArray> {
    415         if (contentType != ContentType.STATUS && decryptedMessageCache.containsKey(clientMessageId)) {
    416             return decryptedMessageCache[clientMessageId]!!
    417         }
    418 
    419         val reader = ProtoReader(messageBuffer)
    420         var outputBuffer = messageBuffer
    421         var outputContentType = fixContentType(contentType, reader) ?: contentType
    422         val conversationParticipants by lazy {
    423             getE2EParticipants(conversationId)
    424         }

    426         fun setDecryptedMessage(buffer: ByteArray) {
    427             outputBuffer = buffer

    443         fun replaceMessageText(text: String) {
    444             outputBuffer = ProtoWriter().apply {
    445                 from(2) {
    446                     addString(1, text)
    447                 }
    448             }.toByteArray()
    449         }
    450 
    451         // decrypt messages
    452         reader.followPath(2, 1) {
    453             val messageTypeId = getVarInt(1)?.toInt() ?: return@followPath
    454             val isMe = context.database.myUserId == senderId
    455 
    456             if (messageTypeId == ENCRYPTED_MESSAGE_ID) {
    457                 runCatching {
    458                     eachBuffer(2) {
    459                         if (decryptedMessageCache.containsKey(clientMessageId)) return@eachBuffer
    460 
    461                         val participantIdHash = getByteArray(1) ?: return@eachBuffer
    462                         val iv = getByteArray(2) ?: return@eachBuffer
    463                         val ciphertext = getByteArray(3) ?: return@eachBuffer
    464 
    465                         if (isMe) {
    466                             if (conversationParticipants.isEmpty()) return@eachBuffer
    467                             val participantId = conversationParticipants.firstOrNull { participantIdHash.contentEquals(hashParticipantId(it, iv)) } ?: return@eachBuffer
    468                             setDecryptedMessage(e2eeInterface.decryptMessage(participantId, ciphertext, iv) ?: run {
    469                                 context.log.warn("Failed to decrypt message for participant $participantId")
    470                                 setWarningMessage()
    471                                 return@eachBuffer
    472                             })
    473                             return@eachBuffer
    474                         }
    475 
    476                         if (!participantIdHash.contentEquals(hashParticipantId(context.database.myUserId, iv))) return@eachBuffer
    477 
    478                         setDecryptedMessage(e2eeInterface.decryptMessage(senderId, ciphertext, iv)?: run {
    479                             setWarningMessage()
    480                             return@eachBuffer
    481                         })
    482                     }
    483                 }.onFailure {
    484                     context.log.error("Failed to decrypt message id: $clientMessageId", it)
    485                     setWarningMessage()
    486                 }
    487 
    488                 return@followPath
    489             }
    490 
    491             val payload = getByteArray(2, 2) ?: return@followPath
    492 
    493             if (senderId == context.database.myUserId) {
    494                 when (messageTypeId) {
    495                     REQUEST_PK_MESSAGE_ID -> {
    496                         replaceMessageText("[${translation["outgoing_pk_message"]}]")
    497                     }
    498                     RESPONSE_SK_MESSAGE_ID -> {
    499                         replaceMessageText("[${translation["outgoing_secret_message"]}]")
    500                     }
    501                 }
    502                 return@followPath
    503             }
    504 
    505             when (messageTypeId) {
    506                 REQUEST_PK_MESSAGE_ID -> {
    507                     pkRequests[clientMessageId] = payload
    508                     replaceMessageText(translation["incoming_pk_message"])
    509                 }
    510                 RESPONSE_SK_MESSAGE_ID -> {
    511                     secretResponses[clientMessageId] = payload
    512                     replaceMessageText(translation["incoming_secret_message"])
    513                 }
    514             }
    515         }
    516 
    517         return outputContentType to outputBuffer
    518     }

    520     private fun messageHook(conversationId: String, messageId: Long, senderId: String, messageContent: MessageContent, committed: Boolean) {
    521         val (contentType, buffer) = tryDecryptMessage(senderId, messageId, conversationId, messageContent.contentType ?: ContentType.CHAT, messageContent.content!!)
    522         messageContent.contentType = contentType
    523         messageContent.content = buffer
    524         // remove messages currently being sent from the cache
    525         if (!committed) {
    526             decryptedMessageCache.remove(messageId)
    527             encryptedMessages.remove(messageId)
    528         }
    529     }

    531     private fun ProtoWriter.writeEncryptedMessage(
    532         participantsIds: List<String>,
    533         messageContent: ByteArray,
    534     ) {
    535         from(2) {
    536             from(1) {
    537                 addVarInt(1, ENCRYPTED_MESSAGE_ID)
    538                 participantsIds.forEach { participantId ->
    539                     val encryptedMessage = e2eeInterface.encryptMessage(participantId,
    540                         messageContent
    541                     ) ?: run {
    542                         throw Exception("Failed to encrypt message for participant $participantId")
    543                     }
    544                     context.log.debug("encrypted message size = ${encryptedMessage.ciphertext.size}")
    545                     from(2) {
    546                         // participantId is hashed with iv to prevent leaking it when sending to multiple conversations
    547                         addBuffer(1, hashParticipantId(participantId, encryptedMessage.iv))
    548                         addBuffer(2, encryptedMessage.iv)
    549                         addBuffer(3, encryptedMessage.ciphertext)
    550                     }
    551                 }
    552                 if (ContentType.fromMessageContainer(ProtoReader(messageContent)) == ContentType.SNAP) {
    553                     addVarInt(5, 1)
    554                 }
    555             }
    556         }
    557     }
