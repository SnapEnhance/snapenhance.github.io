     57     private fun getLat() : Double {
     58         var spoofedLatitude = context.config.global.betterLocation.coordinates.get().first
     59         walkRadius?.let {
     60             spoofedLatitude += randomWalking.current_x

     65     private fun getLong() : Double {
     66         var spoofedLongitude = context.config.global.betterLocation.coordinates.get().second

     73     private fun editClientUpdate(editor: EditorContext) {
     74         val config = context.config.global.betterLocation
     75 
     76         editor.apply {
     77             // SCVSLocationUpdate
     78             edit(1) {
     79                 if (config.spoofLocation.get()) {
     80                     randomWalking.updatePosition()
     81                     remove(1)
     82                     remove(2)
     83                     addFixed32(1, getLat().toFloat()) // lat
     84                     addFixed32(2, getLong().toFloat()) // lng
     85                 }
     86 
     87                 if (config.alwaysUpdateLocation.get()) {
     88                     remove(7)
     89                     addVarInt(7, System.currentTimeMillis()) // timestamp
     90                 }
     91             }
     92 
     93             if (context.config.global.betterLocation.suspendLocationUpdates.get()) {
     94                 remove(1)
     95             }
     96 
     97             // SCVSDeviceData
     98             edit(3) {
     99                 config.spoofBatteryLevel.getNullable()?.takeIf { it.isNotEmpty() }?.let {
    100                     val value = it.toIntOrNull()?.toFloat()?.div(100) ?: return@edit
    101                     remove(2)
    102                     addFixed32(2, value)
    103                     if (value == 100F) {
    104                         remove(3)
    105                         addVarInt(3, 1) // devicePluggedIn
    106                     }
    107                 }
    108 
    109                 if (config.spoofHeadphones.get()) {
    110                     remove(4)
    111                     addVarInt(4, 1) // headphoneOutput
    112                     remove(6)
    113                     addVarInt(6, 1) // isOtherAudioPlaying
    114                 }
    115 
    116                 edit(10) {
    117                     remove(1)
    118                     addVarInt(1, 4) // type = ALWAYS
    119                     remove(2)
    120                     addVarInt(2, 1) // precise = true
    121                 }
    122             }
    123         }
    124     }

    126     private fun onLocationEvent(protoReader: ProtoReader) {
    127         protoReader.eachBuffer(3, 1) {
    128             val userId = UUID(getFixed64(1, 1) ?: return@eachBuffer, getFixed64(1, 2) ?: return@eachBuffer).toString()
    129             val friendCluster = FriendLocation(
    130                 userId = userId,
    131                 latitude = Float.fromBits(getFixed32(4)).toDouble(),
    132                 longitude = Float.fromBits(getFixed32(5)).toDouble(),
    133                 lastUpdated = getVarInt(7, 2) ?: -1L,
    134                 locality = getString(10),
    135                 localityPieces = mutableListOf<String>().also {
    136                     forEach { index, wire ->
    137                         if (index != 11) return@forEach
    138                         it.add((wire.value as ByteArray).toString(Charsets.UTF_8) )
    139                     }
    140                 }
    141             )
    142 
    143             locationHistory[userId] = friendCluster
    144         }
    145     }

    152                 me.rhunk.snapenhance.bridge.location.FriendLocation().also {
    153                     it.username = friendInfo.mutableUsername ?: return@mapNotNull null

    156                     it.bitmojiSelfieId = friendInfo.bitmojiSelfieId
    157                     it.latitude = locationHistory.latitude
