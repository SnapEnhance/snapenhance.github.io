     53     fun distanceTo(other: FriendLocation): Double {
     54         val deltaLat = Math.toRadians(other.latitude - this.latitude)
     55         val deltaLong = Math.toRadians(other.longitude - this.longitude)
     56 
     57         val a = sin(deltaLat / 2) * sin(deltaLat / 2) +
     58                 cos(Math.toRadians(this.latitude)) * cos(Math.toRadians(other.latitude)) *
     59                 sin(deltaLong / 2) * sin(deltaLong / 2)
     60 
     61         return 6371 * 2 * atan2(sqrt(a), sqrt(1 - a))
     62     }

     92     private fun editClientUpdate(editor: EditorContext) {
     93         val config = context.config.global.betterLocation
     94 
     95         editor.apply {
     96             // SCVSLocationUpdate
     97             edit(1) {
     98                 if (config.spoofLocation.get()) {
     99                     randomWalking.updatePosition()
    100                     remove(1)
    101                     remove(2)
    102                     addFixed32(1, getLat().toFloat()) // lat
    103                     addFixed32(2, getLong().toFloat()) // lng
    104                 }
    105 
    106                 if (config.alwaysUpdateLocation.get()) {
    107                     remove(7)
    108                     addVarInt(7, System.currentTimeMillis()) // timestamp
    109                 }
    110             }
    111 
    112             if (context.config.global.betterLocation.suspendLocationUpdates.get()) {
    113                 remove(1)
    114             }
    115 
    116             // SCVSDeviceData
    117             edit(3) {
    118                 config.spoofBatteryLevel.getNullable()?.takeIf { it.isNotEmpty() }?.let {
    119                     val value = it.toIntOrNull()?.toFloat()?.div(100) ?: return@edit
    120                     remove(2)
    121                     addFixed32(2, value)
    122                     if (value == 100F) {
    123                         remove(3)
    124                         addVarInt(3, 1) // devicePluggedIn
    125                     }
    126                 }
    127 
    128                 if (config.spoofHeadphones.get()) {
    129                     remove(4)
    130                     addVarInt(4, 1) // headphoneOutput
    131                     remove(6)
    132                     addVarInt(6, 1) // isOtherAudioPlaying
    133                 }
    134 
    135                 edit(10) {
    136                     remove(1)
    137                     addVarInt(1, 4) // type = ALWAYS
    138                     remove(2)
    139                     addVarInt(2, 1) // precise = true
    140                 }
    141             }
    142         }

    145     private fun onLocationEvent(protoReader: ProtoReader) {
    146         protoReader.eachBuffer(3, 1) {
    147             val clusterId = UUID(getFixed64(1, 1) ?: return@eachBuffer, getFixed64(1, 2) ?: return@eachBuffer).toString()
    148 
    149             val latitude = getFixed32(4)?.let { Float.fromBits(it) }?.toDouble() ?: return@eachBuffer
    150             val longitude = getFixed32(5)?.let { Float.fromBits(it) }?.toDouble() ?: return@eachBuffer
    151 
    152             val locality = getString(10)
    153             val localityPieces = mutableListOf<String>().also {
    154                 forEach { index, wire ->
    155                     if (index != 11) return@forEach
    156                     it.add((wire.value as ByteArray).toString(Charsets.UTF_8) )
    157                 }
    158             }
    159 
    160             eachBuffer(7) friend@{
    161                 val userId = if (contains(1)) UUID(getFixed64(1, 1) ?: return@friend, getFixed64(1, 2) ?: return@friend).toString() else clusterId
    162                 val friendLocation = FriendLocation(
    163                     userId = userId,
    164                     latitude = latitude,
    165                     longitude = longitude,
    166                     lastUpdated = getVarInt(2) ?: -1L,
    167                     locality = locality,
    168                     localityPieces = localityPieces,
    169                     batteryLevel = getFixed32(13)?.let { Float.fromBits(it) } ?: -1F,
    170                 )
    171 
    172                 locationHistory[userId] = friendLocation
    173             }
    174         }
    175     }
