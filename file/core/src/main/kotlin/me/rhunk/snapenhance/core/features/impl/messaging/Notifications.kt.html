    339         fun setNotificationText(text: String) {
    340             val includeUsername = context.database.getDMOtherParticipant(conversationId) == null
    341             cachedMessages.computeIfAbsent(conversationId) {
    342                 sortedMapOf()
    343             }[orderKey] = if (includeUsername) "$senderUsername: $text" else text
    344         }
    345 
    346         if (config.mediaPreview.get().contains(contentType.name)) {
    347             MessageDecoder.decode(message.messageContent!!).firstOrNull()?.also { media ->
    348                 runCatching {
    349                     media.openStream { mediaStream, length ->
    350                         if (mediaStream == null || length > 25 * 1024 * 1024) {
    351                             context.log.error("Failed to open media stream or media is too large")
    352                             sendNotification(message, data, true)
    353                             return@openStream
    354                         }
    355                         val downloadedMedias = mutableMapOf<SplitMediaAssetType, ByteArray>()
    356 
    357                         MediaDownloaderHelper.getSplitElements(mediaStream) { type, inputStream ->
    358                             downloadedMedias[type] = inputStream.readBytes()
    359                         }
    360 
    361                         val originalMedia = downloadedMedias[SplitMediaAssetType.ORIGINAL]!!
    362                         var bitmapPreview = PreviewUtils.createPreview(originalMedia, FileType.fromByteArray(originalMedia).isVideo)!!
    363 
    364                         downloadedMedias[SplitMediaAssetType.OVERLAY]?.let {
    365                             bitmapPreview = PreviewUtils.mergeBitmapOverlay(bitmapPreview, BitmapFactory.decodeByteArray(it, 0, it.size))
    366                         }
    367 
    368                         val notificationBuilder = newNotificationBuilder(data.notification).apply {
    369                             setLargeIcon(bitmapPreview)
    370                             style = Notification.BigPictureStyle().bigPicture(bitmapPreview).bigLargeIcon(null as Bitmap?)
    371                         }
    372                         if (config.mediaCaption.get()) {
    373                             message.serialize()?.let {
    374                                 notificationBuilder.setContentText(it)
    375                             }
    376                         }
    377 
    378                         sendNotification(message, data.copy(notification = notificationBuilder.build()), true)
    379                     }
    380                     return
    381                 }.onFailure {
    382                     context.log.error("Failed to send preview notification", it)
    383                     sendNotification(message, data, true)
    384                     return
    385                 }
    386             }
    387         }
    388 
    389         if (config.chatPreview.get()) {
    390             var isChatMessage = contentType == ContentType.CHAT
    391             var serializedMessage = if (isChatMessage) {
    392                 message.serialize() ?: "[Failed to parse message]"
    393             } else {
    394                 "[${context.translation.getCategory("content_type")[contentType.name]}]${
    395                     if (config.mediaCaption.get()) {
    396                         message.serialize() ?: ""
    397                     } else ""
    398                 }"
    399             }
    400 
    401             if (contentType == ContentType.NOTE && context.config.experimental.betterTranscript.takeIf { it.globalState == true }?.enhancedTranscriptInNotifications?.get() == true) {
    402                 val transcriptApi = context.feature(BetterTranscript::class).transcriptApi
    403                 MessageDecoder.decode(message.messageContent!!).firstOrNull { it.type == AttachmentType.NOTE  }?.also { media ->
    404                     runCatching {
    405                         media.openStream { mediaStream, length ->
    406                             if (mediaStream == null || length > 25 * 1024 * 1024) {
    407                                 context.log.error("Failed to open media stream or media is too large")
    408                                 return@openStream
    409                             }
    410                             val text = transcriptApi.transcribe(
    411                                 mediaStream.readBytes().toRequestBody(),
    412                                 lang = context.config.experimental.betterTranscript.preferredTranscriptionLang.getNullable()?.takeIf { it.isNotBlank() }
    413                             )?.takeIf { it.isNotBlank() } ?: return@openStream

    415                             isChatMessage = true
    416                         }
    417                     }.onFailure {
    418                         context.log.error("Failed to transcribe message", it)
    419                     }
    420                 }
    421             }
    422 
    423             if (isChatMessage || config.stackedMediaMessages.get()) {
    424                 setNotificationText(serializedMessage)
    425             } else {
    426                 sendNotification(message, data, true)
    427                 return
    428             }
    429             computeMessages()
    430         }
    431 
    432         sendNotification(message, data, false)
    433     }

    435     private fun canSendNotification(type: String): Boolean {
    436         val formattedMessageType = type.replaceFirst("mischief_", "")
    437             .replaceFirst("group_your_", "group_")
    438             .replaceFirst("group_other_", "group_")
    439 
    440         return context.config.messaging.notificationBlacklist.get().mapNotNull {
    441             NotificationType.getByKey(it)
    442         }.none {
    443             it.isMatch(formattedMessageType)
    444         }.also {
    445             if (!it) context.log.debug("prevented notification of type $type")
    446         }
    447     }

    461 
    462             if (!canSendNotification(notificationType)) {
    463                 param.setResult(null)
    464                 return@hook
    465             }
    466 
    467             if (notificationType == "addfriend" && config.friendAddSource.get()) {
    468                 val userId = notificationData.notification.shortcutId?.split("|")?.lastOrNull() ?: return@hook
    469                 runBlocking {
    470                     var addSource: String? = null
    471                     withTimeoutOrNull(7000) {
    472                         while (true) {
    473                             addSource = context.feature(FriendMutationObserver::class).getFriendAddSource(userId)
    474                             if (addSource != null) break

    476                         }
    477                     }
    478                     setNotificationText(notificationData.notification, addSource ?: return@runBlocking)
    479                 }
    480                 return@hook
    481             }
    482 
    483             if (!config.chatPreview.get() && config.mediaPreview.isEmpty()) return@hook
    484             if (notificationType.endsWith("typing")) return@hook
    485 
    486             val serverMessageId = extras.getString("message_id") ?: return@hook
    487             val conversationId = extras.getString("conversation_id").also { id ->
    488                 sentNotifications.computeIfAbsent(notificationData.id) { id ?: "" }
    489             } ?: return@hook
    490 
    491             param.setResult(null)
    492             val conversationManager = context.feature(Messaging::class).conversationManager ?: return@hook
    493 
    494             context.coroutineScope.launch(coroutineDispatcher) {
    495                 suspendCoroutine { continuation ->
    496                     conversationManager.fetchMessageByServerId(conversationId, serverMessageId.toLong(), onSuccess = {
    497                         if (it.senderId.toString() == context.database.myUserId) {
    498                             param.invokeOriginal()
    499                             continuation.resumeWith(Result.success(Unit))
    500                             return@fetchMessageByServerId
    501                         }
    502                         onMessageReceived(notificationData, notificationType, it)
    503                         continuation.resumeWith(Result.success(Unit))
    504                     }, onError = {
    505                         context.log.error("Failed to fetch message id ${serverMessageId}: $it")
    506                         param.invokeOriginal()
    507                         continuation.resumeWith(Result.success(Unit))
    508                     })
    509                 }
    510             }
    511         }

    517

    519                 sentNotifications[notificationId]?.let {
    520                     cachedMessages[it]?.clear()
    521                 }
    522                 sentNotifications.remove(notificationId)
    523             }
    524 
    525             notificationManager.activeNotifications.let { notifications ->
    526                 if (notifications.all { it.notification.flags and Notification.FLAG_GROUP_SUMMARY != 0 }) {
    527                     notifications.forEach { param.invokeOriginal(arrayOf(it.tag, it.id, it.user)) }
    528                 }
    529             }
    530         }
    531 
    532         findClass("com.google.firebase.messaging.FirebaseMessagingService").run {
    533             methods.first { it.declaringClass == this && it.returnType == Void::class.javaPrimitiveType && it.parameterCount == 1 && it.parameterTypes[0] == Intent::class.java }
    534                 .hook(HookStage.BEFORE) { param ->

    536                     val messageType = intent.getStringExtra("type") ?: return@hook
    537 
    538                     context.log.debug("received message type: $messageType")

    541                         param.setResult(null)
