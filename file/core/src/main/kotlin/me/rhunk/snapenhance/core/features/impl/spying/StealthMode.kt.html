     11 class StealthMode : MessagingRuleFeature("StealthMode", MessagingRuleType.STEALTH) {
     12     private val displayedMessageQueue = CopyOnWriteArraySet<Long>()
     13     private val snapInteractionQueue = CopyOnWriteArraySet<Long>()
     14 
     15     fun addDisplayedMessageException(clientMessageId: Long) {
     16         displayedMessageQueue.add(clientMessageId)
     17     }
     18 
     19     fun addSnapInteractionException(messageId: Long) {
     20         snapInteractionQueue.add(messageId)
     21     }
     22 
     23 
     24     override fun init() {
     25         val isConversationInStealthMode: (SnapUUID) -> Boolean = { canUseRule(it.toString()) }
     26 
     27         arrayOf("mediaMessagesDisplayed", "displayedMessages").forEach { methodName: String ->
     28             context.classCache.conversationManager.hook(methodName, HookStage.BEFORE) { param ->
     29                 if (displayedMessageQueue.removeIf { param.arg<Long>(1) == it }) return@hook
     30                 if (isConversationInStealthMode(SnapUUID(param.arg(0)))) {
     31                     param.setResult(null)
     32                 }
     33             }
     34         }
     35 
     36         context.event.subscribe(OnSnapInteractionEvent::class) { event ->
     37             if (snapInteractionQueue.removeIf { event.messageId == it }) return@subscribe
     38             if (isConversationInStealthMode(event.conversationId)) {
     39                 event.canceled = true
     40             }
     41         }
     42     }
