     54     private fun isViewGroupButtonMenuContainer(viewGroup: ViewGroup): Boolean {
     55         if (viewGroup !is LinearLayout) return false
     56         val children = ArrayList<View>()
     57         for (i in 0 until viewGroup.getChildCount())
     58             children.add(viewGroup.getChildAt(i))
     59         return if (children.any { view: View? -> view !is LinearLayout })
     60             false
     61         else children.map { view: View -> view as LinearLayout }
     62             .any { linearLayout: LinearLayout ->
     63                 val viewChildren = ArrayList<View>()
     64                 for (i in 0 until linearLayout.childCount) viewChildren.add(
     65                     linearLayout.getChildAt(
     66                         i
     67                     )
     68                 )
     69                 viewChildren.any { viewChild: View ->
     70                     viewChild.javaClass.name.endsWith("SnapFontTextView")
     71                 }
     72             }
     73     }
