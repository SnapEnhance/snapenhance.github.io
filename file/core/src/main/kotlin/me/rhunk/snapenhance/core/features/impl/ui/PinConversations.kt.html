     14 class PinConversations : MessagingRuleFeature("PinConversations", MessagingRuleType.PIN_CONVERSATION) {
     15     override fun init() {
     16         if (!context.config.messaging.unlimitedConversationPinning.get()) return
     17 
     18         context.classCache.feedManager.hook("setPinnedConversationStatus", HookStage.BEFORE) { param ->
     19             val conversationUUID = SnapUUID(param.arg(0))
     20             val isPinned = param.arg<Any>(1).toString() == "PINNED"
     21             setState(conversationUUID.toString(), isPinned)
     22             val callback = param.arg<Any>(2)
     23             mutableSetOf<() -> Unit>().apply {
     24                 addAll(Hooker.ephemeralHookObjectMethod(callback::class.java, callback,"onSuccess", HookStage.BEFORE) {
     25                     forEach { it() }
     26                 })
     27                 addAll(Hooker.ephemeralHookObjectMethod(callback::class.java, callback,"onError", HookStage.BEFORE) { methodParam ->
     28                     methodParam.setResult(null)
     29                     callback::class.java.getDeclaredMethod("onSuccess").invoke(callback)
     30                 })
     31             }
     32         }
     33 
     34         context.classCache.conversation.hookConstructor(HookStage.AFTER) { param ->
     35             val instance = param.thisObject<Any>()
     36             val conversationUUID = SnapUUID(instance.getObjectField("mConversationId"))
     37             if (getState(conversationUUID.toString())) {
     38                 instance.setObjectField("mPinnedTimestampMs", 1L)
     39             }
     40         }
     41 
     42         context.classCache.feedEntry.hookConstructor(HookStage.AFTER) { param ->
     43             val instance = param.thisObject<Any>()
     44             val conversationUUID = SnapUUID(instance.getObjectField("mConversationId") ?: return@hookConstructor)
     45             val isPinned = getState(conversationUUID.toString())
     46             if (isPinned) {
     47                 instance.setObjectField("mPinnedTimestampMs", 1L)
     48             }
     49         }
     50     }
     51 
     52     override fun getRuleState() = RuleState.WHITELIST
