     81     fun runOnUiThread(runnable: () -> Unit) {
     82         if (Looper.myLooper() == Looper.getMainLooper()) {
     83             runnable()
     84             return
     85         }
     86         Handler(Looper.getMainLooper()).post {
     87             runCatching(runnable).onFailure {
     88                 CoreLogger.xposedLog("UI thread runnable failed", it)
     89             }
     90         }
     91     }

     93     fun executeAsync(runnable: suspend ModContext.() -> Unit) {
     94         coroutineScope.launch {

    116     fun softRestartApp(saveSettings: Boolean = false) {
    117         if (saveSettings) {
    118             _config.writeConfig()
    119         }
    120         val intent: Intent? = androidContext.packageManager.getLaunchIntentForPackage(
    121             Constants.SNAPCHAT_PACKAGE_NAME
    122         )
    123         intent?.let {
    124             val mainIntent = Intent.makeRestartActivityTask(intent.component)
    125             androidContext.startActivity(mainIntent)
    126         }
    127         exitProcess(1)
    128     }
