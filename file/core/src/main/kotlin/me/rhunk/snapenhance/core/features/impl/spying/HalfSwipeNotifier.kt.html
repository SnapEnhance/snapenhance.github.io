     38     override fun init() {
     39         if (context.config.messaging.halfSwipeNotifier.globalState != true) return
     40         lateinit var presenceService: Any
     41 
     42         findClass("com.snapchat.talkcorev3.PresenceService\$CppProxy").hookConstructor(HookStage.AFTER) {
     43             presenceService = it.thisObject()
     44         }
     45 
     46         context.mappings.useMapper(CallbackMapper::class) {
     47             callbacks.getClass("PresenceServiceDelegate")?.hook("notifyActiveConversationsChanged", HookStage.BEFORE) {
     48                 val activeConversations = presenceService::class.java.methods.find { it.name == "getActiveConversations" }?.invoke(presenceService) as? Map<*, *> ?: return@hook // conversationId, conversationInfo (this.mPeekingParticipants)
     49 
     50                 if (activeConversations.isEmpty()) {
     51                     peekingConversations.forEach {
     52                         val conversationId = it.key
     53                         val peekingParticipantsIds = it.value
     54                         peekingParticipantsIds.forEach { userId ->
     55                             endPeeking(conversationId, userId)
     56                         }
     57                     }
     58                     peekingConversations.clear()
     59                     return@hook
     60                 }
     61 
     62                 activeConversations.forEach { (conversationId, conversationInfo) ->
     63                     val peekingParticipantsIds = (conversationInfo?.getObjectField("mPeekingParticipants") as? List<*>)?.map { it.toString() } ?: return@forEach
     64                     val cachedPeekingParticipantsIds = peekingConversations[conversationId] ?: emptyList()
     65 
     66                     val newPeekingParticipantsIds = peekingParticipantsIds - cachedPeekingParticipantsIds.toSet()
     67                     val exitedPeekingParticipantsIds = cachedPeekingParticipantsIds - peekingParticipantsIds.toSet()
     68 
     69                     newPeekingParticipantsIds.forEach { userId ->
     70                         startPeeking(conversationId.toString(), userId)
     71                     }
     72 
     73                     exitedPeekingParticipantsIds.forEach { userId ->
     74                         endPeeking(conversationId.toString(), userId)
     75                     }
     76                     peekingConversations[conversationId.toString()] = peekingParticipantsIds
     77                 }
     78             }
     79         }
     80     }

     82     private fun startPeeking(conversationId: String, userId: String) {
     83         startPeekingTimestamps[conversationId + userId] = System.currentTimeMillis()
     84     }
     85 
     86     private fun endPeeking(conversationId: String, userId: String) {
     87         startPeekingTimestamps[conversationId + userId]?.let { startPeekingTimestamp ->
     88             val peekingDuration = (System.currentTimeMillis() - startPeekingTimestamp).milliseconds.inWholeSeconds
     89             val minDuration = context.config.messaging.halfSwipeNotifier.minDuration.get().toLong()
     90             val maxDuration = context.config.messaging.halfSwipeNotifier.maxDuration.get().toLong()
     91 
     92             if (minDuration > peekingDuration || maxDuration < peekingDuration) return
     93 
     94             val groupName = context.database.getFeedEntryByConversationId(conversationId)?.feedDisplayName
     95             val friendInfo = context.database.getFriendInfo(userId) ?: return
     96 
     97             Notification.Builder(context.androidContext, channelId)
     98                 .setContentTitle(groupName ?: friendInfo.displayName ?: friendInfo.mutableUsername)
     99                 .setContentText(if (groupName != null) {
    100                     translation.format("notification_content_group",
    101                         "friend" to (friendInfo.displayName ?: friendInfo.mutableUsername).toString(),
    102                         "group" to groupName,
    103                         "duration" to peekingDuration.toString()
    104                     )
    105                 } else {
    106                     translation.format("notification_content_dm",
    107                         "friend" to (friendInfo.displayName ?: friendInfo.mutableUsername).toString(),
    108                         "duration" to peekingDuration.toString()
    109                     )
    110                 })
    111                 .setContentIntent(
    112                     context.androidContext.packageManager.getLaunchIntentForPackage(
    113                         Constants.SNAPCHAT_PACKAGE_NAME
    114                     )?.let {
    115                         PendingIntent.getActivity(
    116                             context.androidContext,
    117                             0, it, PendingIntent.FLAG_IMMUTABLE
    118                         )
    119                     }
    120                 )
    121                 .setWhen(System.currentTimeMillis())
    122                 .setShowWhen(true)
    123                 .setAutoCancel(true)
    124                 .setSmallIcon(svgEyeDrawable)
    125                 .build()
    126                 .let { notification ->
    127                     notificationManager.notify(System.nanoTime().toInt(), notification)
    128                 }
    129         }
    130     }
