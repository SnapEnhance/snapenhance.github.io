     27 class AutoMarkAsRead : Feature("Auto Mark As Read") {
     28     val canMarkConversationAsRead by lazy { context.config.messaging.autoMarkAsRead.get().contains("conversation_read") }
     29 
     30     fun markConversationsAsRead(conversationIds: List<String>) {
     31         conversationIds.forEach { conversationId ->
     32             val lastClientMessageId = context.database.getMessagesFromConversationId(conversationId, 1)?.firstOrNull()?.clientMessageId?.toLong() ?: Long.MAX_VALUE
     33             context.feature(StealthMode::class).addDisplayedMessageException(lastClientMessageId)
     34             context.feature(Messaging::class).conversationManager?.displayedMessages(conversationId, lastClientMessageId) {
     35                 if (it != null) {
     36                     context.log.warn("Failed to mark message $lastClientMessageId as read in conversation $conversationId")
     37                 }
     38             }
     39         }
     40     }
     41 
     42     suspend fun markSnapAsSeen(conversationId: String, clientMessageId: Long): String? {
     43         return suspendCoroutine { continuation ->
     44             context.feature(Messaging::class).conversationManager?.updateMessage(conversationId, clientMessageId, MessageUpdate.READ) {
     45                 continuation.resume(it)
     46                 if (it != null && it != "DUPLICATEREQUEST") {
     47                     context.log.error("Error marking message as read $it")
     48                 }
     49             }
     50         }
     51     }
     52 
     53     fun markSnapsAsSeen(conversationId: String) {
     54         val messaging = context.feature(Messaging::class)
     55         val messageIds = messaging.getFeedCachedMessageIds(conversationId)?.takeIf { it.isNotEmpty() } ?: run {
     56             context.inAppOverlay.showStatusToast(
     57                 Icons.Default.WarningAmber,
     58                 context.translation["mark_as_seen.no_unseen_snaps_toast"]
     59             )
     60             return
     61         }
     62 
     63         var job: Job? = null
     64         val dialog = ViewAppearanceHelper.newAlertDialogBuilder(context.mainActivity)
     65             .setTitle("Processing...")
     66             .setView(ProgressBar(context.mainActivity).apply {
     67                 setPadding(10, 10, 10, 10)
     68             })
     69             .setOnDismissListener { job?.cancel() }
     70             .show()
     71 
     72         context.coroutineScope.launch(Dispatchers.IO) {
     73             messageIds.forEach { messageId ->
     74                 markSnapAsSeen(conversationId, messageId)
     75                 delay(Random.nextLong(20, 60))
     76                 context.runOnUiThread {
     77                     dialog.setTitle("Processing... (${messageIds.indexOf(messageId) + 1}/${messageIds.size})")
     78                 }
     79             }
     80         }.also { job = it }.invokeOnCompletion {
     81             context.runOnUiThread {
     82                 dialog.dismiss()
     83             }
     84         }
     85     }

     87     override fun init() {
     88         val config by context.config.messaging.autoMarkAsRead
     89         if (config.isEmpty()) return
     90 
     91         if (config.contains("save_snap_in_chat")) {
     92             var lastInteractedSnapClientMessageId = -1L
     93 
     94             context.event.subscribe(OnSnapInteractionEvent::class) { event ->
     95                 lastInteractedSnapClientMessageId = event.messageId
     96             }
     97 
     98             context.classCache.conversationManager.hook("updateMessage", HookStage.BEFORE) { param ->
     99                 if (param.arg<Any>(2).toString() != "SAVE") return@hook
    100 
    101                 val clientMessageId = param.arg<Long>(1)
    102                 if (lastInteractedSnapClientMessageId != clientMessageId) return@hook
    103 
    104                 val conversationId = SnapUUID(param.arg(0))
    105 
    106                 param.setResult(null)
    107 
    108                 val snapManager = context.feature(Messaging::class).snapManager ?: return@hook
    109                 val stealthMode = context.feature(StealthMode::class)
    110 
    111                 // ignore non-stealth mode conversations
    112                 if (!stealthMode.canUseRule(conversationId.toString())) return@hook
    113 
    114                 stealthMode.addSnapInteractionException(clientMessageId)
    115 
    116                 val onSnapInteraction = snapManager.javaClass.methods.firstOrNull { it.name == "onSnapInteraction" } ?: return@hook
    117 
    118                 context.mappings.useMapper(CallbackMapper::class) {
    119                     onSnapInteraction.invoke(snapManager,
    120                         findClass("com.snapchat.client.messaging.SnapInteractionType").enumConstants!!.first { it.toString() == "VIEWING_INITIATED" },
    121                         conversationId.instanceNonNull(),
    122                         clientMessageId,
    123                         CallbackBuilder(callbacks.getClass("SnapInteractionCallback")!!)
    124                             .override("onSuccess") {
    125                                 param.invokeOriginal()
    126                                 stealthMode.addSnapInteractionException(clientMessageId)
    127                             }
    128                             .override("onError") {
    129                                 context.log.verbose("error ${it.arg<Any>(0)}")
    130                             }
    131                             .build()
    132                     )
    133                 }
    134             }
    135         }
    136 
    137         context.event.subscribe(SendMessageWithContentEvent::class) { event ->
    138             event.addCallbackResult("onSuccess") {
    139                 if (canMarkConversationAsRead) {
    140                     markConversationsAsRead(event.destinations.conversations?.map { it.toString() } ?: return@addCallbackResult)
    141                 }
    142 
    143                 if (config.contains("snap_reply")) {
    144                     val quotedMessageId = event.messageContent.instanceNonNull().getObjectFieldOrNull("mQuotedMessageId") as? Long ?: return@addCallbackResult
    145                     val message = context.database.getConversationMessageFromId(quotedMessageId) ?: return@addCallbackResult
    146 
    147                     if (message.contentType == ContentType.SNAP.id) {
    148                         context.coroutineScope.launch {
    149                             markSnapAsSeen(event.destinations.conversations?.firstOrNull()?.toString() ?: return@launch, quotedMessageId)
    150                         }
    151                     }
    152                 }
    153             }
    154         }
    155     }
