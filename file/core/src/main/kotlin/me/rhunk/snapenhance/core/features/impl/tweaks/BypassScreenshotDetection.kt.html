     11 class BypassScreenshotDetection : Feature("BypassScreenshotDetection") {
     12     override fun init() {
     13         if (!context.config.messaging.bypassScreenshotDetection.get()) return
     14         Activity::class.java.hook("registerScreenCaptureCallback", HookStage.BEFORE) { param ->
     15             param.setResult(null)
     16         }
     17         ContentResolver::class.java.methods.first {
     18             it.name == "registerContentObserver" &&
     19             it.parameterTypes.contentEquals(arrayOf(android.net.Uri::class.java, Boolean::class.javaPrimitiveType, ContentObserver::class.java))
     20         }.hook(HookStage.BEFORE) { param ->
     21             val uri = param.arg<Uri>(0)
     22             if (uri.host != "media") return@hook
     23             param.setResult(null)
     24         }
     25     }
