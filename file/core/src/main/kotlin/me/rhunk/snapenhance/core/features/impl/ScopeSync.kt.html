     11 class ScopeSync : Feature("Scope Sync") {
     12     companion object {
     13         private const val DELAY_BEFORE_SYNC = 2000L
     14     }
     15 
     16     private val updateJobs = mutableMapOf<String, Job>()
     17 
     18     private fun sync(conversationId: String) {
     19         context.database.getDMOtherParticipant(conversationId)?.also { participant ->
     20             context.bridgeClient.triggerSync(SocialScope.FRIEND, participant)
     21         } ?: run {
     22             context.bridgeClient.triggerSync(SocialScope.GROUP, conversationId)
     23         }
     24     }
     25 
     26     override fun init() {
     27         context.event.subscribe(SendMessageWithContentEvent::class) { event ->
     28             if (event.messageContent.contentType != ContentType.SNAP) return@subscribe
     29 
     30             event.addCallbackResult("onSuccess") {
     31                 event.destinations.conversations!!.map { it.toString() }.forEach { conversationId ->
     32                     updateJobs[conversationId]?.also { it.cancel() }
     33 
     34                     updateJobs[conversationId] = (context.coroutineScope.launch {
     35                         delay(DELAY_BEFORE_SYNC)
     36                         sync(conversationId)
     37                     })
     38                 }
     39             }
     40         }
     41     }
