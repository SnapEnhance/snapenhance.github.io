     21 fun getChatInputBar(event: AddViewEvent): Lazy<ViewGroup?>? {
     22     if (!event.parent.javaClass.name.endsWith("ChatInputLayout") || !event.viewClassName.endsWith("ViewSwitcher")) return null
     23 
     24     return lazy {
     25         // get the first linear layout in the view switcher
     26         val firstLinearLayout = (event.view as ViewGroup).children()
     27             .firstOrNull { it is LinearLayout } as? ViewGroup ?: return@lazy null
     28         // get the first linear layout with at least 3 children
     29         firstLinearLayout.children()
     30             .firstOrNull { v -> v is LinearLayout && v.childCount > 2 } as? LinearLayout
     31             ?: return@lazy null
     32     }
     33 }

     44     private fun hideStorySection(event: AddViewEvent) {
     45         val parent = event.parent
     46         parent.visibility = View.GONE
     47         val marginLayoutParams = parent.layoutParams as MarginLayoutParams
     48         marginLayoutParams.setMargins(-99999, -99999, -99999, -99999)
     49         event.canceled = true

     52     private fun hideView(view: View) {
     53         view.apply {
     54             visibility = View.GONE
     55             post {
     56                 isEnabled = false
     57                 visibility = View.GONE
     58                 setWillNotDraw(true)
     59             }
     60             addOnLayoutChangeListener { view, _, _, _, _, _, _, _, _ ->
     61                 view.post { view.visibility = View.GONE }
     62             }
     63         }
