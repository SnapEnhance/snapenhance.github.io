     28     private fun hideStorySection(event: AddViewEvent) {
     29         val parent = event.parent
     30         parent.visibility = View.GONE
     31         val marginLayoutParams = parent.layoutParams as MarginLayoutParams
     32         marginLayoutParams.setMargins(-99999, -99999, -99999, -99999)
     33         event.canceled = true
     34     }

     36     private fun hideView(view: View) {
     37         view.apply {
     38             visibility = View.GONE
     39             post {
     40                 isEnabled = false
     41                 visibility = View.GONE
     42                 setWillNotDraw(true)
     43             }
     44             addOnLayoutChangeListener { view, _, _, _, _, _, _, _, _ ->
     45                 view.post { view.visibility = View.GONE }
     46             }
     47         }

    125                         width = 0

    158                 }
    159             }
    160 
    161             if (

    163                 (viewId == chatNoteRecordButton && hiddenElements.contains("hide_voice_record_button")) ||

    165                 (viewId == getId("chat_input_bar_sharing_drawer_button", "id") && hiddenElements.contains("hide_live_location_share_button")) ||
    166                 (viewId == callButtonsStub && hiddenElements.contains("hide_chat_call_buttons"))
