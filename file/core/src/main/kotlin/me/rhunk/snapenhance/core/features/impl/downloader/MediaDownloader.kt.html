    193     private fun handleLocalReferences(path: String) = runBlocking {
    194         Uri.parse(path).let { uri ->
    195             if (uri.scheme == "file" || uri.scheme == null) {
    196                 return@let suspendCoroutine<String> { continuation ->
    197                     context.httpServer.ensureServerStarted()?.let { server ->
    198                         val file = Paths.get(uri.path).toFile()
    199                         val url = server.putDownloadableContent(file.inputStream(), file.length())
    200                         continuation.resumeWith(Result.success(url))
    201                     } ?: run {
    202                         continuation.resumeWith(Result.failure(Exception("Failed to start http server")))
    203                     }
    204                 }
    205             }
    206             path
    207         }
    208     }

    395             fun prettyPrintTime(time: Long): String {
    396                 val seconds = time / 1000
    397                 val minutes = seconds / 60
    398                 val hours = minutes / 60
    399                 return "${(hours % 24).toString().padStart(2, '0')}:${(minutes % 60).toString().padStart(2, '0')}:${(seconds % 60).toString().padStart(2, '0')}"

    401

    404 
    405                 urlIndexes.firstOrNull { index -> index != -1 }?.let { validIndex ->
    406                     it.substring(validIndex)
    407                 } ?: "${RemoteMediaResolver.CF_ST_CDN_D}$it"
    408             }
    409

    506                         val mediaParamMap: ParamMap = operaLayerList.map { Layer(it) }.first().paramMap

    513                         val isVideo = mediaParamMap.containsKey("video_media_info_list")

    517                         )

    520                             mediaInfoMap[SplitMediaAssetType.OVERLAY] =
    521                                 MediaInfo(mediaParamMap["overlay_image_media_info"]!!)
    522                         }
    523 
    524                         val shouldAutoDownload = shouldAutoDownload()
    525 
    526                         if (shouldAutoDownload && lastSeenMediaInfoMap?.get(SplitMediaAssetType.ORIGINAL)?.uri == mediaInfoMap[SplitMediaAssetType.ORIGINAL]?.uri) return@onOperaViewStateCallback
    527 
    528                         lastSeenMapParams = mediaParamMap
    529                         lastSeenMediaInfoMap = mediaInfoMap
    530 
    531                         if (!shouldAutoDownload) {
    532                             return@onOperaViewStateCallback
    533                         }
    534 
    535                         context.executeAsync {
    536                             runCatching {
    537                                 handleOperaMedia(mediaParamMap, mediaInfoMap, false)
    538                             }.onFailure {
    539                                 context.log.error("Failed to handle opera media", it)

    541                             }
    542                         }
    543                     }
    544                 }
    545             }

    564                     creationTimestamp = message.creationTimestamp,
    565                 ).apply {
    566                     downloadInputMedias(
    567                         arrayOf(attachment.createInputMedia()!!)
    568                     )
    569                 }
    570             }.onFailure {
    571                 context.longToast(translations["failed_generic_toast"])
    572                 context.log.error("Failed to download", it)
    573             }
    574         }
    575     }

    581     @SuppressLint("SetTextI18n")
    582     private fun previewAttachment(
    583         attachment: DecodedAttachment
    584     ) {
    585         var previewBitmap: Bitmap? = null
    586         val previewCoroutine = context.coroutineScope.launch {
    587             runCatching {
    588                 attachment.openStream { attachmentStream, _ ->
    589                     val downloadedMediaList = mutableMapOf<SplitMediaAssetType, ByteArray>()
    590 
    591                     MediaDownloaderHelper.getSplitElements(attachmentStream!!) {
    592                             type, inputStream ->
    593                         downloadedMediaList[type] = inputStream.readBytes()
    594                     }
    595 
    596                     val originalMedia = downloadedMediaList[SplitMediaAssetType.ORIGINAL] ?: return@openStream
    597                     val overlay = downloadedMediaList[SplitMediaAssetType.OVERLAY]
    598 
    599                     var bitmap = PreviewUtils.createPreview(originalMedia, isVideo = FileType.fromByteArray(originalMedia).isVideo)
    600                         ?: throw Exception("preview is null")
    601 
    602                     overlay?.also {
    603                         bitmap = PreviewUtils.mergeBitmapOverlay(bitmap, BitmapFactory.decodeByteArray(it, 0, it.size))
    604                     }
    605 
    606                     previewBitmap = bitmap
    607                 }
    608             }.onFailure {
    609                 context.shortToast(translations["failed_to_create_preview_toast"])
    610                 context.log.error("Failed to create preview", it)
    611             }
    612         }
    613 
    614         with(ViewAppearanceHelper.newAlertDialogBuilder(context.mainActivity)) {
    615             val viewGroup = LinearLayout(context).apply {
    616                 layoutParams = MarginLayoutParams(
    617                     MarginLayoutParams.MATCH_PARENT,
    618                     MarginLayoutParams.MATCH_PARENT
    619                 )
    620                 gravity = Gravity.CENTER_HORIZONTAL or Gravity.CENTER_VERTICAL
    621                 addView(ProgressBar(context).apply {
    622                     isIndeterminate = true
    623                 })
    624             }
    625 
    626             setOnDismissListener {
    627                 previewCoroutine.cancel()
    628             }
    629 
    630             previewCoroutine.invokeOnCompletion { cause ->
    631                 if (previewCoroutine.isCancelled) return@invokeOnCompletion
    632                 runOnUiThread {
    633                     viewGroup.removeAllViews()
    634                     if (cause != null) {
    635                         viewGroup.addView(TextView(context).apply {
    636                             text =
    637                                 translations["failed_to_create_preview_toast"] + "\n" + cause.message
    638                             setPadding(30, 30, 30, 30)
    639                         })
    640                         return@runOnUiThread
    641                     }
    642 
    643                     viewGroup.addView(ImageView(context).apply {
    644                         setImageBitmap(previewBitmap)
    645                         layoutParams = LinearLayout.LayoutParams(
    646                             LinearLayout.LayoutParams.MATCH_PARENT,
    647                             LinearLayout.LayoutParams.MATCH_PARENT
    648                         )
    649                         adjustViewBounds = true
    650                     })
    651                 }
    652             }
    653 
    654             runOnUiThread {
    655                 show().apply {
    656                     setContentView(viewGroup)
    657                     window?.setLayout(
    658                         context.resources.displayMetrics.widthPixels,
    659                         context.resources.displayMetrics.heightPixels
    660                     )
    661                 }
    662             }
    663         }
    664     }

    666     @SuppressLint("SetTextI18n")
    667     fun downloadMessageId(messageId: Long, forceAllowDuplicate: Boolean = false, isPreview: Boolean = false, forceDownloadFirst: Boolean = false) {
    668         val messageLogger = context.feature(MessageLogger::class)
    669         val message = context.database.getConversationMessageFromId(messageId) ?: throw Exception("Message not found in database")
    670 
    671         val friendInfo = context.database.getFriendInfo(message.senderId!!) ?: throw Exception("Friend not found in database")
    672         val authorName = friendInfo.usernameForSorting!!
    673 
    674         val decodedAttachments = (
    675             messageLogger.takeIf { it.isEnabled }?.getMessageObject(message.clientConversationId!!, message.clientMessageId.toLong())?.let {
    676                 MessageDecoder.decode(it.getAsJsonObject("mMessageContent"))
    677             } ?: MessageDecoder.decode(
    678                 protoReader = ProtoReader(message.messageContent!!)
    679             ).toMutableList().apply {
    680                 val quotedMessage = message.quotedServerMessageId?.takeIf { it > 0 }?.let { quotedMessageId ->
    681                     context.database.getConversationServerMessage(message.clientConversationId!!, quotedMessageId)
    682                 } ?: return@apply
    683                 addAll(0, MessageDecoder.decode(
    684                     protoReader = ProtoReader(quotedMessage.messageContent ?: return@apply)
    685                 ))
    686             }
    687         ).toMutableList()
    688 
    689         context.feature(Messaging::class).conversationManager?.takeIf {
    690             decodedAttachments.isEmpty()
    691         }?.also { conversationManager ->
    692             runBlocking {
    693                 suspendCoroutine { continuation ->
    694                     conversationManager.fetchMessage(message.clientConversationId!!, message.clientMessageId.toLong(), onSuccess = { message ->
    695                         decodedAttachments.addAll(MessageDecoder.decode(message.messageContent!!))
    696                         continuation.resumeWith(Result.success(Unit))
    697                     }, onError = {
    698                         continuation.resumeWith(Result.success(Unit))
    699                     })
    700                 }
    701             }
    702         }
    703 
    704         if (decodedAttachments.isEmpty()) {
    705             context.shortToast(translations["no_attachments_toast"])
    706             return
    707         }
    708 
    709         if (!isPreview) {
    710             if (forceDownloadFirst ||
    711                 decodedAttachments.size == 1 ||
    712                 context.isMainActivityPaused // we can't show alert dialogs when it downloads from a notification, so it downloads the first one
    713             ) {
    714                 downloadMessageAttachments(friendInfo, message, authorName,
    715                     listOf(decodedAttachments.first()),
    716                     forceAllowDuplicate = forceAllowDuplicate
    717                 )
    718                 return
    719             }
    720 
    721             runOnUiThread {
    722                 ViewAppearanceHelper.newAlertDialogBuilder(context.mainActivity).apply {
    723                     val selectedAttachments = mutableListOf<Int>().apply {
    724                         addAll(decodedAttachments.indices)
    725                     }
    726                     setMultiChoiceItems(
    727                         decodedAttachments.mapIndexed { index, decodedAttachment ->
    728                             "${index + 1}: ${decodedAttachment.getInfo()}"
    729                         }.toTypedArray(),
    730                         decodedAttachments.map { true }.toBooleanArray()
    731                     ) { _, which, isChecked ->
    732                         if (isChecked) {
    733                             selectedAttachments.add(which)
    734                         } else if (selectedAttachments.contains(which)) {
    735                             selectedAttachments.remove(which)
    736                         }
    737                     }
    738                     setTitle(translations["select_attachments_title"])
    739                     setNegativeButton(this@MediaDownloader.context.translation["button.cancel"]) { dialog, _ -> dialog.dismiss() }
    740                     setPositiveButton(this@MediaDownloader.context.translation["button.download"]) { _, _ ->
    741                         downloadMessageAttachments(friendInfo, message, authorName, selectedAttachments.map { decodedAttachments[it] },
    742                             forceAllowDuplicate = forceAllowDuplicate
    743                         )
    744                     }
    745                 }.show()
    746             }
    747             return
    748         }
    749 
    750         if (decodedAttachments.size == 1) {
    751             previewAttachment(decodedAttachments.first())
    752             return
    753         }
    754 
    755         runOnUiThread {
    756             ViewAppearanceHelper.newAlertDialogBuilder(context.mainActivity).apply {
    757                 var selectedAttachment = 0
    758                 setSingleChoiceItems(
    759                     decodedAttachments.mapIndexed { index, decodedAttachment -> "${index + 1}: ${decodedAttachment.getInfo()}" }.toTypedArray(),
    760                     0
    761                 ) { _, which ->
    762                     selectedAttachment = which
    763                 }
    764                 setTitle(translations["select_attachments_title"])
    765                 setNegativeButton(this@MediaDownloader.context.translation["button.cancel"]) { dialog, _ -> dialog.dismiss() }
    766                 setPositiveButton(this@MediaDownloader.context.translation["chat_action_menu.preview_button"]) { _, _ ->
    767                     previewAttachment(decodedAttachments[selectedAttachment])
    768                 }
    769             }.show()
    770         }
    771     }
