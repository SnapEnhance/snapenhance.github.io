    195     private fun handleLocalReferences(path: String) = runBlocking {
    196         Uri.parse(path).let { uri ->
    197             if (uri.scheme == "file" || uri.scheme == null) {
    198                 return@let suspendCoroutine<String> { continuation ->
    199                     context.httpServer.ensureServerStarted()?.let { server ->
    200                         val file = Paths.get(uri.path).toFile()
    201                         val url = server.putDownloadableContent(file.inputStream(), file.length())
    202                         continuation.resumeWith(Result.success(url))
    203                     } ?: run {
    204                         continuation.resumeWith(Result.failure(Exception("Failed to start http server")))
    205                     }
    206                 }
    207             }
    208             path
    209         }
    210     }

    392             fun prettyPrintTime(time: Long): String {

    570     @SuppressLint("SetTextI18n")
    571     private fun previewAttachment(
    572         attachment: DecodedAttachment
    573     ) {
    574         var previewBitmap: Bitmap? = null
    575         val previewCoroutine = context.coroutineScope.launch {
    576             runCatching {
    577                 attachment.openStream { attachmentStream, _ ->
    578                     val downloadedMediaList = mutableMapOf<SplitMediaAssetType, ByteArray>()
    579 
    580                     MediaDownloaderHelper.getSplitElements(attachmentStream!!) {
    581                             type, inputStream ->
    582                         downloadedMediaList[type] = inputStream.readBytes()
    583                     }
    584 
    585                     val originalMedia = downloadedMediaList[SplitMediaAssetType.ORIGINAL] ?: return@openStream
    586                     val overlay = downloadedMediaList[SplitMediaAssetType.OVERLAY]
    587 
    588                     var bitmap = PreviewUtils.createPreview(originalMedia, isVideo = FileType.fromByteArray(originalMedia).isVideo)
    589                         ?: throw Exception("preview is null")
    590 
    591                     overlay?.also {
    592                         bitmap = PreviewUtils.mergeBitmapOverlay(bitmap, BitmapFactory.decodeByteArray(it, 0, it.size))
    593                     }
    594 
    595                     previewBitmap = bitmap
    596                 }
    597             }.onFailure {
    598                 context.shortToast(translations["failed_to_create_preview_toast"])
    599                 context.log.error("Failed to create preview", it)
    600             }
    601         }
    602 
    603         with(ViewAppearanceHelper.newAlertDialogBuilder(context.mainActivity)) {
    604             val viewGroup = LinearLayout(context).apply {
    605                 layoutParams = MarginLayoutParams(
    606                     MarginLayoutParams.MATCH_PARENT,
    607                     MarginLayoutParams.MATCH_PARENT
    608                 )
    609                 gravity = Gravity.CENTER_HORIZONTAL or Gravity.CENTER_VERTICAL
    610                 addView(ProgressBar(context).apply {
    611                     isIndeterminate = true
    612                 })
    613             }
    614 
    615             setOnDismissListener {
    616                 previewCoroutine.cancel()
    617             }
    618 
    619             previewCoroutine.invokeOnCompletion { cause ->
    620                 if (previewCoroutine.isCancelled) return@invokeOnCompletion
    621                 runOnUiThread {
    622                     viewGroup.removeAllViews()
    623                     if (cause != null) {
    624                         viewGroup.addView(TextView(context).apply {
    625                             text =
    626                                 translations["failed_to_create_preview_toast"] + "\n" + cause.message
    627                             setPadding(30, 30, 30, 30)
    628                         })
    629                         return@runOnUiThread
    630                     }
    631 
    632                     viewGroup.addView(ImageView(context).apply {
    633                         setImageBitmap(previewBitmap)
    634                         layoutParams = LinearLayout.LayoutParams(
    635                             LinearLayout.LayoutParams.MATCH_PARENT,
    636                             LinearLayout.LayoutParams.MATCH_PARENT
    637                         )
    638                         adjustViewBounds = true
    639                     })
    640                 }
    641             }
    642 
    643             runOnUiThread {
    644                 show().apply {
    645                     setContentView(viewGroup)
    646                     window?.setLayout(
    647                         context.resources.displayMetrics.widthPixels,
    648                         context.resources.displayMetrics.heightPixels
    649                     )
    650                 }
    651             }
    652         }
    653     }

    655     @SuppressLint("SetTextI18n")
    656     fun downloadMessageId(messageId: Long, forceAllowDuplicate: Boolean = false, isPreview: Boolean = false) {
    657         val messageLogger = context.feature(MessageLogger::class)
    658         val message = context.database.getConversationMessageFromId(messageId) ?: throw Exception("Message not found in database")
    659 
    660         val friendInfo = context.database.getFriendInfo(message.senderId!!) ?: throw Exception("Friend not found in database")
    661         val authorName = friendInfo.usernameForSorting!!
    662 
    663         val decodedAttachments = (
    664             messageLogger.takeIf { it.isEnabled }?.getMessageObject(message.clientConversationId!!, message.clientMessageId.toLong())?.let {
    665                 MessageDecoder.decode(it.getAsJsonObject("mMessageContent"))
    666             } ?: MessageDecoder.decode(
    667                 protoReader = ProtoReader(message.messageContent!!)
    668             ).toMutableList().apply {
    669                 val quotedMessage = message.quotedServerMessageId?.takeIf { it > 0 }?.let { quotedMessageId ->
    670                     context.database.getConversationServerMessage(message.clientConversationId!!, quotedMessageId)
    671                 } ?: return@apply
    672                 addAll(0, MessageDecoder.decode(
    673                     protoReader = ProtoReader(quotedMessage.messageContent ?: return@apply)
    674                 ))
    675             }
    676         ).toMutableList()
    677 
    678         context.feature(Messaging::class).conversationManager?.takeIf {
    679             decodedAttachments.isEmpty()
    680         }?.also { conversationManager ->
    681             runBlocking {
    682                 suspendCoroutine { continuation ->
    683                     conversationManager.fetchMessage(message.clientConversationId!!, message.clientMessageId.toLong(), onSuccess = { message ->
    684                         decodedAttachments.addAll(MessageDecoder.decode(message.messageContent!!))
    685                         continuation.resumeWith(Result.success(Unit))
    686                     }, onError = {
    687                         continuation.resumeWith(Result.success(Unit))
    688                     })
    689                 }
    690             }
    691         }
    692 
    693         if (decodedAttachments.isEmpty()) {
    694             context.shortToast(translations["no_attachments_toast"])
    695             return
    696         }
    697 
    698         if (!isPreview) {
    699             if (decodedAttachments.size == 1 ||
    700                 context.isMainActivityPaused // we can't show alert dialogs when it downloads from a notification, so it downloads the first one
    701             ) {
    702                 downloadMessageAttachments(friendInfo, message, authorName,
    703                     listOf(decodedAttachments.first()),
    704                     forceAllowDuplicate = forceAllowDuplicate
    705                 )
    706                 return
    707             }
    708 
    709             runOnUiThread {
    710                 ViewAppearanceHelper.newAlertDialogBuilder(context.mainActivity).apply {
    711                     val selectedAttachments = mutableListOf<Int>().apply {
    712                         addAll(decodedAttachments.indices)
    713                     }
    714                     setMultiChoiceItems(
    715                         decodedAttachments.mapIndexed { index, decodedAttachment ->
    716                             "${index + 1}: ${decodedAttachment.getInfo()}"
    717                         }.toTypedArray(),
    718                         decodedAttachments.map { true }.toBooleanArray()
    719                     ) { _, which, isChecked ->
    720                         if (isChecked) {
    721                             selectedAttachments.add(which)
    722                         } else if (selectedAttachments.contains(which)) {
    723                             selectedAttachments.remove(which)
    724                         }
    725                     }
    726                     setTitle(translations["select_attachments_title"])
    727                     setNegativeButton(this@MediaDownloader.context.translation["button.cancel"]) { dialog, _ -> dialog.dismiss() }
    728                     setPositiveButton(this@MediaDownloader.context.translation["button.download"]) { _, _ ->
    729                         downloadMessageAttachments(friendInfo, message, authorName, selectedAttachments.map { decodedAttachments[it] },
    730                             forceAllowDuplicate = forceAllowDuplicate
    731                         )
    732                     }
    733                 }.show()
    734             }
    735             return
    736         }
    737 
    738         if (decodedAttachments.size == 1) {
    739             previewAttachment(decodedAttachments.first())
    740             return
    741         }
    742 
    743         runOnUiThread {
    744             ViewAppearanceHelper.newAlertDialogBuilder(context.mainActivity).apply {
    745                 var selectedAttachment = 0
    746                 setSingleChoiceItems(
    747                     decodedAttachments.mapIndexed { index, decodedAttachment -> "${index + 1}: ${decodedAttachment.getInfo()}" }.toTypedArray(),
    748                     0
    749                 ) { _, which ->
    750                     selectedAttachment = which
    751                 }
    752                 setTitle(translations["select_attachments_title"])
    753                 setNegativeButton(this@MediaDownloader.context.translation["button.cancel"]) { dialog, _ -> dialog.dismiss() }
    754                 setPositiveButton(this@MediaDownloader.context.translation["chat_action_menu.preview_button"]) { _, _ ->
    755                     previewAttachment(decodedAttachments[selectedAttachment])
    756                 }
    757             }.show()
    758         }
    759     }

    775     fun onMessageActionMenu(isPreviewMode: Boolean, forceAllowDuplicate: Boolean = false) {
