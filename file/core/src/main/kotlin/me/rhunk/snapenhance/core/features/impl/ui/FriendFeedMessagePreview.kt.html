     52                 message.messageContent
     53                     ?.let { ProtoReader(it) }
     54                     ?.followPath(4, 4)?.let {
     55                         if (hasE2EE) endToEndEncryption.decryptDatabaseMessage(message) else it
     56                     }
     57                     ?: return@mapNotNull null
     58 
     59             val contentType = ContentType.fromMessageContainer(messageContainer) ?: ContentType.fromId(message.contentType)
     60             val messageString = messageContainer.getBuffer().getMessageText(contentType) ?: "[${context.translation.getCategory("content_type")[contentType.name]}]"
     61 
     62             val friendName = friendNameCache.getOrPut(message.senderId ?: return@mapNotNull null) {
     63                 context.database.getFriendInfo(message.senderId ?: return@mapNotNull null)?.let { it.displayName?: it.mutableUsername } ?: "Unknown"
     64             }
     65             "$friendName: $messageString"
     66         }?.takeIf { it.isNotEmpty() }?.reversed()
     67 
     68         withContext(Dispatchers.Main) {
     69             messages?.also { messageCache[conversationId] = it } ?: run {
     70                 messageCache.remove(conversationId)
     71             }
     72             callback()
     73         }
     74     }
