    157     private fun filterFriends(friends: List<FriendInfo>, filter: Filter, nameFilter: String): List<FriendInfo> {
    158         val userIdBlacklist = arrayOf(
    159             context.database.myUserId,
    160             "b42f1f70-5a8b-4c53-8c25-34e7ec9e6781", // myai
    161             "84ee8839-3911-492d-8b94-72dd80f3713a", // teamsnapchat
    162         )
    163 
    164         return friends.filter { friend ->
    165             friend.userId !in userIdBlacklist && when (filter) {
    166                 Filter.ALL -> true
    167                 Filter.MY_FRIENDS -> friend.friendLinkType == FriendLinkType.MUTUAL.value && friend.addedTimestamp > 0
    168                 Filter.BLOCKED -> friend.friendLinkType == FriendLinkType.BLOCKED.value
    169                 Filter.REMOVED_ME -> friend.friendLinkType == FriendLinkType.OUTGOING.value && friend.addedTimestamp > 0 && friend.businessCategory == 0 // ignore followed accounts
    170                 Filter.SUGGESTED -> friend.friendLinkType == FriendLinkType.SUGGESTED.value
    171                 Filter.DELETED -> friend.friendLinkType == FriendLinkType.DELETED.value
    172                 Filter.BUSINESS_ACCOUNTS -> friend.businessCategory > 0
    173                 Filter.STREAKS -> friend.friendLinkType == FriendLinkType.MUTUAL.value && friend.addedTimestamp > 0 && friend.streakLength != 0
    174                 Filter.NON_STREAKS -> friend.friendLinkType == FriendLinkType.MUTUAL.value&& friend.addedTimestamp > 0 && friend.streakLength == 0
    175                 Filter.LOCATION_ON_MAP -> betterLocation.locationHistory.contains(friend.userId)
    176             } && nameFilter.takeIf { it.isNotBlank() }?.let { name ->
    177                 friend.mutableUsername?.contains(
    178                     name,
    179                     ignoreCase = true
    180                 ) == true || friend.displayName?.contains(name, ignoreCase = true) == true
    181             } ?: true
    182         }
    183     }
