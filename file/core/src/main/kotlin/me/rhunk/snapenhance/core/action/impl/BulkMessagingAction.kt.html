    153     private fun filterFriends(friends: List<FriendInfo>, filter: Filter, nameFilter: String): List<FriendInfo> {
    154         val userIdBlacklist = arrayOf(
    155             context.database.myUserId,
    156             "b42f1f70-5a8b-4c53-8c25-34e7ec9e6781", // myai
    157             "84ee8839-3911-492d-8b94-72dd80f3713a", // teamsnapchat
    158         )
    159         return friends.filter { friend ->
    160             friend.userId !in userIdBlacklist && when (filter) {
    161                 Filter.ALL -> true
    162                 Filter.MY_FRIENDS -> friend.friendLinkType == FriendLinkType.MUTUAL.value && friend.addedTimestamp > 0
    163                 Filter.BLOCKED -> friend.friendLinkType == FriendLinkType.BLOCKED.value
    164                 Filter.REMOVED_ME -> friend.friendLinkType == FriendLinkType.OUTGOING.value && friend.addedTimestamp > 0 && friend.businessCategory == 0 // ignore followed accounts
    165                 Filter.SUGGESTED -> friend.friendLinkType == FriendLinkType.SUGGESTED.value
    166                 Filter.DELETED -> friend.friendLinkType == FriendLinkType.DELETED.value
    167                 Filter.BUSINESS_ACCOUNTS -> friend.businessCategory > 0
    168                 Filter.STREAKS -> friend.friendLinkType == FriendLinkType.MUTUAL.value && friend.addedTimestamp > 0 && friend.streakLength != 0
    169                 Filter.NON_STREAKS -> friend.friendLinkType == FriendLinkType.MUTUAL.value&& friend.addedTimestamp > 0 && friend.streakLength == 0
    170             } && nameFilter.takeIf { it.isNotBlank() }?.let { name ->
    171                 friend.mutableUsername?.contains(
    172                     name,
    173                     ignoreCase = true
    174                 ) == true || friend.displayName?.contains(name, ignoreCase = true) == true
    175             } ?: true
    176         }
    177     }

    596         conversationId: String,
    597         setDialogMessage: (String) -> Unit
    598     ) {
    599         val messageCount = mutableIntStateOf(0)
    600         MessagingTask(
    601             context.messagingBridge,
    602             conversationId,
    603             taskType = MessagingTaskType.DELETE,
    604             constraints = listOf(MessagingConstraints.MY_USER_ID(context.messagingBridge), {
    605                 contentType != ContentType.STATUS.id
    606             }),
    607             processedMessageCount = messageCount,
    608             onSuccess = {
    609                 setDialogMessage("${messageCount.intValue} deleted messages")
    610             },
    611         ).run()
    612     }
