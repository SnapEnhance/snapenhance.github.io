    266     private suspend fun fetchMessagesPaginated(conversationId: String, lastMessageId: Long, amount: Int): List<Message> = runBlocking {
    267         for (i in 0..5) {
    268             val messages: List<Message>? = suspendCancellableCoroutine { continuation ->
    269                 context.feature(Messaging::class).conversationManager?.fetchConversationWithMessagesPaginated(conversationId,
    270                     lastMessageId,
    271                     amount, onSuccess = { messages ->
    272                         continuation.resumeWith(Result.success(messages))
    273                     }, onError = {
    274                         continuation.resumeWith(Result.success(null))
    275                     }) ?: continuation.resumeWith(Result.success(null))
    276             }
    277             if (messages != null) return@runBlocking messages
    278             logDialog("Retrying in 1 second...")
    279             delay(1000)
    280         }
    281         logDialog("Failed to fetch messages")
    282         emptyList()
    283     }

    285     private fun exportChatForConversations(
    286         conversations: List<FriendFeedEntry>,
    287         exportParams: ExportParams,
    288     ) {
    289         dialogLogs.clear()
    290         val jobs = mutableListOf<Job>()
    291 
    292         currentActionDialog = ViewAppearanceHelper.newAlertDialogBuilder(context.mainActivity)
    293             .setTitle(translation["exporting_chats"])
    294             .setCancelable(false)
    295             .setMessage("")
    296             .create()
    297         
    298         val conversationSize = translation.format("processing_chats", "amount" to conversations.size.toString())
    299         
    300         logDialog(conversationSize)
    301 
    302         context.coroutineScope.launch {
    303             conversations.forEach { conversation ->
    304                 launch {
    305                     runCatching {
    306                         exportFullConversation(conversation, exportParams)
    307                     }.onFailure {
    308                         logDialog(translation.format("export_fail", "conversation" to conversation.key.toString()))
    309                         logDialog(it.stackTraceToString())
    310                         CoreLogger.xposedLog(it)
    311                     }
    312                 }.also { jobs.add(it) }
    313             }
    314             jobs.joinAll()
    315             logDialog(translation["finished"])
    316         }.also {
    317             currentActionDialog?.setButton(DialogInterface.BUTTON_POSITIVE, translation["dialog_negative_button"]) { dialog, _ ->
    318                 it.cancel()
    319                 jobs.forEach { it.cancel() }
    320                 dialog.dismiss()
    321             }
    322         }
    323 
    324         currentActionDialog!!.also {
    325             it.setCanceledOnTouchOutside(false)
    326         }.show()
    327     }

    329     private suspend fun exportFullConversation(
    330         feedEntry: FriendFeedEntry,
    331         exportParams: ExportParams,
    332     ) {
    333         //first fetch the first message
    334         val conversationId = feedEntry.key!!
    335         val conversationName = feedEntry.feedDisplayName ?: feedEntry.friendDisplayName!!.split("|").lastOrNull() ?: "unknown"
    336         val conversationParticipants = context.database.getConversationParticipants(feedEntry.key!!, useCache = false)
    337             ?.mapNotNull {
    338                 context.database.getFriendInfo(it)
    339             }?.associateBy { it.userId!! } ?: emptyMap()
    340 
    341         val publicFolder = File(Environment.getExternalStoragePublicDirectory(Environment.DIRECTORY_DOWNLOADS), "SnapEnhance").also { if (!it.exists()) it.mkdirs() }
    342         val outputFile = publicFolder.resolve("conversation_${conversationName}_${System.currentTimeMillis()}.${exportParams.exportFormat.extension}")
    343 
    344         logDialog(translation.format("exporting_message", "conversation" to conversationName))
    345 
    346         val conversationExporter = ConversationExporter(
    347             context = context,
    348             friendFeedEntry = feedEntry,
    349             conversationParticipants = conversationParticipants,
    350             exportParams = exportParams,
    351             cacheFolder = publicFolder.resolve("cache").also { if (!it.exists()) it.mkdirs() },
    352             outputFile = outputFile,
    353         ).apply { init(); printLog = {
    354             logDialog(it.toString())
    355         } }
    356 
    357         var foundMessageCount = 0
    358 
    359         var lastMessageId = fetchMessagesPaginated(conversationId, Long.MAX_VALUE, amount = 1).firstOrNull()?.also {
    360             conversationExporter.readMessage(it)
    361             foundMessageCount++
    362         }?.messageDescriptor?.messageId ?: run {
    363             logDialog(translation["no_messages_found"])
    364             return
    365         }
    366 
    367         while (true) {
    368             val fetchedMessages = fetchMessagesPaginated(conversationId, lastMessageId, amount = 500).toMutableList()
    369             if (fetchedMessages.isEmpty()) break
    370 
    371             fetchedMessages.firstOrNull()?.let {
    372                 lastMessageId = it.messageDescriptor!!.messageId!!
    373             }
    374 
    375             exportParams.messageTypeFilter?.let { filter ->
    376                 fetchedMessages.removeIf { message ->
    377                     !filter.contains(message.messageContent?.contentType ?: return@removeIf false)
    378                 }
    379             }
    380 
    381             foundMessageCount += fetchedMessages.size
    382 
    383             if (exportParams.amountOfMessages != null && foundMessageCount >= exportParams.amountOfMessages) {
    384                 fetchedMessages.reversed().subList(0, exportParams.amountOfMessages - (foundMessageCount - fetchedMessages.size)).forEach { message ->
    385                     conversationExporter.readMessage(message)
    386                 }
    387                 break
    388             }
    389 
    390             fetchedMessages.reversed().forEach { message ->
    391                 conversationExporter.readMessage(message)
    392             }
    393 
    394             setStatus("Exporting (found ${foundMessageCount})")
    395         }
    396 
    397         if (exportParams.exportFormat == ExportFormat.HTML) conversationExporter.awaitDownload()
    398         conversationExporter.close()
    399         logDialog(translation["writing_output"])
    400         dialogLogs.clear()
    401         logDialog("\n" + translation.format("exported_to",
    402             "path" to outputFile.absolutePath.toString()
    403         ) + "\n")
    404     }
