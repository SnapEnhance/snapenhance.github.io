     24 class SecurityFeatures : Feature("Security Features") {
     25     private fun transact(option: Int, option2: Long) = kotlin.runCatching { Os.prctl(option, option2, 0, 0, 0) }.getOrNull()
     26 
     27     private val token by lazy {
     28         transact(0, 0)
     29     }
     30 
     31     private fun getStatus() = token?.run {
     32         transact(this, 0)?.toString(2)?.padStart(32, '0')?.count { it == '1' }
     33     }
     34 
     35     override fun init() {
     36         token // pre init token
     37 
     38         context.event.subscribe(UnaryCallEvent::class) { event ->
     39             if (!event.uri.contains("/Login")) return@subscribe
     40 
     41             // intercept login response
     42             event.addResponseCallback {
     43                 val response = ProtoReader(buffer)
     44                 val isBlocked = when {
     45                     event.uri.contains("TLv") -> response.getVarInt(1) == 14L
     46                     else -> response.getVarInt(1) == 16L
     47                 }
     48 
     49                 val errorDataIndex = when {
     50                     response.contains(11) -> 11
     51                     response.contains(10) -> 10
     52                     response.contains(8) -> 8
     53                     else -> return@addResponseCallback
     54                 }
     55 
     56                 if (isBlocked) {
     57                     val status = transact(token ?: return@addResponseCallback, 1)?.let {
     58                         val buffer = ByteArray(8192)
     59                         val fd = FileDescriptor().apply {
     60                             setObjectField("descriptor", it)
     61                         }
     62                         val read = Os.read(fd, buffer, 0, buffer.size)
     63                         Os.close(fd)
     64                         buffer.copyOfRange(0, read).decodeToString()
     65                     }!!
     66 
     67                     buffer = ProtoEditor(buffer).apply {
     68                         edit(errorDataIndex) {
     69                             remove(1)
     70                             addString(1, status)
     71                         }
     72                     }.toByteArray()
     73                 }
     74             }
     75         }
     76 
     77         context.inAppOverlay.addCustomComposable {
     78             var statusText by remember {
     79                 mutableStateOf("")
     80             }
     81             var textColor by remember {
     82                 mutableStateOf(Color.Red)
     83             }
     84 
     85             LaunchedEffect(Unit) {
     86                 withContext(Dispatchers.IO) {
     87                     while (true) {
     88                         val status = getStatus()
     89                         withContext(Dispatchers.Main) {

     91                                 textColor = Color.Red
     92                                 statusText = "SIF not loaded!"
     93                             } else {
     94                                 textColor = Color.Green
     95                                 statusText = "SIF = $status"
     96                             }
     97                         }
     98                         delay(1000)
     99                     }
    100                 }
    101             }
    102 
    103             Text(

    112         }
    113     }
