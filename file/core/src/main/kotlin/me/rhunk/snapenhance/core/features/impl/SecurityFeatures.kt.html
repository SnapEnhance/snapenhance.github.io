     23 class SecurityFeatures : Feature("Security Features") {
     24     private fun transact(option: Int) = Os.prctl(option, 0, 0, 0, 0)
     25 
     26     private val token by lazy {
     27         runCatching { transact(0) }.getOrNull()
     28     }
     29 
     30     private fun getStatus() = token?.run {
     31         transact(this).toString(2).padStart(32, '0').count { it == '1' }
     32     }
     33 
     34     override fun init() {
     35         token // pre init token
     36 
     37         context.inAppOverlay.addCustomComposable {
     38             var statusText by remember {
     39                 mutableStateOf("")
     40             }
     41             var textColor by remember {
     42                 mutableStateOf(Color.Red)
     43             }
     44 
     45             LaunchedEffect(Unit) {
     46                 withContext(Dispatchers.IO) {
     47                     while (true) {
     48                         val status = getStatus()
     49                         withContext(Dispatchers.Main) {
     50                             if (status == null || status == 0) {
     51                                 textColor = Color.Red
     52                                 statusText = "sif not loaded. Can't get status"
     53                             } else {
     54                                 textColor = Color.Green
     55                                 statusText = "sif = $status"
     56                             }
     57                         }
     58                         delay(1000)
     59                     }
     60                 }
     61             }
     62 
     63             Text(
     64                 text = statusText,
     65                 modifier = Modifier
     66                     .align(Alignment.BottomCenter)
     67                     .background(Color.Black, shape = RoundedCornerShape(5.dp))
     68                     .padding(3.dp),
     69                 fontSize = 10.sp,
     70                 color = textColor
     71             )
     72         }
     73     }
     74 }
