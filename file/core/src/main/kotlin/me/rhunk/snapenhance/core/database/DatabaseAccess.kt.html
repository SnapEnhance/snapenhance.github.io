     42     private fun useDatabase(database: DatabaseType, writeMode: Boolean = false): SQLiteDatabase? {
     43         // only cache read-only databases
     44         if (!writeMode && openedDatabases.containsKey(database) && openedDatabases[database]?.isOpen == true) {
     45             return openedDatabases[database]
     46         }
     47 
     48         val dbPath = context.androidContext.getDatabasePath(database.fileName)
     49         if (!dbPath.exists()) return null
     50         return runCatching {
     51             SQLiteDatabase.openDatabase(
     52                 dbPath,
     53                 OpenParams.Builder()
     54                     .setOpenFlags(
     55                         if (writeMode) SQLiteDatabase.OPEN_READWRITE or SQLiteDatabase.ENABLE_WRITE_AHEAD_LOGGING
     56                         else SQLiteDatabase.OPEN_READONLY
     57                     )
     58                     .setErrorHandler {
     59                         context.androidContext.deleteDatabase(dbPath.absolutePath)
     60                         context.softRestartApp()
     61                     }.build()
     62             )
     63         }.onFailure {
     64             context.log.error("Failed to open database ${database.fileName}!", it)
     65         }.getOrNull()?.also {
     66             if (!writeMode) openedDatabases[database] = it
     67         }
     68     }

     70     private fun <T> SQLiteDatabase.performOperation(query: SQLiteDatabase.() -> T?): T? {
     71         return runCatching {
     72             if (NativeLib.initialized && openedDatabases[DatabaseType.ARROYO] == this) {
     73                 var result: T? = null
     74                 context.native.lockNativeDatabase(DatabaseType.ARROYO.fileName) {
     75                     result = query()
     76                 }
     77                 result
     78             } else synchronized(this) {
     79                 query()
     80             }
     81         }.onFailure {
     82             context.log.error("Database operation failed", it)
     83         }.getOrNull()
     84     }

     86     private fun SQLiteDatabase.safeRawQuery(query: String, args: Array<String>? = null): Cursor? {
     87         return runCatching {
     88             rawQuery(query, args)
     89         }.onFailure {
     90             if (it !is SQLiteDatabaseCorruptException) {
     91                 context.log.error("Failed to execute query $query", it)
     92                 return@onFailure
     93             }
     94             context.longToast("Database ${this.path} is corrupted! Restarting ...")
     95             context.androidContext.deleteDatabase(this.path)
     96             context.crash("Database ${this.path} is corrupted!", it)
     97         }.getOrNull()
     98     }

    167             useDatabase(type, writeMode = true)?.apply {
    168                 rawQuery("PRAGMA integrity_check", null).use { query ->
    169                     if (!query.moveToFirst() || query.getString(0).lowercase() != "ok") {
    170                         context.log.error("Failed to perform integrity check on ${type.fileName}")
    171                         context.androidContext.deleteDatabase(type.fileName)
    172                     }
    173                 }
    174             }?.close()
    175         }

    187     ): T? = this.safeRawQuery("SELECT * FROM $table WHERE $where", args)?.use {

    192             obj.write(it)

    195         }
    196         obj
