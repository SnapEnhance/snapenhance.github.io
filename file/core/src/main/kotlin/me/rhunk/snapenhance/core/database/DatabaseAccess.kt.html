     38     private fun useDatabase(database: DatabaseType, writeMode: Boolean = false): SQLiteDatabase? {
     39         // only cache read-only databases
     40         if (!writeMode && openedDatabases.containsKey(database) && openedDatabases[database]?.isOpen == true) {
     41             return openedDatabases[database]
     42         }
     43 
     44         val dbPath = context.androidContext.getDatabasePath(database.fileName)
     45         if (!dbPath.exists()) return null
     46         return runCatching {
     47             SQLiteDatabase.openDatabase(
     48                 dbPath,
     49                 OpenParams.Builder()
     50                     .setOpenFlags(
     51                         if (writeMode) SQLiteDatabase.OPEN_READWRITE or SQLiteDatabase.ENABLE_WRITE_AHEAD_LOGGING
     52                         else SQLiteDatabase.OPEN_READONLY
     53                     )
     54                     .setErrorHandler {
     55                         context.androidContext.deleteDatabase(dbPath.absolutePath)
     56                         context.softRestartApp()
     57                     }.build()
     58             )
     59         }.onFailure {
     60             context.log.error("Failed to open database ${database.fileName}!", it)
     61         }.getOrNull()?.also {
     62             if (!writeMode) openedDatabases[database] = it
     63         }
     64     }

     66     private fun <T> SQLiteDatabase.performOperation(query: SQLiteDatabase.() -> T?): T? {
     67         return runCatching {
     68             if (NativeLib.initialized && openedDatabases[DatabaseType.ARROYO] == this) {
     69                 var result: T? = null
     70                 context.native.lockNativeDatabase(DatabaseType.ARROYO.fileName) {
     71                     result = query()
     72                 }
     73                 result
     74             } else synchronized(this) {
     75                 query()
     76             }
     77         }.onFailure {
     78             context.log.error("Database operation failed", it)
     79         }.getOrNull()
     80     }

     82     private fun SQLiteDatabase.safeRawQuery(query: String, args: Array<String>? = null): Cursor? {
     83         return runCatching {
     84             rawQuery(query, args)
     85         }.onFailure {
     86             if (it !is SQLiteDatabaseCorruptException) {
     87                 context.log.error("Failed to execute query $query", it)
     88                 return@onFailure
     89             }
     90             context.longToast("Database ${this.path} is corrupted! Restarting ...")
     91             context.androidContext.deleteDatabase(this.path)
     92             context.crash("Database ${this.path} is corrupted!", it)
     93         }.getOrNull()
     94     }

     96     private val dmOtherParticipantCache by lazy {
     97         (useDatabase(DatabaseType.ARROYO)?.performOperation {
     98             safeRawQuery(
     99                 "SELECT client_conversation_id, conversation_type, user_id FROM user_conversation WHERE user_id != ?",
    100                 arrayOf(myUserId)
    101             )?.use { query ->
    102                 val participants = mutableMapOf<String, String?>()
    103                 if (!query.moveToFirst()) {
    104                     return@performOperation null
    105                 }
    106                 do {
    107                     val conversationId = query.getStringOrNull("client_conversation_id") ?: continue
    108                     val userId = query.getStringOrNull("user_id") ?: continue
    109                     participants[conversationId] = when (query.getIntOrNull("conversation_type")) {
    110                         0 -> userId
    111                         else -> null
    112                     }
    113                     participants[userId] = null
    114                 } while (query.moveToNext())
    115                 participants
    116             }
    117         } ?: emptyMap()).toMutableMap()
    118     }

    126             useDatabase(type, writeMode = true)?.apply {
    127                 rawQuery("PRAGMA integrity_check", null).use { query ->
    128                     if (!query.moveToFirst() || query.getString(0).lowercase() != "ok") {
    129                         context.log.error("Failed to perform integrity check on ${type.fileName}")
    130                         context.androidContext.deleteDatabase(type.fileName)
    131                     }
    132                 }
    133             }?.close()

    141     private fun <T : DatabaseObject> SQLiteDatabase.readDatabaseObject(
    142         obj: T,
    143         table: String,
    144         where: String,
    145         args: Array<String>
    146     ): T? = this.safeRawQuery("SELECT * FROM $table WHERE $where", args)?.use {
    147         if (!it.moveToFirst()) {
    148             return null
    149         }
    150         try {
    151             obj.write(it)
    152         } catch (e: Throwable) {
    153             context.log.error("Failed to read database object", e)
    154         }
    155         obj
    156     }

    337     fun getConversationParticipants(conversationId: String, useCache: Boolean = true): List<String>? {
    338         if (dmOtherParticipantCache[conversationId] != null && useCache) return dmOtherParticipantCache[conversationId]?.let { listOf(myUserId, it) }
    339         return useDatabase(DatabaseType.ARROYO)?.performOperation {
    340             safeRawQuery(
    341                 "SELECT user_id, conversation_type FROM user_conversation WHERE client_conversation_id = ?",
    342                 arrayOf(conversationId)
    343             )?.use { cursor ->
    344                 if (!cursor.moveToFirst()) {
    345                     return@performOperation null
    346                 }
    347                 val participants = mutableListOf<String>()
    348                 var conversationType = -1
    349                 do {
    350                     if (conversationType == -1) conversationType = cursor.getInteger("conversation_type")
    351                     participants.add(cursor.getStringOrNull("user_id")!!)
    352                 } while (cursor.moveToNext())
    353 
    354                 if (!dmOtherParticipantCache.containsKey(conversationId)) {
    355                     dmOtherParticipantCache[conversationId] = when (conversationType) {
    356                         0 -> participants.firstOrNull { it != myUserId }
    357                         else -> null
    358                     }
    359                 }
    360                 participants
    361             }
    362         }
    363     }

    436                     services[service] = token
    437                 }
    438

    440             }
    441         }

    463     fun updatePinnedBestFriendStatus(userId: String, friendmoji: String) {
    464         useDatabase(DatabaseType.MAIN, writeMode = true)?.apply {
    465             val numberOneBestFriends = getBestFriends().filter { friend ->
    466                 friend.friendmojiCategories?.split(",")?.any { it.startsWith("number_one") } == true
    467             }

    469             numberOneBestFriends.forEach { friendInfo ->

    476                             }?.joinToString(",") ?: "")
    477                             put("isPinnedBestFriend", 0)
    478                         },
    479                         "userId = ?",
    480                         arrayOf(friendInfo.userId)
    481                     )
    482                 }
    483             }
    484 
    485             val friend = getFriendInfo(userId) ?: return@apply
    486             performOperation {
    487                 update(
    488                     "Friend",
    489                     ContentValues().apply {

    493                         put("isPinnedBestFriend", 1)
