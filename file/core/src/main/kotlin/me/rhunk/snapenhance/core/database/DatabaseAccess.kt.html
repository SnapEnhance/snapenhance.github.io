     35     private fun useDatabase(database: DatabaseType, writeMode: Boolean = false): SQLiteDatabase? {
     36         // only cache read-only databases
     37         if (!writeMode && openedDatabases.containsKey(database) && openedDatabases[database]?.isOpen == true) {
     38             return openedDatabases[database]
     39         }
     40 
     41         val dbPath = context.androidContext.getDatabasePath(database.fileName)
     42         if (!dbPath.exists()) return null
     43         return runCatching {
     44             SQLiteDatabase.openDatabase(
     45                 dbPath,
     46                 OpenParams.Builder()
     47                     .setOpenFlags(
     48                         if (writeMode) SQLiteDatabase.OPEN_READWRITE or SQLiteDatabase.ENABLE_WRITE_AHEAD_LOGGING
     49                         else SQLiteDatabase.OPEN_READONLY
     50                     )
     51                     .setErrorHandler {
     52                         context.androidContext.deleteDatabase(dbPath.absolutePath)
     53                         context.softRestartApp()
     54                     }.build()
     55             )
     56         }.onFailure {
     57             context.log.error("Failed to open database ${database.fileName}!", it)
     58         }.getOrNull()?.also {
     59             if (!writeMode) openedDatabases[database] = it
     60         }
     61     }

     63     private fun <T> SQLiteDatabase.performOperation(query: SQLiteDatabase.() -> T?): T? {
     64         return runCatching {
     65             if (NativeLib.initialized && openedDatabases[DatabaseType.ARROYO] == this) {
     66                 var result: T? = null
     67                 context.native.lockNativeDatabase(DatabaseType.ARROYO.fileName) {
     68                     result = query()
     69                 }
     70                 result
     71             } else synchronized(this) {
     72                 query()
     73             }
     74         }.onFailure {
     75             context.log.error("Database operation failed", it)
     76         }.getOrNull()
     77     }

     79     private fun SQLiteDatabase.safeRawQuery(query: String, args: Array<String>? = null): Cursor? {
     80         return runCatching {
     81             rawQuery(query, args)
     82         }.onFailure {
     83             if (it !is SQLiteDatabaseCorruptException) {
     84                 context.log.error("Failed to execute query $query", it)
     85                 return@onFailure
     86             }
     87             context.longToast("Database ${this.path} is corrupted! Restarting ...")
     88             context.androidContext.deleteDatabase(this.path)
     89             context.crash("Database ${this.path} is corrupted!", it)
     90         }.getOrNull()
     91     }

     93     private val dmOtherParticipantCache by lazy {
     94         (useDatabase(DatabaseType.ARROYO)?.performOperation {
     95             safeRawQuery(
     96                 "SELECT client_conversation_id, conversation_type, user_id FROM user_conversation WHERE user_id != ?",
     97                 arrayOf(myUserId)
     98             )?.use { query ->
     99                 val participants = mutableMapOf<String, String?>()
    100                 if (!query.moveToFirst()) {
    101                     return@performOperation null
    102                 }
    103                 do {
    104                     val conversationId = query.getStringOrNull("client_conversation_id") ?: continue
    105                     val userId = query.getStringOrNull("user_id") ?: continue
    106                     participants[conversationId] = when (query.getIntOrNull("conversation_type")) {
    107                         0 -> userId
    108                         else -> null
    109                     }
    110                     participants[userId] = null
    111                 } while (query.moveToNext())
    112                 participants
    113             }
    114         } ?: emptyMap()).toMutableMap()
    115     }

    120     fun init() {
    121         // perform integrity check on databases
    122         DatabaseType.entries.forEach { type ->
    123             useDatabase(type, writeMode = true)?.apply {
    124                 rawQuery("PRAGMA integrity_check", null).use { query ->
    125                     if (!query.moveToFirst() || query.getString(0).lowercase() != "ok") {
    126                         context.log.error("Failed to perform integrity check on ${type.fileName}")
    127                         context.androidContext.deleteDatabase(type.fileName)
    128                     }
    129                 }
    130             }?.close()
    131         }
    132     }

    138     private fun <T : DatabaseObject> SQLiteDatabase.readDatabaseObject(
    139         obj: T,
    140         table: String,
    141         where: String,
    142         args: Array<String>
    143     ): T? = this.safeRawQuery("SELECT * FROM $table WHERE $where", args)?.use {
    144         if (!it.moveToFirst()) {
    145             return null
    146         }
    147         try {
    148             obj.write(it)
    149         } catch (e: Throwable) {
    150             context.log.error("Failed to read database object", e)
    151         }
    152         obj
    153     }

    324     fun getConversationParticipants(conversationId: String, useCache: Boolean = true): List<String>? {
    325         if (dmOtherParticipantCache[conversationId] != null && useCache) return dmOtherParticipantCache[conversationId]?.let { listOf(myUserId, it) }
    326         return useDatabase(DatabaseType.ARROYO)?.performOperation {
    327             safeRawQuery(
    328                 "SELECT user_id, conversation_type FROM user_conversation WHERE client_conversation_id = ?",
    329                 arrayOf(conversationId)
    330             )?.use { cursor ->
    331                 if (!cursor.moveToFirst()) {
    332                     return@performOperation null
    333                 }
    334                 val participants = mutableListOf<String>()
    335                 var conversationType = -1
    336                 do {
    337                     if (conversationType == -1) conversationType = cursor.getInteger("conversation_type")
    338                     participants.add(cursor.getStringOrNull("user_id")!!)
    339                 } while (cursor.moveToNext())
    340 
    341                 if (!dmOtherParticipantCache.containsKey(conversationId)) {
    342                     dmOtherParticipantCache[conversationId] = when (conversationType) {
    343                         0 -> participants.firstOrNull { it != myUserId }
    344                         else -> null
    345                     }
    346                 }
    347                 participants
    348             }
    349         }
    350     }

    418                 val services = mutableMapOf<String, String>()

    420                 reader.eachBuffer(1) {
    421                     val token = getString(1) ?: return@eachBuffer

    423                     services[service] = token
    424                 }

    450     fun updatePinnedBestFriendStatus(userId: String, friendmoji: String) {
    451         useDatabase(DatabaseType.MAIN, writeMode = true)?.apply {
    452             val numberOneBestFriends = getBestFriends().filter { friend ->
    453                 friend.friendmojiCategories?.split(",")?.any { it.startsWith("number_one") } == true
    454             }

    456             numberOneBestFriends.forEach { friendInfo ->

    470             }
