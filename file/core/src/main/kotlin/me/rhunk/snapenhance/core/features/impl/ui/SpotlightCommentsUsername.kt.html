     15 class SpotlightCommentsUsername : Feature("SpotlightCommentsUsername", loadParams =  FeatureLoadParams.ACTIVITY_CREATE_SYNC) {
     16     private val usernameCache = EvictingMap<String, String>(150)
     17 
     18     @SuppressLint("SetTextI18n")
     19     override fun onActivityCreate() {
     20         if (!context.config.global.spotlightCommentsUsername.get()) return
     21 
     22         val messaging = context.feature(Messaging::class)
     23         val commentsCreatorBadgeTimestampId = context.resources.getId("comments_creator_badge_timestamp")
     24 
     25         context.event.subscribe(BindViewEvent::class) { event ->
     26             val commentsCreatorBadgeTimestamp = event.view.findViewById<TextView>(commentsCreatorBadgeTimestampId) ?: return@subscribe
     27 
     28             val posterUserId = event.prevModel.toString().takeIf { it.startsWith("Comment") }
     29                 ?.substringAfter("posterUserId=")?.substringBefore(",")?.substringBefore(")") ?: return@subscribe
     30 
     31             fun setUsername(username: String) {
     32                 usernameCache[posterUserId] = username
     33                 if (commentsCreatorBadgeTimestamp.text.contains(username)) return
     34                 commentsCreatorBadgeTimestamp.text = " (${username})" + commentsCreatorBadgeTimestamp.text.toString()
     35             }
     36 
     37             usernameCache[posterUserId]?.let {
     38                 setUsername(it)
     39                 return@subscribe
     40             }
     41 
     42             context.coroutineScope.launch {
     43                 val username = runCatching {
     44                     messaging.fetchSnapchatterInfos(listOf(posterUserId)).firstOrNull()
     45                 }.onFailure {
     46                     context.log.error("Failed to fetch snapchatter info for user $posterUserId", it)
     47                 }.getOrNull()?.username ?: return@launch
     48 
     49                 withContext(Dispatchers.Main) {
     50                     setUsername(username)
     51                 }
     52             }
     53         }
     54     }
