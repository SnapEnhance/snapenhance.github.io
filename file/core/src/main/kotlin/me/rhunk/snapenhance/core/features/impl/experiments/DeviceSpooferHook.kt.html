     14 class DeviceSpooferHook: Feature("Device Spoofer")  {

     17 			param.setResult("com.android.vending")
     18 		}
     19 	}

     21 	@SuppressLint("MissingPermission")

     46 					"android_id" -> {

     78 					val capabilities = instance.getNetworkCapabilities(network) ?: return@filter false
     79 					!capabilities.hasTransport(NetworkCapabilities.TRANSPORT_VPN)
     80 				}.toTypedArray())
     81 			}
     82 		}
     83 
     84 		if (fingerprint.isNotEmpty()) {
     85 			Build.FINGERPRINT // init fingerprint field
     86 			Build::class.java.getField("FINGERPRINT").apply {
     87 				isAccessible = true
     88 				set(null, fingerprint)
     89 				isAccessible = false
     90 			}
     91 		}
     92 	}
