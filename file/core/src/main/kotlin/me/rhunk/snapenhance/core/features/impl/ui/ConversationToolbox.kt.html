     48 class ConversationToolbox : Feature("Conversation Toolbox", loadParams = FeatureLoadParams.ACTIVITY_CREATE_SYNC) {
     49     private val composableList = mutableListOf<ComposableMenu>()

     52     fun addComposable(title: String, filter: (conversationId: String) -> Boolean = { true }, composable: @Composable (alertDialog: AlertDialog, conversationId: String) -> Unit) {
     53         composableList.add(
     54             ComposableMenu(title, filter, composable)

     58     @SuppressLint("SetTextI18n")
     59     override fun onActivityCreate() {
     60         val defaultInputBarId = context.resources.getId("default_input_bar")
     61 
     62         context.event.subscribe(AddViewEvent::class) { event ->
     63             if (event.view.id != defaultInputBarId) return@subscribe
     64             if (composableList.isEmpty()) return@subscribe
     65 
     66             (event.view as ViewGroup).addView(FrameLayout(event.view.context).apply {
     67                 layoutParams = LinearLayout.LayoutParams(
     68                     ViewGroup.LayoutParams.WRAP_CONTENT,
     69                     (52 * context.resources.displayMetrics.density).toInt(),
     70                 ).apply {
     71                     gravity = Gravity.BOTTOM
     72                 }
     73                 setPadding(25, 0, 25, 0)
     74 
     75                 addView(TextView(event.view.context).apply {
     76                     layoutParams = FrameLayout.LayoutParams(
     77                         ViewGroup.LayoutParams.WRAP_CONTENT,
     78                         ViewGroup.LayoutParams.WRAP_CONTENT,
     79                     ).apply {
     80                         gravity = Gravity.CENTER_VERTICAL
     81                     }
     82                     setOnClickListener {
     83                         openToolbox()
     84                     }
     85                     textSize = 21f
     86                     text = "\uD83E\uDDF0"
     87                 })
     88             })
     89         }
     90 
     91         context.scriptRuntime.eachModule {
     92             val interfaceManager = getBinding(InterfaceManager::class)?.takeIf {
     93                 it.hasInterface(EnumScriptInterface.CONVERSATION_TOOLBOX)
     94             } ?: return@eachModule
     95             addComposable("\uD83D\uDCDC ${moduleInfo.displayName}") { alertDialog, conversationId ->
     96                 ScriptInterface(remember {
     97                     interfaceManager.buildInterface(EnumScriptInterface.CONVERSATION_TOOLBOX, mapOf(
     98                         "alertDialog" to alertDialog,
     99                         "conversationId" to conversationId,
    100                     ))
    101                 } ?: return@addComposable)
    102             }
    103         }
    104     }

    106     private fun openToolbox() {
    107         val openedConversationId = context.feature(Messaging::class).openedConversationUUID?.toString() ?: run {
    108             context.shortToast("You must open a conversation first")
    109             return
    110         }
    111 
    112         createComposeAlertDialog(context.mainActivity!!) { alertDialog ->
    113             Column(
    114                 modifier = Modifier
    115                     .fillMaxWidth()
    116                     .heightIn(
    117                         min = 100.dp,
    118                         max = LocalConfiguration.current.screenHeightDp * 0.8f.dp
    119                     )
    120                     .verticalScroll(rememberScrollState())
    121             ) {
    122                 Text("Conversation Toolbox", fontSize = 20.sp, modifier = Modifier
    123                     .fillMaxWidth()
    124                     .padding(10.dp), textAlign = TextAlign.Center)
    125                 Spacer(modifier = Modifier.height(10.dp))
    126 
    127                 composableList.reversed().forEach { (title, filter, composable) ->
    128                     if (!filter(openedConversationId)) return@forEach
    129                     Card(
    130                         modifier = Modifier
    131                             .fillMaxWidth()
    132                             .padding(5.dp),
    133                         shape = MaterialTheme.shapes.medium
    134                     ) {
    135                         Row(
    136                             modifier = Modifier
    137                                 .clickable {
    138                                     expandedComposableCache[title] = !(expandedComposableCache[title] ?: false)
    139                                 }
    140                                 .fillMaxWidth()
    141                                 .padding(10.dp),
    142                             horizontalArrangement = Arrangement.spacedBy(10.dp),
    143                             verticalAlignment = Alignment.CenterVertically,
    144                         ) {
    145                             Image(
    146                                 imageVector = if (expandedComposableCache[title] == true) Icons.Filled.KeyboardArrowDown else Icons.Filled.KeyboardArrowUp,
    147                                 contentDescription = null,
    148                                 colorFilter = ColorFilter.tint(MaterialTheme.colorScheme.onSurface),
    149                             )
    150                             Text(title, fontSize = 16.sp, fontStyle = FontStyle.Italic)
    151                         }
    152                         if (expandedComposableCache[title] == true) {
    153                             Column(
    154                                 modifier = Modifier
    155                                     .fillMaxWidth()
    156                                     .padding(10.dp)
    157                             ) {
    158                                 runCatching {
    159                                     composable(alertDialog, openedConversationId)
    160                                 }.onFailure { throwable ->
    161                                     Text("Failed to load composable: ${throwable.message}")
    162                                     context.log.error("Failed to load composable: ${throwable.message}", throwable)
    163                                 }
    164                             }
    165                         }
    166                     }
    167                 }
    168             }
    169         }.show()
    170     }
