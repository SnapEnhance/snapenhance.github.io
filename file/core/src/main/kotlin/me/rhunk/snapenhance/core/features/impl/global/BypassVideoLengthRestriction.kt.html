     15 class BypassVideoLengthRestriction :
     16     Feature("BypassVideoLengthRestriction", loadParams = FeatureLoadParams.ACTIVITY_CREATE_ASYNC) {
     17     private lateinit var fileObserver: FileObserver
     18 
     19     override fun asyncOnActivityCreate() {
     20         val mode = context.config.global.bypassVideoLengthRestriction.getNullable()
     21 
     22         if (mode == "single") {
     23             //fix black videos when story is posted
     24             if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
     25                 val postedStorySnapFolder =
     26                     File(context.androidContext.filesDir, "file_manager/posted_story_snap")
     27 
     28                 fileObserver = (object : FileObserver(postedStorySnapFolder, MOVED_TO) {
     29                     override fun onEvent(event: Int, path: String?) {
     30                         if (event != MOVED_TO || path?.endsWith("posted_story_snap.2") != true) return
     31                         fileObserver.stopWatching()
     32 
     33                         val file = File(postedStorySnapFolder, path)
     34                         runCatching {
     35                             val fileContent = JsonParser.parseReader(file.reader()).asJsonObject
     36                             if (fileContent["timerOrDuration"].asLong < 0) file.delete()
     37                         }.onFailure {
     38                             context.log.error("Failed to read story metadata file", it)
     39                         }
     40                     }
     41                 })
     42 
     43                 context.event.subscribe(SendMessageWithContentEvent::class) { event ->
     44                     if (event.destinations.stories!!.isEmpty()) return@subscribe
     45                     fileObserver.startWatching()
     46                 }
     47             }
     48 
     49             context.mappings.useMapper(DefaultMediaItemMapper::class) {
     50                 defaultMediaItem.getAsClass()?.hookConstructor(HookStage.BEFORE) { param ->
     51                     //set the video length argument
     52                     param.setArg(5, -1L)
     53                 }
     54             }
     55         }
     56 
     57         //TODO: allow split from any source
     58         if (mode == "split") {
     59             // memories grid
     60             context.mappings.useMapper(DefaultMediaItemMapper::class) {
     61                 cameraRollMediaId.getAsClass()?.hookConstructor(HookStage.AFTER) { param ->
     62                     //set the durationMs field
     63                     param.thisObject<Any>()
     64                         .setObjectField(durationMsField.get()!!, -1L)
     65                 }
     66             }
     67 
     68             // chat camera roll grid
     69             findClass("com.snap.composer.memories.MemoriesPickerVideoDurationConfig").hookConstructor(HookStage.AFTER) { param ->
     70                 param.thisObject<Any>().apply {
     71                     setObjectField("_maxSingleItemDurationMs", null)
     72                     setObjectField("_maxTotalDurationMs", null)
     73                 }
     74             }
     75         }
     76     }
