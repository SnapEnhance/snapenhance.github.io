    261     private fun login(userId: String, username: String) {
    262         val accountData = context.bridgeClient.getAccountStorage().getAccountData(userId)?.let { pfd ->
    263             ParcelFileDescriptor.AutoCloseInputStream(pfd).use { it.readBytes() }
    264         }
    265         if (accountData == null) {
    266             context.shortToast("Account data not found")
    267             return
    268         }
    269 
    270         arrayOf(
    271             context.androidContext.filesDir,
    272             context.androidContext.cacheDir,
    273             context.androidContext.dataDir.resolve("databases"),
    274         ).forEach { dir -> dir.listFiles()?.forEach { it.deleteRecursively() } }
    275 
    276         val zipInputStream = ZipInputStream(accountData.inputStream())
    277         var entry: ZipEntry?
    278         while (zipInputStream.nextEntry.also { entry = it } != null) {
    279             val file = context.androidContext.dataDir.resolve(entry!!.name)
    280             if (file.exists()) {
    281                 file.delete()
    282             } else {
    283                 file.parentFile?.mkdirs()
    284             }
    285             context.log.debug("Extracting ${file.absolutePath}")
    286             file.outputStream().use {
    287                 zipInputStream.copyTo(it)
    288             }
    289         }
    290 
    291         context.log.debug("Account data restored")
    292         context.shortToast("Logged in as $username")
    293         context.softRestartApp()
    294     }

    296     private fun getCurrentAccountData(): ParcelFileDescriptor {
    297         val pfd = ParcelFileDescriptor.createPipe()
    298 
    299         context.coroutineScope.launch(Dispatchers.IO) {
    300             val zipOutputStream = ZipOutputStream(ParcelFileDescriptor.AutoCloseOutputStream(pfd[1]))
    301 
    302             fun addFile(path: String, file: File) {
    303                 file.takeIf { it.exists() }?.inputStream()?.use {
    304                     context.log.verbose("Adding $file to zip")
    305                     zipOutputStream.putNextEntry(ZipEntry(path))
    306                     it.copyTo(zipOutputStream)
    307                     zipOutputStream.closeEntry()
    308                 }
    309             }
    310 
    311             for (path in arrayOf(
    312                 "databases/main.db",
    313                 "databases/main.db-shm",
    314                 "databases/main.db-wal",
    315                 "databases/core.db",
    316                 "databases/core.db-wal",
    317                 "databases/core.db-shm",
    318                 "databases/fidelius_database.db",
    319                 "databases/fidelius_database.db-wal",
    320                 "databases/fidelius_database.db-shm",
    321                 "shared_prefs/user_session_shared_pref.xml",
    322                 "shared_prefs/user_device_identity_keys.xml",
    323                 "shared_prefs/com.google.android.gms.appid.xml",
    324             )) {
    325                 addFile(path, context.androidContext.dataDir.resolve(path))
    326             }
    327 
    328             context.androidContext.dataDir.resolve("databases").listFiles()?.filter {
    329                 it.name.contains("_fidelius.db")
    330             }?.forEach {
    331                 addFile("databases/${it.name}", it)
    332             }
    333 
    334             zipOutputStream.flush()
    335             zipOutputStream.close()
    336         }
    337 
    338         return pfd[0]
    339     }

    355     private fun importAccount(fileUri: Uri) {
    356         var tempZip: File? = null
    357         var mainDbFile: File? = null
    358         var mainDbWalFile: File? = null
    359         var mainDbShmFile: File? = null
    360 
    361         runCatching {
    362             // copy zip file
    363             context.mainActivity!!.contentResolver.openInputStream(fileUri)?.use { input ->
    364                 val bufferedInputStream = input.buffered()
    365                 val fileType = MediaDownloaderHelper.getFileType(bufferedInputStream)
    366 
    367                 if (fileType != FileType.ZIP) {
    368                     throw Exception("Invalid file type")
    369                 }
    370 
    371                 context.androidContext.cacheDir.resolve(System.currentTimeMillis().toString()).also {
    372                     tempZip = it
    373                 }.outputStream().use { output ->
    374                     bufferedInputStream.copyTo(output)
    375                 }
    376             }
    377 
    378             context.log.verbose("Extracting account data")
    379 
    380             // extract main.db in cache
    381             tempZip?.inputStream().use { fileInputStream ->
    382                 val zipInputStream = ZipInputStream(fileInputStream)
    383                 var entry: ZipEntry?
    384                 while (zipInputStream.nextEntry.also { entry = it } != null) {
    385                     val fileName = entry?.name?.substringAfterLast('/') ?: continue
    386                     if (!fileName.startsWith("main.db")) continue
    387 
    388                     val file = context.androidContext.cacheDir.resolve(fileName)
    389                     context.log.verbose("Found ${entry!!.name} in zip file")
    390 
    391                     when (fileName) {
    392                         "main.db" -> mainDbFile = file
    393                         "main.db-wal" -> mainDbWalFile = file
    394                         "main.db-shm" -> mainDbShmFile = file
    395                     }
    396 
    397                     file.outputStream().use {
    398                         zipInputStream.copyTo(it)
    399                     }
    400                 }
    401             }
    402 
    403             assert(mainDbFile != null) { "main.db not found in zip file" }
    404 
    405             SQLiteDatabase.openDatabase(mainDbFile!!.absolutePath, null, SQLiteDatabase.OPEN_READONLY).use {  sqliteDatabase ->
    406                 val userId = sqliteDatabase.rawQuery("SELECT userId FROM SnapToken", null).use {
    407                     if (!it.moveToFirst()) throw Exception("userId not found in main.db")
    408                     it.getString(0)
    409                 }
    410                 context.log.verbose("Found userId $userId")
    411                 val username = sqliteDatabase.rawQuery("SELECT username FROM Friend WHERE userId = ?", arrayOf(userId)).use {
    412                     if (!it.moveToFirst()) throw Exception("username not found in main.db")
    413                     it.getString(0)
    414                 }
    415                 context.log.verbose("Found username $username")
    416                 tempZip?.inputStream()?.use {
    417                     context.bridgeClient.getAccountStorage().addAccount(
    418                         userId,
    419                         username,
    420                         it.toParcelFileDescriptor(context.coroutineScope)
    421                     )
    422                 }
    423                 context.shortToast("Imported $username!")
    424                 updateUsers()
    425             }
    426         }.onFailure {
    427             context.shortToast("Failed to import account: ${it.message}")
    428             context.log.error("Failed to import account", it)
    429         }
    430 
    431         tempZip?.delete()
    432         mainDbFile?.delete()
    433         mainDbWalFile?.delete()
    434         mainDbShmFile?.delete()
    435     }

    437     @SuppressLint("SetTextI18n")
    438     override fun init() {
    439         if (context.config.experimental.accountSwitcher.globalState != true) return
    440 
    441         onNextActivityCreate {
    442             val hovaHeaderSearchIcon = context.resources.getId("hova_header_search_icon")
    443 
    444             context.event.subscribe(AddViewEvent::class) { event ->
    445                 if (event.view.id != hovaHeaderSearchIcon) return@subscribe
    446 
    447                 event.view.setOnLongClickListener {
    448                     context.mainActivity!!.vibrateLongPress()
    449                     showManagementPopup()
    450                     false
    451                 }
    452             }
    453         }
    454 
    455         context.event.subscribe(ActivityResultEvent::class) { event ->
    456             if (importRequestCode == event.requestCode) {
    457                 importRequestCode = null
    458                 if (event.resultCode != Activity.RESULT_OK) return@subscribe
    459                 event.canceled = true
    460                 val uri = event.intent.data ?: return@subscribe
    461 
    462                 context.coroutineScope.launch { importAccount(uri) }
    463             }
    464 
    465             if (exportCallback?.first == event.requestCode) {
    466                 val userId = exportCallback?.second
    467                 exportCallback = null
    468                 event.canceled = true
    469                 if (event.resultCode != Activity.RESULT_OK) return@subscribe
    470 
    471                 context.coroutineScope.launch {
    472                     runCatching {
    473                         event.intent.data?.let { uri ->
    474                             val accountDataPfd = context.bridgeClient.getAccountStorage().getAccountData(userId) ?: throw Exception("Account data not found")
    475                             context.androidContext.contentResolver.openOutputStream(uri)?.use { outputStream ->
    476                                 ParcelFileDescriptor.AutoCloseInputStream(accountDataPfd).use {
    477                                     it.copyTo(outputStream)
    478                                 }
    479                             }
    480                             context.shortToast("Account exported!")
    481                         }
    482                     }.onFailure {
    483                         context.shortToast("Failed to export account. Check logs for more info.")
    484                         context.log.error("Failed to export account", it)
    485                     }
    486                 }
    487             }
    488         }
    489 
    490         findClass("com.snap.identity.service.ForcedLogoutBroadcastReceiver").hook("onReceive", HookStage.BEFORE) { param ->
    491             val intent = param.arg<Intent>(1)
    492             if (isLoginActivity) return@hook
    493             if (intent.getBooleanExtra("forced", false) && !context.config.experimental.preventForcedLogout.get()) {
    494                 runCatching {
    495                     val accountStorage = context.bridgeClient.getAccountStorage()
    496 
    497                     if (accountStorage.isAccountExists(context.database.myUserId)) {
    498                         accountStorage.removeAccount(context.database.myUserId)
    499                         context.shortToast("Removed account due to forced logout")
    500                     }
    501                 }
    502                 return@hook
    503             }
    504 
    505             if (context.config.experimental.accountSwitcher.autoBackupCurrentAccount.get()) {
    506                 backupCurrentAccount()
    507             }
    508         }
    509 
    510         findClass("com.snap.identity.loginsignup.ui.LoginSignupActivity").apply {
    511             hook("onPostCreate", HookStage.AFTER) {
    512                 context.mainActivity!!.findViewById<FrameLayout>(android.R.id.content).addView(createComposeView(context.mainActivity!!) {
    513                     Row(
    514                         modifier = Modifier.fillMaxWidth(),
    515                     ) {
    516                         Button(
    517                             onClick = { showManagementPopup() },
    518                             modifier = Modifier.padding(16.dp)
    519                         ) {
    520                             Text("Switch Account")
    521                         }
    522                     }
    523                 }.apply {
    524                     layoutParams = FrameLayout.LayoutParams(FrameLayout.LayoutParams.WRAP_CONTENT, FrameLayout.LayoutParams.WRAP_CONTENT).apply {
    525                         gravity = android.view.Gravity.TOP or android.view.Gravity.START
    526                     }
    527                 })
    528             }
    529         }
    530     }
