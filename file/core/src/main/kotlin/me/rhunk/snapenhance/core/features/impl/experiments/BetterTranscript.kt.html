     18 class BetterTranscript: Feature("Better Transcript", loadParams = FeatureLoadParams.ACTIVITY_CREATE_SYNC) {

     21     override fun onActivityCreate() {

     23         val config = context.config.experimental.betterTranscript
     24         val preferredTranscriptionLang = config.preferredTranscriptionLang.getNullable()?.takeIf {
     25             it.isNotBlank()
     26         }
     27 
     28         if (config.forceTranscription.get()) {
     29             context.event.subscribe(BuildMessageEvent::class, priority = 104) { event ->
     30                 if (event.message.messageContent?.contentType != ContentType.NOTE) return@subscribe
     31                 event.message.messageContent!!.content = ProtoEditor(event.message.messageContent!!.content!!).apply {
     32                     edit(6, 1) {
     33                         if (firstOrNull(3) == null) {
     34                             addString(3, context.getConfigLocale())
     35                         }
     36                     }
     37                 }.toByteArray()
     38             }
     39         }
     40 
     41         findClass("com.snapchat.client.voiceml.IVoiceMLSDK\$CppProxy").hook("asrTranscribe", HookStage.BEFORE) { param ->
     42             if (config.enhancedTranscript.get()) {
     43                 val buffer = param.arg<ByteBuffer>(2).let {
     44                     it.rewind()
     45                     ByteArray(it.remaining()).also { it1 -> it.get(it1); it.rewind() }
     46                 }
     47                 val result = runCatching {
     48                     transcriptApi.transcribe(
     49                         buffer.toRequestBody(),
     50                         lang = config.preferredTranscriptionLang.getNullable()?.takeIf {
     51                             it.isNotBlank()
     52                         }?.uppercase()
     53                     )
     54                 }.onFailure {
     55                     context.log.error("Failed to transcribe audio", it)
     56                     context.shortToast("Failed to transcribe audio! Check logcat for more details.")
     57                 }.getOrNull()
     58 
     59                 param.setResult(
     60                     (param.method() as Method).returnType.dataBuilder {
     61                         set("mError", result == null)
     62                         set("mNlpResponses", ArrayList<Any>())
     63                         set("mWordInfo", ArrayList<Any>())
     64                         set("mTranscription", result)
     65                     }
     66                 )
     67                 return@hook
     68             }
     69             preferredTranscriptionLang?.lowercase()?.let {
     70                 val asrConfig = param.arg<Any>(1)
     71                 asrConfig.getObjectFieldOrNull("mBaseConfig")?.apply {
     72                     setObjectField("mLanguageModel", it)
     73                     setObjectField("mUiLanguage", it)
     74                 }
     75             }
     76         }
     77     }
