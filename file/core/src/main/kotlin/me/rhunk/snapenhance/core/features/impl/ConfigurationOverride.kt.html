     28             fun getConfigKeyInfo(key: Any?) = runCatching {
     29                 if (key == null) return@runCatching null
     30                 val keyClassMethods = key::class.java.methods
     31                 val keyName = keyClassMethods.firstOrNull { it.name == "getName" }?.invoke(key)?.toString() ?: key.toString()
     32                 val category = keyClassMethods.firstOrNull { it.name == configEnumMapping["getCategory"]?.get().toString() }?.invoke(key)?.toString() ?: return null
     33                 val valueHolder = keyClassMethods.firstOrNull { it.name == configEnumMapping["getValue"]?.get().toString() }?.invoke(key) ?: return null
     34                 val defaultValue = valueHolder.getObjectField(configEnumMapping["defaultValueField"]?.get().toString()) ?: return null
     35                 ConfigKeyInfo(category, keyName, defaultValue)
     36             }.onFailure {
     37                 context.log.error("Failed to get config key info", it)
     38             }.getOrNull()

     78             classReference.getAsClass()?.hook(
     79                 getProperty.getAsString()!!,
     80                 HookStage.AFTER
     81             ) { param ->
     82                 val propertyKey = getConfigKeyInfo(param.argNullable<Any>(0)) ?: return@hook
     83

     97                     val valueHolder = XposedHelpers.callMethod(enumData, configEnumMapping["getValue"]?.getAsString())
     98                     valueHolder.setObjectField(configEnumMapping["defaultValueField"]?.getAsString()!!, value)
     99                 }
    100 
    101                 propertyOverrides[key]?.let { (filter, value) ->
    102                     val keyInfo = getConfigKeyInfo(enumData) ?: return@let
    103                     if (!filter(keyInfo)) return@let
    104                     setValue(value(keyInfo))
    105                 }

    111                 appExperimentProvider["getBooleanAppExperimentClass"]?.getAsClass()
    112                     ?.hook("invoke", HookStage.BEFORE) { param ->
    113                         val keyInfo = getConfigKeyInfo(param.arg(1)) ?: return@hook
    114                         if (customBooleanPropertyRules.any { it(keyInfo) }) {
    115                             param.setResult(true)
    116                             return@hook
    117                         }
    118                         propertyOverrides[keyInfo.name]?.let { (filter, value, isAppExperiment) ->
    119                             if (isAppExperiment != true || !filter(keyInfo)) return@let
    120                             param.setResult(value(keyInfo))

    122                     }

    129                     val appExperimentProviderInstance = instance::class.java.fields.firstOrNull {
    130                         appExperimentProvider["class"]?.getAsClass()?.isAssignableFrom(it.type) == true
    131                     }?.get(instance) ?: return@ephemeralHookConstructor
    132 
    133                     appExperimentProviderInstance::class.java.methods.first {
    134                         it.name == appExperimentProvider["hasExperimentMethod"]?.getAsString().toString()
    135                     }.hook(HookStage.BEFORE) { param ->
    136                         val keyInfo = getConfigKeyInfo(param.arg(0)) ?: return@hook
    137                         if (customBooleanPropertyRules.any { it(keyInfo) }) {
    138                             param.setResult(true)
    139                             return@hook
    140                         }
    141 
    142                         val propertyOverride = propertyOverrides[keyInfo.name] ?: return@hook
    143                         propertyOverride.isAppExperiment.takeIf { propertyOverride.filter(keyInfo) }?.let { param.setResult(it) }
    144                     }
    145                 }
    146             }.onFailure {
