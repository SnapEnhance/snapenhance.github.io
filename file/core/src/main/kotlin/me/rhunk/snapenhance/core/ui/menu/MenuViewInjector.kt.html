     20 class MenuViewInjector : Feature("MenuViewInjector") {
     21     private val menuMap by lazy {
     22         arrayOf(
     23             NewChatActionMenu(),
     24             OperaContextActionMenu(),
     25             OperaViewerIcons(),
     26             SettingsGearInjector(),

     28             ChatActionMenu(),

     33             it::class

     37     @Suppress("UNCHECKED_CAST")
     38     fun <T: AbstractMenu> menu(menuClass: KClass<T>): T? {
     39         return menuMap[menuClass] as? T
     40     }

     42     override fun init() {

     45

     47 
     48             val actionSheetItemsContainerLayoutId = context.resources.getIdentifier("action_sheet_items_container", "id")

     50             val actionMenu = context.resources.getIdentifier("action_menu", "id")
     51             val componentsHolder = context.resources.getIdentifier("components_holder", "id")
     52             val feedNewChat = context.resources.getIdentifier("feed_new_chat", "id")
     53             val hovaNavMapIcon = context.resources.getIdentifier("hova_header_search_icon", "id")

     59             context.event.subscribe(AddViewEvent::class) { event ->
     60                 val originalAddView: (View) -> Unit = {
     61                     event.adapter.invokeOriginal(arrayOf(it, -1,
     62                         FrameLayout.LayoutParams(
     63                             ViewGroup.LayoutParams.MATCH_PARENT,
     64                             ViewGroup.LayoutParams.MATCH_PARENT
     65                         ))
     66                     )
     67                 }
     68 
     69                 val viewGroup: ViewGroup = event.parent
     70                 val childView: View = event.view

     72 
     73                 if (event.view.id == actionSheetItemsContainerLayoutId) {
     74                     event.view.post {
     75                         if (event.parent.findParent(4) {
     76                                 it.findViewById<View>(actionMenuTitle) != null
     77                             } == null) return@post
     78 
     79                         val views = mutableListOf<View>()

     81                             views.add(it)
     82                         }

     84                     }
     85                 }

     88                     menuMap[OperaViewerIcons::class]!!.inject(viewGroup, childView, originalAddView)
     89                 }
     90 
     91                 if (event.parent.id == componentsHolder && (childView.id == feedNewChat || childView.id == hovaNavMapIcon)) {
     92                     menuMap[SettingsGearInjector::class]!!.inject(viewGroup, childView, originalAddView)
     93                     return@subscribe
     94                 }
     95 
     96                 if (viewGroup !is LinearLayout && childView.id == chatActionMenu && context.isDeveloper) {
     97                     event.view = LinearLayout(childView.context).apply {
     98                         orientation = LinearLayout.VERTICAL
     99                         addView(
    100                             (menuMap[NewChatActionMenu::class]!! as NewChatActionMenu).createDebugInfoView(childView.context)
    101                         )
    102                         addView(event.view)
    103                     }
    104                 }
    105 
    106                 if (childView.javaClass.name.endsWith("ChatActionMenuComponent") && hasV2ActionMenu()) {
    107                     (menuMap[NewChatActionMenu::class]!! as NewChatActionMenu).handle(event)
    108                     return@subscribe
    109                 }
    110 
    111                 if (viewGroup.javaClass.name.endsWith("ActionMenuChatItemContainer") && !hasV2ActionMenu()) {
    112                     if (viewGroup.parent == null || viewGroup.parent.parent == null) return@subscribe
    113                     menuMap[ChatActionMenu::class]!!.inject(viewGroup, childView, originalAddView)
    114                     return@subscribe
    115                 }
    116 
    117                 if (viewGroup !is LinearLayout && childView.id == actionMenu && messaging.lastFocusedConversationType == 1) {
    118                     val injectedLayout = LinearLayout(childView.context).apply {
    119                         orientation = LinearLayout.VERTICAL
    120                         gravity = Gravity.BOTTOM
    121                         layoutParams = ViewGroup.LayoutParams(ViewGroup.LayoutParams.MATCH_PARENT, ViewGroup.LayoutParams.MATCH_PARENT)
    122                         addView(childView)
    123                     }
    124 
    125                     event.parent.post {
    126                         injectedLayout.addView(ScrollView(injectedLayout.context).apply {
    127                             layoutParams = LinearLayout.LayoutParams(
    128                                 ViewGroup.LayoutParams.MATCH_PARENT,
    129                                 ViewGroup.LayoutParams.WRAP_CONTENT
    130                             ).apply {

    132                                 setMargins(0, 100, 0, 0)
    133                             }
    134 
    135                             addView(LinearLayout(context).apply {
    136                                 orientation = LinearLayout.VERTICAL
    137                                 menuMap[FriendFeedInfoMenu::class]?.inject(event.parent, injectedLayout) { view ->
    138                                     view.layoutParams = LinearLayout.LayoutParams(
    139                                         ViewGroup.LayoutParams.MATCH_PARENT,
    140                                         ViewGroup.LayoutParams.WRAP_CONTENT
    141                                     ).apply {
    142                                         setMargins(0, 5, 0, 5)
    143                                     }
    144                                     addView(view)
    145                                 }
    146                             })
    147                         }, 0)
    148                     }

    150                     event.view = injectedLayout

    152             }
