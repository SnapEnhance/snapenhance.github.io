     11     inline fun newMethodHook(
     12         stage: HookStage,
     13         crossinline consumer: (HookAdapter) -> Unit,
     14         crossinline filter: ((HookAdapter) -> Boolean) = { true }
     15     ): XC_MethodHook {
     16         return if (stage == HookStage.BEFORE) object : XC_MethodHook() {
     17             override fun beforeHookedMethod(param: MethodHookParam<*>) {
     18                 runCatching {
     19                     HookAdapter(param).takeIf(filter)?.also(consumer)
     20                 }.onFailure {
     21                     AbstractLogger.directError("Failed to execute before hook", it)
     22                 }
     23             }
     24         } else object : XC_MethodHook() {
     25             override fun afterHookedMethod(param: MethodHookParam<*>) {
     26                 runCatching {
     27                     HookAdapter(param).takeIf(filter)?.also(consumer)
     28                 }.onFailure {
     29                     AbstractLogger.directError("Failed to execute after hook", it)
     30                 }
     31             }
     32         }
     33     }

     82     inline fun hookObjectMethod(
     83         clazz: Class<*>,
     84         instance: Any,
     85         methodName: String,
     86         stage: HookStage,
     87         crossinline hookConsumer: (HookAdapter) -> Unit
     88     ): List<() -> Unit> {
     89         val unhooks = mutableSetOf<XC_MethodHook.Unhook>()
     90         hook(clazz, methodName, stage) { param->
     91             if (param.nullableThisObject<Any>().let {
     92                 if (it == null) unhooks.forEach { u -> u.unhook() }
     93                 it != instance
     94             }) return@hook
     95             hookConsumer(param)
     96         }.also { unhooks.addAll(it) }
     97         return unhooks.map {
     98             { it.unhook() }
     99         }
    100     }

    121     ): Set<() -> Unit> {
