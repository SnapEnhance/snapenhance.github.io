     62     suspend fun connect(onFailure: (Throwable) -> Unit): Boolean? {
     63         if (this::service.isInitialized && service.asBinder().pingBinder()) {
     64             return true
     65         }
     66 
     67         val connectionTimeout = 15000L
     68         val retryDelay = 3000L
     69 
     70         return withTimeoutOrNull(connectionTimeout) {
     71             var result: Boolean? = null
     72 
     73             for (retry in 0.. (connectionTimeout / retryDelay).toInt()) {
     74                 result = withTimeoutOrNull(retryDelay) {
     75                     suspendCancellableCoroutine { cancellableContinuation ->
     76                         continuation = cancellableContinuation
     77                         with(context.androidContext) {
     78                             //ensure the remote process is running
     79                             runCatching {
     80                                 startActivity(Intent()
     81                                     .setClassName(Constants.SE_PACKAGE_NAME, "me.rhunk.snapenhance.bridge.ForceStartActivity")
     82                                     .addFlags(Intent.FLAG_ACTIVITY_NEW_TASK or Intent.FLAG_ACTIVITY_MULTIPLE_TASK)
     83                                 )
     84                             }
     85 
     86                             runCatching {
     87                                 val intent = Intent()
     88                                     .setClassName(Constants.SE_PACKAGE_NAME, "me.rhunk.snapenhance.bridge.BridgeService")
     89                                 runCatching {
     90                                     if (this@BridgeClient::service.isInitialized) {
     91                                         unbindService(this@BridgeClient)
     92                                     }
     93                                 }
     94                                 if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
     95                                     bindService(
     96                                         intent,
     97                                         Context.BIND_AUTO_CREATE,
     98                                         Executors.newSingleThreadExecutor(),
     99                                         this@BridgeClient
    100                                     )
    101                                 } else {
    102                                     XposedHelpers.callMethod(
    103                                         this,
    104                                         "bindServiceAsUser",
    105                                         intent,
    106                                         this@BridgeClient,
    107                                         Context.BIND_AUTO_CREATE,
    108                                         Handler(HandlerThread("BridgeClient").apply {
    109                                             start()
    110                                         }.looper),
    111                                         Process.myUserHandle()
    112                                     )
    113                                 }
    114                             }.onFailure {
    115                                 onFailure(it)
    116                                 resumeContinuation(false)
    117                             }
    118                         }
    119                     }
    120                 }
    121                 if (result != null) break
    122             }
    123 
    124             result
    125         }
    126     }

    128     override fun onServiceConnected(name: ComponentName, service: IBinder) {
    129         this.service = BridgeInterface.Stub.asInterface(service)
    130         runBlocking {
    131             onConnectedCallbacks.forEach {
    132                 runCatching {
    133                     it()
    134                 }.onFailure {
    135                     context.log.error("Failed to run onConnectedCallback", it)
    136                 }
    137             }
    138         }
    139         cacheSnapEnhanceApkPath = this.service.applicationApkPath.also {
    140             if (cacheSnapEnhanceApkPath != null && cacheSnapEnhanceApkPath != it) {
    141                 context.log.verbose("Restarting Snapchat due to SnapEnhance update")
    142                 context.softRestartApp()
    143                 return
    144             }
    145         }
    146         resumeContinuation(true)
    147     }

    157     private fun tryReconnect() {
    158         runBlocking {
    159             reconnectSemaphore.withPermit {
    160                 if (service.asBinder().pingBinder()) return@runBlocking
    161                 Log.d("BridgeClient", "service is dead, restarting")
    162                 val canLoad = connect {
    163                     Log.e("BridgeClient", "connection failed", it)
    164                     context.softRestartApp()
    165                 }
    166                 if (canLoad != true) {
    167                     Log.e("BridgeClient", "failed to reconnect to service, result=$canLoad")
    168                     context.softRestartApp()
    169                 }
    170             }
    171         }
    172     }

    174     private fun <T> safeServiceCall(block: () -> T): T {
    175         return runCatching {
    176             block()
    177         }.getOrElse { throwable ->
    178             if (throwable is DeadObjectException) {
    179                 tryReconnect()
    180                 return@getOrElse runCatching {
    181                     block()
    182                 }.getOrElse {
    183                     Log.e("BridgeClient", "service call failed", it)
    184                     if (it is DeadObjectException) {
    185                         context.softRestartApp()
    186                     }
    187                     throw it
    188                 }
    189             }
    190             throw throwable
    191         }
    192     }
