     53     private fun resumeContinuation(state: Boolean) {
     54         runBlocking {
     55             connectSemaphore.withPermit {

     58             }

     62     suspend fun connect(onFailure: (Throwable) -> Unit): Boolean? {
     63         if (this::service.isInitialized && service.asBinder().pingBinder()) {
     64             return true
     65         }
     66 
     67         return withTimeoutOrNull(10000L) {
     68             suspendCancellableCoroutine { cancellableContinuation ->
     69                 continuation = cancellableContinuation
     70                 with(context.androidContext) {
     71                     //ensure the remote process is running
     72                     runCatching {
     73                         startActivity(Intent()
     74                             .setClassName(Constants.SE_PACKAGE_NAME, "me.rhunk.snapenhance.bridge.ForceStartActivity")
     75                             .addFlags(Intent.FLAG_ACTIVITY_NEW_TASK or Intent.FLAG_ACTIVITY_MULTIPLE_TASK)
     76                         )
     77                     }
     78 
     79                     runCatching {
     80                         val intent = Intent()
     81                             .setClassName(Constants.SE_PACKAGE_NAME, "me.rhunk.snapenhance.bridge.BridgeService")
     82                         runCatching {
     83                             if (this@BridgeClient::service.isInitialized) {
     84                                 unbindService(this@BridgeClient)
     85                             }
     86                         }
     87                         if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.Q) {
     88                             bindService(
     89                                 intent,
     90                                 Context.BIND_AUTO_CREATE,
     91                                 Executors.newSingleThreadExecutor(),
     92                                 this@BridgeClient
     93                             )
     94                         } else {
     95                             XposedHelpers.callMethod(
     96                                 this,
     97                                 "bindServiceAsUser",
     98                                 intent,
     99                                 this@BridgeClient,
    100                                 Context.BIND_AUTO_CREATE,
    101                                 Handler(HandlerThread("BridgeClient").apply {
    102                                     start()
    103                                 }.looper),
    104                                 Process.myUserHandle()
    105                             )
    106                         }
    107                     }.onFailure {
    108                         onFailure(it)
    109                         resumeContinuation(false)
    110                     }
    111                 }
    112             }
    113         }
    114     }

    116     override fun onServiceConnected(name: ComponentName, service: IBinder) {
    117         this.service = BridgeInterface.Stub.asInterface(service)
    118         runBlocking {
    119             onConnectedCallbacks.forEach {
    120                 runCatching {
    121                     it()
    122                 }.onFailure {
    123                     context.log.error("Failed to run onConnectedCallback", it)
    124                 }
    125             }
    126         }
    127         cacheSnapEnhanceApkPath = this.service.applicationApkPath.also {
    128             if (cacheSnapEnhanceApkPath != null && cacheSnapEnhanceApkPath != it) {
    129                 context.log.verbose("Restarting Snapchat due to SnapEnhance update")
    130                 context.softRestartApp()
    131                 return
    132             }
    133         }
    134         resumeContinuation(true)
    135     }

    145     private fun tryReconnect() {
    146         runBlocking {
    147             reconnectSemaphore.withPermit {
    148                 if (service.asBinder().pingBinder()) return@runBlocking
    149                 Log.d("BridgeClient", "service is dead, restarting")
    150                 val canLoad = connect {
    151                     Log.e("BridgeClient", "connection failed", it)
    152                     context.softRestartApp()
    153                 }
    154                 if (canLoad != true) {
    155                     Log.e("BridgeClient", "failed to reconnect to service, result=$canLoad")
    156                     context.softRestartApp()
    157                 }
    158             }
    159         }
    160     }

    162     private fun <T> safeServiceCall(block: () -> T): T {
    163         return runCatching {
    164             block()
    165         }.getOrElse { throwable ->
    166             if (throwable is DeadObjectException) {
    167                 tryReconnect()
    168                 return@getOrElse runCatching {
    169                     block()
    170                 }.getOrElse {
    171                     Log.e("BridgeClient", "service call failed", it)
    172                     if (it is DeadObjectException) {
    173                         context.softRestartApp()
    174                     }
    175                     throw it
    176                 }
    177             }
    178             throw throwable
    179         }
    180     }
