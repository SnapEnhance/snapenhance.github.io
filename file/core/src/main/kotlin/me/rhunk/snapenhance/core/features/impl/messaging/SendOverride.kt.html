     41                             if (stripSnapMetadata.contains("hide_caption_text")) {

     46                                 }
     47                             }
     48                             if (stripSnapMetadata.contains("hide_snap_filters")) {

     51                             }
     52                             if (stripSnapMetadata.contains("hide_extras")) {
     53                                 remove(13)
     54                                 edit(5, 1) {
     55                                     remove(2)
     56                                 }
     57                             }
     58                         }
     59                     }

     62                             edit(6, 1, 1) {
     63                                 remove(13)

     66                         if (stripSnapMetadata.contains("remove_audio_note_transcript_capability")) {
     67                             edit(6, 1) {
     68                                 remove(3)
     69                             }
     70                         }
     71                     }

     77         }
     78 
     79         val configOverrideType = context.config.messaging.galleryMediaSendOverride.getNullable() ?: return
     80 
     81         context.event.subscribe(MediaUploadEvent::class) { event ->
     82             ProtoReader(event.localMessageContent.content!!).followPath(11, 5)?.let { snapDocPlayback ->
     83                 event.onMediaUploaded { result ->
     84                     result.messageContent.content = ProtoEditor(result.messageContent.content!!).apply {
     85                         edit(11, 5) {
     86                             // remove media upload hint when viewing snap
     87                             edit(1) {
     88                                 edit(1) {
     89                                     remove(27)
     90                                     addBuffer(26, byteArrayOf())
     91                                 }
     92                             }
     93 
     94                             remove(2)
     95                             snapDocPlayback.getByteArray(2)?.let {
     96                                 addBuffer(2, it)
     97                             }
     98                         }
     99                     }.toByteArray()
    100                 }
    101             }
    102         }
    103 
    104         context.event.subscribe(NativeUnaryCallEvent::class) { event ->
    105             if (event.uri != "/messagingcoreservice.MessagingCoreService/CreateContentMessage") return@subscribe
    106             postSavePolicy?.let { savePolicy ->
    107                 context.log.verbose("post save policy $savePolicy")
    108                 event.buffer = ProtoEditor(event.buffer).apply {
    109                     edit {
    110                         edit(4) {
    111                             remove(7)
    112                             addVarInt(7, savePolicy)

    114                         add(6) {
    115                             from(9) {
    116                                 addVarInt(1, 1)
    117                             }
    118                         }
    119                     }
    120                 }.toByteArray()
    121             }
    122         }
    123 
    124         context.event.subscribe(SendMessageWithContentEvent::class) { event ->
    125             postSavePolicy = null
    126             if (event.destinations.stories?.isNotEmpty() == true && event.destinations.conversations?.isEmpty() == true) return@subscribe
    127             val localMessageContent = event.messageContent
    128             if (localMessageContent.contentType != ContentType.EXTERNAL_MEDIA) return@subscribe
    129 
    130             //prevent story replies
    131             val messageProtoReader = ProtoReader(localMessageContent.content!!)

    136             fun sendMedia(overrideType: String): Boolean {
    137                 if (overrideType != "ORIGINAL" && messageProtoReader.followPath(3)?.getCount(3) != 1) {
    138                     context.inAppOverlay.showStatusToast(
    139                         icon = Icons.Default.WarningAmber,
    140                         context.translation["gallery_media_send_override.multiple_media_toast"]
    141                     )
    142                     return false
    143                 }
    144 
    145                 when (overrideType) {
    146                     "SNAP", "SAVEABLE_SNAP" -> {
    147                         postSavePolicy = if (overrideType == "SAVEABLE_SNAP") 3 /* VIEW_SESSION */ else 1 /* PROHIBITED */
    148 
    149                         val extras = messageProtoReader.followPath(3, 3, 13)?.getBuffer()
    150                         localMessageContent.contentType = ContentType.SNAP
    151                         localMessageContent.content = ProtoWriter().apply {
    152                             from(11) {
    153                                 from(5) {
    154                                     from(1) {
    155                                         from(1) {
    156                                             addVarInt(2, 0)
    157                                             addVarInt(12, 0)
    158                                             addVarInt(15, 0)
    159                                         }
    160                                         addVarInt(6, 0)
    161                                     }
    162                                     messageProtoReader.getByteArray(3, 3, 5, 2)?.let {
    163                                         addBuffer(2, it)
    164                                     }
    165                                 }
    166                                 extras?.let {
    167                                     addBuffer(13, it)
    168                                 }
    169                             }
    170                         }.toByteArray()
    171                     }
    172                     "NOTE" -> {
    173                         localMessageContent.contentType = ContentType.NOTE
    174                         localMessageContent.content =
    175                             MessageSender.audioNoteProto(
    176                                 messageProtoReader.getVarInt(3, 3, 5, 1, 1, 15) ?: context.feature(MediaFilePicker::class).lastMediaDuration ?: 0,
    177                                 Locale.getDefault().toLanguageTag()
    178                             )
    179                     }
    180                 }
    181 
    182                 return true
    183             }
    184 
    185             if (configOverrideType != "always_ask") {
    186                 if (sendMedia(configOverrideType)) {
    187                     event.invokeOriginal()
    188                 }
    189                 return@subscribe
    190             }
    191 
    192             context.runOnUiThread {
    193                 ViewAppearanceHelper.newAlertDialogBuilder(context.mainActivity!!)
    194                     .setItems(typeNames.values.map {
    195                         context.translation["features.options.gallery_media_send_override.$it"]
    196                     }.toTypedArray()) { dialog, which ->
    197                         dialog.dismiss()
    198                         if (sendMedia(typeNames.keys.toTypedArray()[which])) {
    199                             event.invokeOriginal()
    200                         }
    201                     }
    202                     .setTitle(context.translation["send_override_dialog.title"])
    203                     .setNegativeButton(context.translation["button.cancel"], null)

    205             }
    206         }
