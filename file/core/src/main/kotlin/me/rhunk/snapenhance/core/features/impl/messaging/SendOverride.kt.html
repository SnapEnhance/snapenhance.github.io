     17 class SendOverride : Feature("Send Override") {
     18     private var isLastSnapSavable = false
     19     private val typeNames by lazy {
     20         mutableListOf("ORIGINAL", "SNAP", "NOTE").also {
     21             if (NativeLib.initialized) {
     22                 it.add("SAVABLE_SNAP")
     23             }
     24         }.associateWith { it }
     25     }

     27     override fun init() {
     28         val stripSnapMetadata = context.config.messaging.stripMediaMetadata.get()
     29 
     30         context.event.subscribe(SendMessageWithContentEvent::class, {
     31             stripSnapMetadata.isNotEmpty()
     32         }) { event ->
     33             val contentType = event.messageContent.contentType ?: return@subscribe
     34 
     35             val newMessageContent = ProtoEditor(event.messageContent.content!!).apply {
     36                 when (contentType) {
     37                     ContentType.SNAP, ContentType.EXTERNAL_MEDIA -> {
     38                         edit(*(if (contentType == ContentType.SNAP) intArrayOf(11) else intArrayOf(3, 3))) {
     39                             if (stripSnapMetadata.contains("hide_caption_text")) {
     40                                 edit(5) {
     41                                     editEach(1) {
     42                                         remove(2)
     43                                     }
     44                                 }
     45                             }
     46                             if (stripSnapMetadata.contains("hide_snap_filters")) {
     47                                 remove(9)
     48                                 remove(11)
     49                             }
     50                             if (stripSnapMetadata.contains("hide_extras")) {
     51                                 remove(13)
     52                                 edit(5, 1) {
     53                                     remove(2)
     54                                 }
     55                             }
     56                         }
     57                     }
     58                     ContentType.NOTE -> {
     59                         if (stripSnapMetadata.contains("remove_audio_note_duration")) {
     60                             edit(6, 1, 1) {
     61                                 remove(13)
     62                             }
     63                         }
     64                         if (stripSnapMetadata.contains("remove_audio_note_transcript_capability")) {
     65                             edit(6, 1) {
     66                                 remove(3)
     67                             }
     68                         }
     69                     }
     70                     else -> return@subscribe
     71                 }
     72             }.toByteArray()
     73 
     74             event.messageContent.content = newMessageContent
     75         }
     76 
     77         val configOverrideType = context.config.messaging.galleryMediaSendOverride.getNullable() ?: return
     78 
     79         context.event.subscribe(NativeUnaryCallEvent::class) { event ->
     80             if (event.uri != "/messagingcoreservice.MessagingCoreService/CreateContentMessage") return@subscribe
     81             if (isLastSnapSavable) {
     82                 event.buffer = ProtoEditor(event.buffer).apply {
     83                     edit {
     84                         edit(4) {
     85                             remove(7)
     86                             addVarInt(7, 3) // savePolicy = VIEW_SESSION
     87                         }
     88                         add(6) {
     89                             from(9) {
     90                                 addVarInt(1, 1)
     91                             }
     92                         }
     93                     }
     94                 }.toByteArray()
     95             }
     96         }
     97 
     98         context.event.subscribe(SendMessageWithContentEvent::class) { event ->
     99             isLastSnapSavable = false
    100             if (event.destinations.stories?.isNotEmpty() == true && event.destinations.conversations?.isEmpty() == true) return@subscribe
    101             val localMessageContent = event.messageContent
    102             if (localMessageContent.contentType != ContentType.EXTERNAL_MEDIA) return@subscribe
    103 
    104             //prevent story replies
    105             val messageProtoReader = ProtoReader(localMessageContent.content!!)
    106             if (messageProtoReader.contains(7)) return@subscribe
    107 
    108             event.canceled = true
    109 
    110             fun sendMedia(overrideType: String): Boolean {
    111                 if (overrideType != "ORIGINAL" && messageProtoReader.followPath(3)?.getCount(3) != 1) {
    112                     context.inAppOverlay.showStatusToast(
    113                         icon = Icons.Default.WarningAmber,
    114                         context.translation["gallery_media_send_override.multiple_media_toast"]
    115                     )
    116                     return false
    117                 }
    118 
    119                 when (overrideType) {
    120                     "SNAP", "SAVABLE_SNAP" -> {
    121                         val extras = messageProtoReader.followPath(3, 3, 13)?.getBuffer()
    122 
    123                         localMessageContent.contentType = ContentType.SNAP
    124                         localMessageContent.content = MessageSender.redSnapProto(extras)
    125                         if (overrideType == "SAVABLE_SNAP") {
    126                             isLastSnapSavable = true
    127                         }
    128                     }
    129                     "NOTE" -> {
    130                         localMessageContent.contentType = ContentType.NOTE
    131                         localMessageContent.content =
    132                             MessageSender.audioNoteProto(
    133                                 messageProtoReader.getVarInt(3, 3, 5, 1, 1, 15) ?: context.feature(MediaFilePicker::class).lastMediaDuration ?: 0,
    134                                 Locale.getDefault().toLanguageTag()
    135                             )
    136                     }
    137                 }
    138 
    139                 return true
    140             }
    141 
    142             if (configOverrideType != "always_ask") {
    143                 if (sendMedia(configOverrideType)) {
    144                     event.invokeOriginal()
    145                 }
    146                 return@subscribe
    147             }
    148 
    149             context.runOnUiThread {
    150                 ViewAppearanceHelper.newAlertDialogBuilder(context.mainActivity!!)
    151                     .setItems(typeNames.values.map {
    152                         context.translation["features.options.gallery_media_send_override.$it"]
    153                     }.toTypedArray()) { dialog, which ->
    154                         dialog.dismiss()
    155                         if (sendMedia(typeNames.keys.toTypedArray()[which])) {
    156                             event.invokeOriginal()
    157                         }
    158                     }
    159                     .setNegativeButton(context.translation["button.cancel"], null)
    160                     .show()
    161             }
    162         }
    163     }
