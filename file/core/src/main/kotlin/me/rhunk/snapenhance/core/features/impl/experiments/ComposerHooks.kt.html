     92     private fun newComposerFunction(block: ComposerMarshaller.() -> Boolean): Any? {
     93         val composerFunctionClass = findClass("com.snap.composer.callable.ComposerFunction")
     94         return Proxy.newProxyInstance(
     95             composerFunctionClass.classLoader,
     96             arrayOf(composerFunctionClass)
     97         ) { _, method, args ->
     98             if (method.name != "perform") return@newProxyInstance null
     99             block(ComposerMarshaller(args?.get(0) ?: return@newProxyInstance false))
    100         }
    101     }

    104     override fun init() {
    105         if (config.globalState != true) return

    107         val importedFunctions = mutableMapOf<String, Any?>()

    109         fun composerFunction(name: String, block: ComposerMarshaller.() -> Unit) {
    110             importedFunctions[name] = newComposerFunction {
    111                 block(this)
    112                 true
    113             }
    114         }

    116         composerFunction("getConfig") {

    118                 "operaDownloadButton" to context.config.downloader.operaDownloadButton.get(),
    119                 "bypassCameraRollLimit" to config.bypassCameraRollLimit.get(),
    120                 "showFirstCreatedUsername" to config.showFirstCreatedUsername.get(),
    121                 "composerLogs" to config.composerLogs.get()

    123         }

    125         composerFunction("showToast") {
    126             if (getSize() < 1) return@composerFunction
    127             context.shortToast(getUntyped(0) as? String ?: return@composerFunction)
    128         }
    129 
    130         composerFunction("downloadLastOperaMedia") {
    131             context.feature(MediaDownloader::class).downloadLastOperaMediaAsync(getUntyped(0) == true)
    132         }
    133 
    134         composerFunction("getFriendInfoByUsername") {
    135             if (getSize() < 1) return@composerFunction
    136             val username = getUntyped(0) as? String ?: return@composerFunction
    137             runCatching {
    138                 pushUntyped(context.database.getFriendInfoByUsername(username)?.let {
    139                     context.gson.toJson(it)

    141             }.onFailure {

    146         composerFunction("log") {
    147             if (getSize() < 2) return@composerFunction
    148             val logLevel = getUntyped(0) as? String ?: return@composerFunction
    149             val message = getUntyped(1) as? String ?: return@composerFunction
    150 
    151             val tag = "ComposerLogs"
    152 
    153             when (logLevel) {
    154                 "log" -> context.log.verbose(message, tag)
    155                 "debug" -> context.log.debug(message, tag)
    156                 "info" -> context.log.info(message, tag)
    157                 "warn" -> context.log.warn(message, tag)
    158                 "error" -> context.log.error(message, tag)
    159             }
    160         }

    162         fun loadHooks() {
    163             if (!NativeLib.initialized) {
    164                 context.log.error("ComposerHooks cannot be loaded without NativeLib")
    165                 return
    166             }
    167             val loaderScript = runCatching {
    168                 context.fileHandlerManager.getFileHandle(FileHandleScope.COMPOSER.key, "loader.js").toWrapper().readBytes().toString(Charsets.UTF_8)
    169             }.onFailure {
    170                 context.log.error("Failed to load composer loader script", it)
    171             }.getOrNull() ?: return
    172             context.native.setComposerLoader("""
    173                 const i = setInterval(() => {
    174                     try {
    175                         require('composer_core/src/DeviceBridge').getDisplayWidth();
    176                         clearInterval(i);
    177                         (() => { const _getImportsFunctionName = "$getImportsFunctionName"; $loaderScript })();
    178                     } catch (e) {}
    179                 }, 200)
    180             """.trimIndent().trim())
    181         }
