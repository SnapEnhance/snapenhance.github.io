    133     private fun newComposerFunction(block: ComposerMarshaller.() -> Boolean): Any? {
    134         val composerFunctionClass = findClass("com.snap.composer.callable.ComposerFunction")
    135         return Proxy.newProxyInstance(
    136             composerFunctionClass.classLoader,
    137             arrayOf(composerFunctionClass)
    138         ) { _, method, args ->
    139             if (method.name != "perform") return@newProxyInstance null
    140             block(ComposerMarshaller(args?.get(0) ?: return@newProxyInstance false))
    141         }
    142     }

    145     override fun init() {
    146         if (config.globalState != true) return

    148         val importedFunctions = mutableMapOf<String, Any?>()

    150         fun composerFunction(name: String, block: ComposerMarshaller.() -> Unit) {
    151             importedFunctions[name] = newComposerFunction {
    152                 block(this)
    153                 true

    157         composerFunction("getConfig") {

    175         composerFunction("getFriendInfoByUsername") {
    176             if (getSize() < 1) return@composerFunction

    187         composerFunction("log") {
    188             if (getSize() < 2) return@composerFunction
    189             val logLevel = getUntyped(0) as? String ?: return@composerFunction
    190             val message = getUntyped(1) as? String ?: return@composerFunction
    191 
    192             val tag = "ComposerLogs"
    193 
    194             when (logLevel) {
    195                 "log" -> context.log.verbose(message, tag)
    196                 "debug" -> context.log.debug(message, tag)
    197                 "info" -> context.log.info(message, tag)
    198                 "warn" -> context.log.warn(message, tag)
    199                 "error" -> context.log.error(message, tag)
    200             }
    201         }

    203         fun loadHooks() {
    204             if (!NativeLib.initialized) {
    205                 context.log.error("ComposerHooks cannot be loaded without NativeLib")
    206                 return
    207             }
    208             val loaderScript = runCatching {
    209                 context.fileHandlerManager.getFileHandle(FileHandleScope.COMPOSER.key, "loader.js").toWrapper().readBytes().toString(Charsets.UTF_8)
    210             }.onFailure {
    211                 context.log.error("Failed to load composer loader script", it)
    212             }.getOrNull() ?: return
    213             context.native.setComposerLoader("""
    214                 (() => { const _getImportsFunctionName = "$getImportsFunctionName"; $loaderScript })();
    215             """.trimIndent().trim())
    216         }
    217 
    218         loadHooks()
    219
