     29 fun Context.getTextFromClipboard(): String? {
     30     return runCatching {
     31         getSystemService(android.content.ClipboardManager::class.java).primaryClip
     32             ?.takeIf { it.itemCount > 0 }
     33             ?.getItemAt(0)
     34             ?.text?.toString()
     35     }.getOrNull()
     36 }

     42 fun InputStream.toParcelFileDescriptor(coroutineScope: CoroutineScope): ParcelFileDescriptor {
     43     val pfd = ParcelFileDescriptor.createPipe()

     45 
     46     coroutineScope.launch(Dispatchers.IO) {
