     25     private fun readVarInt(): Long {
     26         var result = 0L
     27         var shift = 0
     28         while (true) {
     29             val b = readByte()
     30             result = result or ((b.toLong() and 0x7F) shl shift)
     31             if (b.toInt() and 0x80 == 0) {
     32                 break
     33             }
     34             shift += 7
     35         }
     36         return result
     37     }

     39     private fun read() {
     40         while (offset < buffer.size) {
     41             try {
     42                 val tag = readVarInt().toInt()
     43                 val id = tag ushr 3
     44                 val type = WireType.fromValue(tag and 0x7) ?: break
     45                 val value = when (type) {
     46                     WireType.VARINT -> readVarInt()
     47                     WireType.FIXED64 -> {
     48                         val bytes = ByteArray(8)
     49                         for (i in 0..7) {
     50                             bytes[i] = readByte()
     51                         }
     52                         bytes
     53                     }
     54                     WireType.CHUNK -> {
     55                         val length = readVarInt().toInt()
     56                         val bytes = ByteArray(length)
     57                         for (i in 0 until length) {
     58                             bytes[i] = readByte()
     59                         }
     60                         bytes
     61                     }
     62                     WireType.START_GROUP -> {
     63                         val bytes = mutableListOf<Byte>()
     64                         while (true) {
     65                             val b = readByte()
     66                             if (b.toInt() == WireType.END_GROUP.value) {
     67                                 break
     68                             }
     69                             bytes.add(b)
     70                         }
     71                         bytes.toByteArray()
     72                     }
     73                     WireType.FIXED32 -> {
     74                         val bytes = ByteArray(4)
     75                         for (i in 0..3) {
     76                             bytes[i] = readByte()
     77                         }
     78                         bytes
     79                     }
     80                     WireType.END_GROUP -> continue
     81                 }
     82                 values.getOrPut(id) { mutableListOf() }.add(Wire(id, type, value))
     83             } catch (t: Throwable) {
     84                 values.clear()
     85                 break
     86             }
     87         }
     88     }

    204     private fun prettyPrint(tabSize: Int): String {
    205         val tabLine = "    ".repeat(tabSize)
    206         val stringBuilder = StringBuilder()
    207         values.forEach v@{ (id, wires) ->
    208             wires.forEach { wire ->
    209                 stringBuilder.append(tabLine)
    210                 stringBuilder.append("$id <${wire.type.name.lowercase()}> = ")
    211                 when (wire.type) {
    212                     WireType.VARINT -> stringBuilder.append("${wire.value}\n")
    213                     WireType.FIXED64, WireType.FIXED32 -> {
    214                         val byteBuffer = ByteBuffer.wrap(wire.value as ByteArray).order(java.nio.ByteOrder.LITTLE_ENDIAN)
    215                         val hexValue = wire.value.joinToString("") { byte -> "%02x".format(byte) }
    216                         val intValue = if (wire.type == WireType.FIXED32) byteBuffer.int else byteBuffer.long
    217                         byteBuffer.position(0)
    218                         val decimalValue = if (wire.type == WireType.FIXED32) byteBuffer.float else byteBuffer.double
    219                         stringBuilder.append("$intValue/0x$hexValue/$decimalValue\n")
    220                     }
    221                     WireType.CHUNK -> {
    222                         val array = (wire.value as? ByteArray) ?: return@forEach
    223 
    224                         fun printArray() {
    225                             // auto detect uuids
    226                             if (array.size == 16) {
    227                                 val longs = LongArray(2)
    228                                 for (i in 0 .. 7) {
    229                                     longs[0] = longs[0] or ((array[i].toLong() and 0xFF) shl ((7 - i) * 8))
    230                                 }
    231                                 for (i in 8 .. 15) {
    232                                     longs[1] = longs[1] or ((array[i].toLong() and 0xFF) shl ((15 - i) * 8))
    233                                 }
    234                                 stringBuilder.append("uuid: ${UUID(longs[0], longs[1])}\n")
    235                                 return
    236                             }
    237 
    238                             //auto detect ascii strings
    239                             if (array.all { it in (0x20..0x7E) || it == 0x0A.toByte() || it == 0x0D.toByte() }) {
    240                                 stringBuilder.append("string: ${array.toString(Charsets.UTF_8)}\n")
    241                                 return
    242                             }
    243 
    244                             stringBuilder.append("\n")
    245                             stringBuilder.append("$tabLine    ")
    246                             stringBuilder.append(array.joinToString(" ") { byte -> "%02x".format(byte) })
    247                             stringBuilder.append("\n")
    248                         }
    249 
    250                         runCatching {
    251                             if (array.isEmpty()) {
    252                                 stringBuilder.append("empty\n")
    253                                 return@runCatching
    254                             }
    255 
    256                             ProtoReader(array).prettyPrint(tabSize + 1).takeIf { it.isNotEmpty() }?.let {
    257                                 stringBuilder.append("message:\n")
    258                                 stringBuilder.append(it)
    259                             } ?: printArray()
    260                         }.onFailure {
    261                             printArray()
    262                         }
    263                     }
    264                     else -> stringBuilder.append("unknown\n")
    265                 }
    266             }
    267         }
    268 
    269         return stringBuilder.toString()
    270     }
