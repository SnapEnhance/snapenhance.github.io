    137     fun propertyOption(translation: LocaleWrapper, key: String): String {
    138         if (key == "null") {
    139             return translation[params.disabledKey?.let { disabledKey ->
    140                 params.customOptionTranslationPath?.let {
    141                     "$it.$disabledKey"
    142                 } ?: key
    143             } ?: "manager.sections.features.disabled"]
    144         }
    145 
    146         return if (!params.flags.contains(ConfigFlag.NO_TRANSLATE))
    147             translation[params.customOptionTranslationPath?.let {
    148                 "$it.$key"
    149             } ?: "features.options.${name}.$key"]
    150         else key
    151     }
    152 
    153     fun propertyName() = propertyTranslationPath() + ".name"
    154     fun propertyDescription() = propertyTranslationPath() + ".description"
    155 
    156     fun propertyTranslationPath(): String {
    157         params.customTranslationPath?.let {
    158             return it
    159         }
    160         return parentKey?.let {
    161             "${it.propertyTranslationPath()}.properties.$name"
    162         } ?: "features.properties.$name"
