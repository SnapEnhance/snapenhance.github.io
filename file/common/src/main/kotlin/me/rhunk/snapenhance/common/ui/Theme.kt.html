    141 @Composable
    142 fun AppMaterialTheme(
    143     isDarkTheme: Boolean = isSystemInDarkTheme(),
    144     content: @Composable () -> Unit
    145 ) {
    146     val dynamicColor = Build.VERSION.SDK_INT >= Build.VERSION_CODES.S
    147     val colorScheme = when {
    148         dynamicColor && isDarkTheme -> {
    149             dynamicDarkColorScheme(LocalContext.current)
    150         }
    151         dynamicColor && !isDarkTheme -> {
    152             dynamicLightColorScheme(LocalContext.current)
    153         }
    154         !isDarkTheme -> LightThemeColors
    155         else -> DarkThemeColors
    156     }
    157 
    158     MaterialTheme(
    159         colorScheme = colorScheme,
    160         content = content
    161     )
