     40     fun load(block: ScriptableObject.() -> Unit) {
     41         contextScope {
     42             val classLoader = scriptRuntime.androidContext.classLoader
     43             moduleObject = initSafeStandardObjects()
     44             moduleObject.putConst("module", moduleObject, scriptableObject {
     45                 putConst("info", this, scriptableObject {
     46                     putConst("name", this, moduleInfo.name)
     47                     putConst("version", this, moduleInfo.version)
     48                     putConst("displayName", this, moduleInfo.displayName)
     49                     putConst("description", this, moduleInfo.description)
     50                     putConst("author", this, moduleInfo.author)
     51                     putConst("minSnapchatVersion", this, moduleInfo.minSnapchatVersion)
     52                     putConst("minSEVersion", this, moduleInfo.minSEVersion)
     53                     putConst("grantedPermissions", this, moduleInfo.grantedPermissions)
     54                 })
     55             })
     56 
     57             scriptRuntime.logger.apply {
     58                 moduleObject.putConst("console", moduleObject, scriptableObject {
     59                     putFunction("log") { info(argsToString(it)) }
     60                     putFunction("warn") { warn(argsToString(it)) }
     61                     putFunction("error") { error(argsToString(it)) }
     62                     putFunction("debug") { debug(argsToString(it)) }
     63                     putFunction("info") { info(argsToString(it)) }
     64                     putFunction("trace") { verbose(argsToString(it)) }
     65                     putFunction("verbose") { verbose(argsToString(it)) }
     66                 })
     67             }
     68 
     69             registerBindings(
     70                 JavaInterfaces(),
     71                 InterfaceManager(),
     72                 Networking(),
     73                 Protobuf()
     74             )
     75 
     76             moduleObject.putFunction("setField") { args ->
     77                 val obj = args?.get(0) ?: return@putFunction Undefined.instance
     78                 val name = args[1].toString()
     79                 val value = args[2]
     80                 val field = obj.javaClass.declaredFields.find { it.name == name } ?: return@putFunction Undefined.instance
     81                 field.isAccessible = true
     82                 field.set(obj, value.toPrimitiveValue(lazy { field.type.name }))
     83                 Undefined.instance
     84             }
     85 
     86             moduleObject.putFunction("getField") { args ->
     87                 val obj = args?.get(0) ?: return@putFunction Undefined.instance
     88                 val name = args[1].toString()
     89                 val field = obj.javaClass.declaredFields.find { it.name == name } ?: return@putFunction Undefined.instance
     90                 field.isAccessible = true
     91                 field.get(obj)
     92             }
     93 
     94             moduleObject.putFunction("sleep") { args ->
     95                 val time = args?.get(0) as? Number ?: return@putFunction Undefined.instance
     96                 Thread.sleep(time.toLong())
     97                 Undefined.instance
     98             }
     99 
    100             moduleObject.putFunction("findClass") {
    101                 val className = it?.get(0).toString()
    102                 val useModClassLoader = it?.getOrNull(1) as? Boolean ?: false
    103                 if (useModClassLoader) moduleInfo.ensurePermissionGranted(Permissions.UNSAFE_CLASSLOADER)
    104 
    105                 runCatching {
    106                     if (useModClassLoader) this::class.java.classLoader?.loadClass(className)
    107                     else classLoader.loadClass(className)
    108                 }.onFailure { throwable ->
    109                     scriptRuntime.logger.error("Failed to load class $className", throwable)
    110                 }.getOrNull()
    111             }
    112 
    113             moduleObject.putFunction("type") { args ->
    114                 val className = args?.get(0).toString()
    115                 val useModClassLoader = args?.getOrNull(1) as? Boolean ?: false
    116                 if (useModClassLoader) moduleInfo.ensurePermissionGranted(Permissions.UNSAFE_CLASSLOADER)
    117 
    118                 val clazz = runCatching {
    119                     if (useModClassLoader) this::class.java.classLoader?.loadClass(className) else classLoader.loadClass(className)
    120                 }.getOrNull() ?: return@putFunction Undefined.instance
    121 
    122                 scriptableObject("JavaClassWrapper") {
    123                     val newInstance: (Array<out Any?>?) -> Any? = { args ->
    124                         val constructor = clazz.declaredConstructors.find {
    125                             (args ?: emptyArray()).isSameParameters(it.parameterTypes)
    126                         }?.also { it.isAccessible = true } ?: throw IllegalArgumentException("Constructor not found with args ${argsToString(args)}")
    127                         constructor.newInstance(*args ?: emptyArray())
    128                     }
    129                     putFunction("__new__") { newInstance(it) }
    130 
    131                     clazz.declaredMethods.filter { Modifier.isStatic(it.modifiers) }.forEach { method ->
    132                         putFunction(method.name) { args ->
    133                             val declaredMethod = clazz.declaredMethods.find {
    134                                 it.name == method.name && (args ?: emptyArray()).isSameParameters(it.parameterTypes)
    135                             }?.also { it.isAccessible = true } ?: throw IllegalArgumentException("Method ${method.name} not found with args ${argsToString(args)}")
    136                             declaredMethod.invoke(null, *args ?: emptyArray())
    137                         }
    138                     }
    139 
    140                     clazz.declaredFields.filter { Modifier.isStatic(it.modifiers) }.forEach { field ->
    141                         field.isAccessible = true
    142                         defineProperty(field.name, { field.get(null) }, { value -> field.set(null, value) }, 0)
    143                     }
    144 
    145                     if (get("newInstance") == null) {
    146                         putFunction("newInstance") { newInstance(it) }
    147                     }
    148                 }
    149             }
    150 
    151             moduleObject.putFunction("logInfo") { args ->
    152                 scriptRuntime.logger.info(argsToString(args))
    153                 Undefined.instance
    154             }
    155 
    156             moduleObject.putFunction("logError") { args ->
    157                 scriptRuntime.logger.error(argsToString(arrayOf(args?.get(0))), args?.getOrNull(1) as? Throwable ?: Throwable())
    158                 Undefined.instance
    159             }
    160 
    161             for (toastFunc in listOf("longToast", "shortToast")) {
    162                 moduleObject.putFunction(toastFunc) { args ->
    163                     Handler(scriptRuntime.androidContext.mainLooper).post {
    164                         Toast.makeText(
    165                             scriptRuntime.androidContext,
    166                             args?.joinToString(" ") ?: "",
    167                             if (toastFunc == "longToast") Toast.LENGTH_LONG else Toast.LENGTH_SHORT
    168                         ).show()
    169                     }
    170                     Undefined.instance
    171                 }
    172             }
    173 
    174             block(moduleObject)
    175 
    176             moduleBindings.forEach { (_, instance) ->
    177                 instance.context = moduleBindingContext
    178 
    179                 runCatching {
    180                     instance.onInit()
    181                 }.onFailure {
    182                     scriptRuntime.logger.error("Failed to init binding ${instance.name}", it)
    183                 }
    184             }
    185 
    186             moduleObject.putFunction("require") { args ->
    187                 val bindingName = args?.get(0).toString()
    188                 val (namespace, path) = bindingName.takeIf {
    189                     it.startsWith("@") && it.contains("/")
    190                 }?.let {
    191                     it.substring(1).substringBefore("/") to it.substringAfter("/")
    192                 } ?: (null to "")
    193 
    194                 when (namespace) {
    195                     "modules" -> scriptRuntime.getModuleByName(path)?.moduleObject?.scriptable("module")?.scriptable("exports")
    196                     else -> moduleBindings[bindingName]?.getObject()
    197                 }
    198             }
    199         }
    200 
    201         contextScope(shouldOptimize = true) {
    202             evaluateReader(moduleObject, reader, moduleInfo.name, 1, null)
    203         }
    204     }

    206     fun unload() {
    207         callFunction("module.onUnload")
    208         moduleBindings.entries.removeIf { (name, binding) ->
    209             runCatching {
    210                 binding.onDispose()
    211             }.onFailure {
    212                 scriptRuntime.logger.error("Failed to dispose binding $name", it)
    213             }
    214             true
    215         }
    216     }

    218     fun callFunction(name: String, vararg args: Any?) {
    219         contextScope {
    220             name.split(".").also { split ->
    221                 val function = split.dropLast(1).fold(moduleObject) { obj, key ->
    222                     obj.get(key, obj) as? ScriptableObject ?: return@contextScope Unit
    223                 }.get(split.last(), moduleObject) as? Function ?: return@contextScope Unit
    224 
    225                 runCatching {
    226                     function.call(this, moduleObject, moduleObject, args)
    227                 }.onFailure {
    228                     scriptRuntime.logger.error("Error while calling function $name", it)
    229                 }
    230             }
    231         }
    232     }

    234     fun registerBindings(vararg bindings: AbstractBinding) {
    235         bindings.forEach {
    236             moduleBindings[it.name] = it.apply {
    237                 context = moduleBindingContext
    238             }
    239         }
    240     }

    242     fun onBridgeConnected(reloaded: Boolean = false) {
    243         if (reloaded) {
    244             moduleBindings.values.forEach { binding ->
    245                 runCatching {
    246                     binding.onBridgeReloaded()
    247                 }.onFailure {
    248                     scriptRuntime.logger.error("Failed to call onBridgeConnected for binding ${binding.name}", it)
    249                 }
    250             }
    251         }
    252 
    253         callFunction("module.onBridgeConnected", reloaded)
    254     }

    256     @Suppress("UNCHECKED_CAST")
    257     fun <T : Any> getBinding(clazz: KClass<T>): T? {
    258         return moduleBindings.values.find { clazz.isInstance(it) } as? T
    259     }

    261     private fun argsToString(args: Array<out Any?>?): String {
    262         return args?.joinToString(" ") {
    263             when (it) {
    264                 is Wrapper -> it.unwrap().let { value ->
    265                     if (value is Throwable) value.message + "\n" + value.stackTraceToString()
    266                     else value.toString()
    267                 }
    268                 else -> it.toString()
    269             }
    270         } ?: "null"
    271     }
