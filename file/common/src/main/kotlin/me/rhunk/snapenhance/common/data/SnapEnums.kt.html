     97         fun fromMessageContainer(protoReader: ProtoReader?): ContentType? {
     98             if (protoReader == null) return null
     99             return protoReader.run {
    100                 when {
    101                     contains(8) -> STATUS
    102                     contains(2) -> CHAT
    103                     contains(11) -> SNAP
    104                     contains(6) -> NOTE
    105                     contains(3) -> EXTERNAL_MEDIA
    106                     contains(4) -> STICKER
    107                     contains(5) -> SHARE
    108                     contains(7) -> EXTERNAL_MEDIA // story replies
    109                     contains(20) -> MAP_REACTION
    110                     else -> null
    111                 }
    112             }
    113         }
    114     }
