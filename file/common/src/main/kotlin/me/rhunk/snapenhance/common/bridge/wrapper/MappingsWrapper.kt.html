     47         )

     53     private fun loadCached() {
     54         if (!exists()) {
     55             throw Exception("Mappings file does not exist")
     56         }
     57         val mappingsObject = JsonParser.parseString(readBytes().toString(Charsets.UTF_8)).asJsonObject.also {
     58             mappingUniqueHash = it["unique_hash"].asLong
     59         }
     60 
     61         mappingsObject.entrySet().forEach { (key, value) ->
     62             mappers.values.firstOrNull { it.mapperName == key }?.let { mapper ->
     63                 mapper.readFromJson(value.asJsonObject)
     64                 mapper.classLoader = context.classLoader
     65             }
     66         }
     67         isMappingsLoaded = true
     68     }
     69 
     70     fun refresh() {
     71         mappingUniqueHash = getUniqueBuildId()
     72 
     73         // reset native signature cache
     74         fileHandleManager.value.getFileHandle(FileHandleScope.INTERNAL.key, InternalFileHandleType.NATIVE_SIG_CACHE.key).delete()
     75 
     76         val classMapper = ClassMapper(*mappers.values.toTypedArray())
     77 
     78         runCatching {
     79             classMapper.loadApk(getSnapchatPackageInfo()?.applicationInfo?.sourceDir ?: throw Exception("Failed to get APK"))
     80         }.onFailure {
     81             throw Exception("Failed to load APK", it)
     82         }
     83 
     84         runBlocking {
     85             val result = classMapper.run().apply {
     86                 addProperty("unique_hash", mappingUniqueHash)
     87             }
     88             writeBytes(result.toString().toByteArray())
     89         }
     90     }
