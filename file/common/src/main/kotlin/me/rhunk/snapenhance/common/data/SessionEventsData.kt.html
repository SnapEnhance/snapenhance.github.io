     77 @Parcelize
     78 class TrackerEventsResult(
     79     val rules: Map<ScopedTrackerRule, List<TrackerRuleEvent>>,
     80 ): Parcelable {
     81     fun getActions(): Map<TrackerRuleAction, TrackerRuleActionParams> {
     82         return rules.flatMap {
     83             it.value
     84         }.fold(mutableMapOf()) { acc, ruleEvent ->
     85             ruleEvent.actions.forEach { action ->
     86                 acc[action] = acc[action]?.merge(ruleEvent.params) ?: ruleEvent.params
     87             }
     88             acc
     89         }
     90     }
     91 
     92     fun canTrackOn(conversationId: String?, userId: String?): Boolean {
     93         return rules.any { (scopedRule, events) ->
     94             if (!events.any { it.enabled }) return@any false
     95             val scopes = scopedRule.scopes
     96 
     97             when (scopes[userId]) {
     98                 TrackerScopeType.WHITELIST -> return@any true
     99                 TrackerScopeType.BLACKLIST -> return@any false
    100                 else -> {}
    101             }
    102 
    103             when (scopes[conversationId]) {
    104                 TrackerScopeType.WHITELIST -> return@any true
    105                 TrackerScopeType.BLACKLIST -> return@any false
    106                 else -> {}
    107             }
    108 
    109             return@any scopes.isEmpty() || scopes.any { it.value == TrackerScopeType.BLACKLIST }
    110         }
    111     }
    112 }
