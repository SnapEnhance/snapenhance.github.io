    166     override fun addMessage(bridgeLoggedMessage: BridgeLoggedMessage) {
    167         val hasMessage = database.rawQuery("SELECT message_id FROM messages WHERE conversation_id = ? AND message_id = ?", arrayOf(bridgeLoggedMessage.conversationId, bridgeLoggedMessage.messageId.toString())).use {
    168             it.moveToFirst()
    169             it.count > 0
    170         }
    171 
    172         if (!hasMessage) {
    173             runBlocking(coroutineScope.coroutineContext) {
    174                 database.insert("messages", null, ContentValues().apply {
    175                     put("message_id", bridgeLoggedMessage.messageId)
    176                     put("conversation_id", bridgeLoggedMessage.conversationId)
    177                     put("user_id", bridgeLoggedMessage.userId)
    178                     put("username", bridgeLoggedMessage.username)
    179                     put("send_timestamp", bridgeLoggedMessage.sendTimestamp)
    180                     put("added_timestamp", System.currentTimeMillis())
    181                     put("group_title", bridgeLoggedMessage.groupTitle)
    182                     put("message_data", bridgeLoggedMessage.messageData)
    183                 })
    184             }
    185         }
    186 
    187         // handle message edits
    188         runBlocking(coroutineScope.coroutineContext) {
    189             runCatching {
    190                 val messageObject = gson.fromJson(
    191                     bridgeLoggedMessage.messageData.toString(Charsets.UTF_8),
    192                     JsonObject::class.java
    193                 )
    194                 if (messageObject.getAsJsonObject("mMessageContent")
    195                         ?.getAsJsonPrimitive("mContentType")?.asString != "CHAT"
    196                 ) return@runBlocking
    197 
    198                 val metadata = messageObject.getAsJsonObject("mMetadata")
    199                 if (metadata.get("mIsEdited")?.asBoolean != true) return@runBlocking
    200 
    201                 val messageTextContent =
    202                     messageObject.getAsJsonObject("mMessageContent")?.getAsJsonArray("mContent")
    203                         ?.map { it.asByte }?.toByteArray()?.let {
    204                             ProtoReader(it).getString(2, 1)
    205                         } ?: return@runBlocking
    206 
    207                 database.rawQuery(
    208                     "SELECT MAX(edit_number), message_text FROM chat_edits WHERE conversation_id = ? AND message_id = ?",
    209                     arrayOf(bridgeLoggedMessage.conversationId, bridgeLoggedMessage.messageId.toString())
    210                 ).use {
    211                     it.moveToFirst()
    212                     val editNumber = it.getInt(0)
    213                     val lastEditedMessage = it.getString(1)
    214 
    215                     if (lastEditedMessage == messageTextContent) return@runBlocking
    216 
    217                     database.insert("chat_edits", null, ContentValues().apply {
    218                         put("edit_number", editNumber + 1)
    219                         put("added_timestamp", System.currentTimeMillis())
    220                         put("conversation_id", bridgeLoggedMessage.conversationId)
    221                         put("message_id", bridgeLoggedMessage.messageId)
    222                         put("message_text", messageTextContent)
    223                     })
    224                 }
    225             }.onFailure {
    226                 AbstractLogger.directDebug("Failed to handle message edit: ${it.message}")
    227             }
    228         }
    229     }

    420     override fun getChatEdits(conversationId: String, messageId: Long): List<LoggedChatEdit> {
    421         val edits = mutableListOf<LoggedChatEdit>()
    422         database.rawQuery(
    423             "SELECT added_timestamp, message_text FROM chat_edits WHERE conversation_id = ? AND message_id = ? ORDER BY added_timestamp ASC",
    424             arrayOf(conversationId, messageId.toString())
    425         ).use { cursor ->
    426             while (cursor.moveToNext()) {
    427                 edits.add(LoggedChatEdit().apply {
    428                     timestamp = cursor.getLongOrNull("added_timestamp") ?: return@apply
    429                     message = cursor.getStringOrNull("message_text")
    430                 }.takeIf { it.timestamp > 0L } ?: continue)
    431             }
    432         }
    433 
    434         if (edits.isNotEmpty()) {
    435             // append original message
    436             database.rawQuery("SELECT added_timestamp, message_data FROM messages WHERE conversation_id = ? AND message_id = ?", arrayOf(conversationId, messageId.toString())).use { cursor ->
    437                 if (!cursor.moveToFirst()) return@use
    438 
    439                 val originalMessage = cursor.getBlobOrNull("message_data") ?: return@use
    440                 val addedTimestamp = cursor.getLongOrNull("added_timestamp") ?: return@use
    441 
    442                 val messageObject = gson.fromJson(
    443                     originalMessage.toString(Charsets.UTF_8),
    444                     JsonObject::class.java
    445                 )
    446 
    447                 val messageTextContent =
    448                     messageObject.getAsJsonObject("mMessageContent")?.getAsJsonArray("mContent")
    449                         ?.map { it.asByte }?.toByteArray()?.let {
    450                             ProtoReader(it).getString(2, 1)
    451                         } ?: return@use
    452 
    453                 if (edits.firstOrNull()?.message != messageTextContent) {
    454                     edits.add(0, LoggedChatEdit().apply {
    455                         timestamp = addedTimestamp
    456                         message = messageTextContent
    457                     })
    458                 }
    459             }
    460         }
    461 
    462         return edits
    463     }

    465     fun fetchMessages(
    466         conversationId: String,
    467         fromTimestamp: Long,
    468         limit: Int,
    469         reverseOrder: Boolean = true,
    470         filter: ((LoggedMessage) -> Boolean)? = null
    471     ): List<LoggedMessage> {
    472         val messages = mutableListOf<LoggedMessage>()
    473         database.rawQuery(
    474             "SELECT * FROM messages WHERE conversation_id = ? AND send_timestamp ${if (reverseOrder) "<" else ">"} ? ORDER BY send_timestamp ${if (reverseOrder) "DESC" else "ASC"}",
    475             arrayOf(conversationId, fromTimestamp.toString())
    476         ).use {
    477             while (it.moveToNext() && messages.size < limit) {
    478                 val message = LoggedMessage(
    479                     messageId = it.getLongOrNull("message_id") ?: continue,
    480                     conversationId = it.getStringOrNull("conversation_id") ?: continue,
    481                     userId = it.getStringOrNull("user_id") ?: continue,
    482                     username = it.getStringOrNull("username") ?: continue,
    483                     sendTimestamp = it.getLongOrNull("send_timestamp") ?: continue,
    484                     addedTimestamp = it.getLongOrNull("added_timestamp") ?: continue,
    485                     groupTitle = it.getStringOrNull("group_title"),
    486                     messageData = it.getBlobOrNull("message_data") ?: continue
    487                 )
    488                 if (filter != null && !filter(message)) continue
    489                 messages.add(message)
    490             }
    491         }
    492         return messages
